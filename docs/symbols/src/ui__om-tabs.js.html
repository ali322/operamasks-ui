904 	    /**
905 	     * 选中特定的页签
906 	     * n 可为页签的索引（从0开始计数），或者页签的tabId TODO n 需要支持first  和 last 表示选中第一个和最后一个
907 	     */
908      	_activate : function(n) {
909 	        var $self = this.element,
910 	            options = this.options,
911 	            items = this.items,
912 	            url;
913 	        var $ul = $self.find('>div.om-tabs-headers ul');
914 	        if ( options.activeTabId === n || options.active === n ) {
915 	               return false;
916 	        }
917 	        n = n || 0;
918 	        var $anchor , tid = n;
919 	        if ( n == 'next' ) {
920 	            n = (options.active + 1) % items.length ;
921 	        } else if ( n == 'prev' ) {
922 	            n = (options.active - 1) % items.length ;
923 	        } 
924 	        if (typeof n == 'number') {
925 	            tid = this._getAlter(n);
926 	        } else if (typeof n == 'string') {
927 	            n = this._getAlter(n);
928 	        }
929 	        if (options.onBeforeActivate && this._trigger("onBeforeActivate",null,n) === false) {
930 	            return false;
931 	        }
932 	        $anchor = $ul.find('li a[tabId=' + tid + ']');
933 	        $anchor.parent().addClass(activatedCls)
934 	            .siblings().removeClass(activatedCls);
935 	        options.activeTabId = tid;
936 	        options.active = n;
937 	        var i = items.length;
938 	        // 保证切换面板时先显示后隐藏，防止页面抖动的现象
939 	        for(i=items.length;i--;i>=0){
940 	        	var $target = items[i];
941 	        	if ($target.omPanel('option' , 'tabId')== tid) {
942 	        		$target.omPanel('open');
943 	        		if(url=this._getUnloadedUrl(tid)){
944 	        			$target.omPanel("reload" , url);
945 	        			this._removeLoadInfo(tid);
946 	        		}
947 	        	}
948 	        }
949 	        for(i=items.length;i--;i>=0){
950 	        	var $target = items[i];
951 	        	if ($target.omPanel('option' , 'tabId') != tid) {
952 	        		$target.omPanel('close');
953 	        	}
954 	        }
955 	        //当选中了一个并未完全显示的页签,需要滚动让他完全显示出来
956 	        if (this._checkScroller()) {
957 	            //stop every animation.
958 	            $ul.stop(true, true);
959 	            $self.clearQueue();
960 	            var $lScroller = $ul.prev();
961 	            var $rScroller = $ul.next();
962 	            var lBorder = $anchor.parent().offset().left;
963 	            var rBorder = lBorder + $anchor.parent().outerWidth(true);
964 	            var lDiff = $lScroller.offset().left + $lScroller.outerWidth(true) + 4 - lBorder ;
965 	            var rDiff = $rScroller.offset().left - rBorder ;
966 	            if (lDiff >= 0) {
967 	                this._scroll(lDiff, this._scrollCbFn());
968 	            } else if (rDiff <= 0) {
969 	                this._scroll(rDiff, this._scrollCbFn());
970 	            } else {
971 	                this._scrollCbFn()();
972 	            }
973 	        }
974 	        var his = this.history,
975 	        	index = $.inArray(tid, his);
976 	        index === -1 ? his.push(tid) : his.push(his.splice(index , 1)[0]);
977 	        options.onActivate && this._trigger("onActivate",null,n);
978     	},
