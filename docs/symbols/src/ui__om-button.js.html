 14 ;(function($) {
 15 	/**
 16      * @name omButton
 17      * @class 按钮组件。类似于html中的button、input[type=submit]、input[type=button]，使用背景图片实现圆角，支持icon，可以只显示icon。<br/><br/>
 18      * <b>特点：</b><br/>
 19      * <ol>
 20      *      <li>实现圆角</li>
 21      *      <li>支持左icon和右icon，可同时出现左右icon，也可以只显示icon不显示label</li>
 22      *      <li>按钮文字数目不限，也可以任意设置按钮宽度</li>
 23      *      <li>支持input[type=button]、input[type=submit]、input[type=reset]、button、a五种标签形式</li>
 24      * </ol>
 25      * 
 26      * <b>示例：</b><br/>
 27      * <pre>
 28      * <script type="text/javascript" >
 29      * $(document).ready(function() {
 30      *     $("#btn").omButton({
 31      *         icons : {left:"images/help.png",right:"images/edit_add.png"},
 32      *         width : 150,
 33      *         disabled : "disabled",
 34      *         onClick : function(event){
 35      *             // do something
 36      *         }
 37      *     });
 38      * });
 39      * </script>
 40      * 
 41      * <input id="btn" type="submit" />
 42      * </pre>
 43      * 
 44      * @constructor
 45      * @description 构造函数. 
 46      * @param p 标准config对象：{}
 47      */
 48     $.omWidget("om.omButton", {
 49     	
 50         options: /**@lends omButton# */{
 51         	/**
 52         	 * 是否禁用组件。可通过("#id").attr("disabled")判断按钮是否禁用。
 53              * @type String 
 54              * @default null 
 55              * @example
 56              * $("#button").omButton({disabled: true}); 
 57              */
 58             disabled : null ,
 59             /**
 60              * 显示文本。label值可以写到dom元素里面，也可以设置在属性里，设置在属性里优先级最高。
 61              * @type String 
 62              * @default ""
 63              * @example
 64              * $("#button").omButton({label: "apusic"});
 65              */
 66             label : null ,
 67             /**
 68         	 * 按钮图标。left 左图标，right 右图标，取值为图片路径。
 69              * @type Object 
 70              * @default null 
 71              * @example
 72              * $("#button").omButton({
 73              *     icons: {
 74              *         left: "images/help.png",
 75              *         right: "images/edit_add.png"
 76              *     }
 77              * });
 78              */
 79             icons: {
 80     			left: null,
 81     			right: null
 82     		},
 83     		/**
 84         	 * 按钮宽度。设置固定宽度后按钮不会随文字多少而改变。
 85              * @type Number 
 86              * @default null 
 87              * @example
 88              * width : 150
 89              */
 90             width : null ,
 91             /**
 92         	 * 点击按钮时触发的事件。
 93              * @event
 94              * @param event jQuery.Event对象。
 95              * @name omButton#onClick 
 96              * @example
 97              * onClick : function(event){
 98              *    //do something
 99              * }
100              */
101             onClick : null
102         },
103         
104         _create : function() {
105             this._initElemType();
106             var wrapperSpan = $("<span>").addClass("om-btn om-state-default").css("border", "none"),
107             leftSpan = $("<span>").addClass("om-btn-bg om-btn-left"),
108             centerSpan = $("<span>").addClass("om-btn-bg om-btn-center"),
109             rightSpan = $("<span>").addClass("om-btn-bg om-btn-right");
110             // TODO: 删除对apuaic具体类型的依赖
111             if(this.element.hasClass("apusic-btn-deepblue")){
112             	wrapperSpan.addClass("apusic-btn-deepblue");
113             }
114             
115             this.element.addClass( "om-btn-txt" )
116                 .css( {"background-position":"left center","background-repeat":"no-repeat"} )
117                 .wrap( wrapperSpan )
118                 .before( leftSpan )
119                 .after( rightSpan )
120                 .wrap( centerSpan );
121         },
122         
123         _init : function(){
124             var self = this,
125                 options = this.options,
126                 element = this.element;
127             if ( typeof options.disabled != "boolean" ) {
128                 options.disabled = element.propAttr( "disabled" );
129     		}
130             if ( element.attr("disabled") == "disabled" || options.disabled == "disabled") {
131     			options.disabled = true;
132     		}
133             this._initButton();
134             if(options.disabled){
135             	self._addClass("disabled");
136             	element.css("cursor","default");
137             }
138             var $newelement = element.parent().parent();
139             $newelement.bind("mouseenter", function( event ){
140             					if ( options.disabled ) {
141             						return false;
142             					}
143             					self._addClass("hover");
144             				}).bind( "mouseleave", function( event ) {
145             					if ( options.disabled ) {
146             						return false;
147             					}
148             					self._removeClass("hover");
149             					self._removeClass("active");
150             				}).bind( "click", function( event ){
151             					if ( options.disabled ) {
152             						event.preventDefault();
153             						event.stopImmediatePropagation();
154             						return false;
155             					}else if(self.options.onClick){
156             						self._trigger("onClick",event);
157             					}
158             				}).bind( "mousedown", function( event ) {
159             					if ( options.disabled ) {
160             						return false;
161             					}
162             					self._addClass("active");
163             				    self._removeClass("focus");
164             				})
165             				.bind( "mouseup", function( event ) {
166             					if ( options.disabled ) {
167             						return false;
168             					}
169             					self._removeClass("focus");
170             					self._removeClass("active");
171             				})
172             				.bind( "keydown", function(event) {
173             					if ( options.disabled ) {
174             						return false;
175             					}
176             					if ( event.keyCode == $.om.keyCode.SPACE || event.keyCode == $.om.keyCode.ENTER ) {
177             						self._addClass("active");
178             					}
179             					if( event.keyCode == $.om.keyCode.SPACE){
180             						var onClick = options.onClick;
181             		                if ( onClick && self._trigger("onClick",event) === false ) {
182             		                    return;
183             		                }
184             					}
185             				})
186             				.bind("keyup", function() {
187 								self._removeClass("active");
188 							});
189 	            element.bind( "focus.button", function( event ) {
190 								if ( options.disabled ) {
191 									return false;
192 								}
193 								self._addClass("focus");
194 							}).bind( "blur.button", function( event ) {
195 	        					if ( options.disabled ) {
196 	        						return false;
197 	        					}
198 	        					self._removeClass("focus");
199 	        				});
200 	            element.next().bind("click", function(event){
201 	            	if(element.is("input")&&element.attr("type")=="submit"){	
202 	            		element.submit();
203 	            	}
204 	            });
205         },
206         /**
207          * 启用组件。
208          * @name omButton#enable
209          * @function
210          * @example
211          * $("#btn").omButton("enable");
212          */
213         enable : function(){
214             this._removeClass("disabled");
215             this.options.disabled = false;
216             this.element.css("cursor","pointer")
217                         .removeAttr("disabled");
218         },
219         /**
220          * 禁用组件。
221          * @name omButton#disable
222          * @function
223          * @example
224          * $("#btn").omButton("disable");
225          */
226         disable : function(){
227         	this._addClass("disabled");
228             this.options.disabled = true;
229             this.element.css("cursor","default");
230             if(this.type == "input" || this.type == "button"){
231             	this.element.attr("disabled", "disabled");
232             }
233         },
234         /**
235          * 触发点击事件。
236          * @name omButton#click
237          * @function
238          * @example
239          * $("#btn").omButton("click");
240          */
241         click : function(){
242         	if(!this.options.disabled && this.options.onClick){
243         		this._trigger("onClick");
244             }
245         },
246         /**
247          * 改变按钮的label属性。
248          * @name omButton#changeLabel
249          * @function
250          * @param label 按钮文本
251          * @example
252          * $("#btn").omButton("changeLabel","按钮label");
253          */
254         changeLabel : function(label){
255             if(this.type == "a"){
256             	this.element.text(label) ;
257             }else if( this.type == "input" ){
258             	this.element.val(label) ;
259             }else if ( this.type == "button" ){
260             	this.element.html(label) ;
261             }
262         },
263         /**
264          * 改变按钮的icon属性。
265          * @name omButton#changeIcons
266          * @function
267          * @param icons 图标路径
268          * @example
269          * $("#btn").omButton("changeIcons",{
270          *     left: "images/help.png",
271          *     right: "images/edit_add.png"
272          * });
273          */
274         changeIcons : function( icons ){
275         	if( !this.options.disabled ){
276 	            if( icons ){
277 	            	icons.left?this.element.css( "backgroundImage","url( "+icons.left+" )" ):null;
278 	            	icons.right?this.element.next().attr( "src",icons.right ):null;
279 	            }
280             }
281         },
282         destroy : function(){
283         	$el = this.element;
284         	$el.closest(".om-btn").after($el).remove();
285         },
286         _addClass : function( state ){
287         	this.element.parent().parent().addClass( "om-state-"+state );
288         },
289         _removeClass : function( state ){
290         	this.element.parent().parent().removeClass( "om-state-"+state );
291         },
292         _initButton : function(){
293         	var options = this.options,
294         	    label = this._getLabel(),
295         	    element = this.element;
296         	
297             element.removeClass("om-btn-icon om-btn-only-icon")
298                 .next("img").remove();
299         	
300         	if( options.width > 10 ){
301         		element.css( "width",parseInt( options.width )-10 );
302         	}
303         	if( this.type == "a" || this.type == "button" ){
304         	    element.html( label );
305         	}else{
306         	    element.val( label );
307         	}
308         	
309         	if( options.icons.left ){
310         	    if( label ){
311         	        element.addClass( "om-btn-icon" ).css( "background-image","url("+options.icons.left+")" );
312         	    }else{
313         	        element.addClass( "om-btn-only-icon" ).css("background-image","url("+options.icons.left+")" );
314         	    }
315         	}
316         	if( options.icons.right ){
317         	    if( label != "" ){
318         	        $( "<img>" ).attr( "src",options.icons.right ).css( {"vertical-align":"baseline","padding-left":"3px"} ).insertAfter( element );
319         	    }else{
320         	        $( "<img>" ).attr( "src",options.icons.right ).css( "vertical-align","baseline" ).insertAfter( element );
321         	    }
322             }
323         },
324         _getLabel : function(){
325         	return this.options.label || this.element.html() || this.element.text() || this.element.val();
326         },
327         _initElemType: function() {
328     		if ( this.element.is("input") ) {
329     			this.type = "input";
330     		}  else if ( this.element.is("a") ) {
331     			this.type = "a";
332     		} else if ( this.element.is("button") ){
333     			this.type = "button";
334     		}
335     	}
336     });
337 })(jQuery);
