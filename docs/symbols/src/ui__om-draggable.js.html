205 	_mouseCapture: function(event) {
206 
207 		var o = this.options;
208 
209 		// among others, prevent a drag on a resizable-handle
210 		if (this.helper || o.disabled || $(event.target).is('.om-resizable-handle'))

221 	_mouseStart: function(event) {
222 
223 		var o = this.options;
224 
225 		//Create and append the visible helper
226 		this.helper = this._createHelper(event);
227 
228 		//Cache the helper size
229 		this._cacheHelperProportions();
230 
231 		//If ddmanager is used for droppables, set the global draggable
232 		if($.om.ddmanager)
233 			$.om.ddmanager.current = this;
234 
235 		/*
236 		 * - Position generation -
237 		 * This block generates everything position related - it's the core of draggables.
238 		 */
239 
240 		//Cache the margins of the original element
241 		this._cacheMargins();
242 
243 		//Store the helper's css position
244 		this.cssPosition = this.helper.css("position");
245 		this.scrollParent = this.helper.scrollParent();
246 
247 		//The element's absolute position on the page minus margins
248 		this.offset = this.positionAbs = this.element.offset();
249 		this.offset = {
250 			top: this.offset.top - this.margins.top,
251 			left: this.offset.left - this.margins.left
252 		};
253 
254 		$.extend(this.offset, {
255 			click: { //Where the click happened, relative to the element
256 				left: event.pageX - this.offset.left,
257 				top: event.pageY - this.offset.top
258 			},
259 			parent: this._getParentOffset(),
260 			relative: this._getRelativeOffset() //This is a relative to absolute position minus the actual position calculation - only used for relative positioned helper
261 		});
262 
263 		//Generate the original position
264 		this.originalPosition = this.position = this._generatePosition(event);
265 		this.originalPageX = event.pageX;
266 		this.originalPageY = event.pageY;
267 
268 		//Set a containment if given in the options
269 		if(o.containment)
270 			this._setContainment();
271 
272 		//Trigger event + callbacks
273 		if(this._trigger("onStart", event) === false) {
274 			this._clear();
275 			return false;
276 		}
277 
278 		//Recache the helper size
279 		this._cacheHelperProportions();
280 
281 		//Prepare the droppable offsets
282 		if ($.om.ddmanager && !o.dropBehaviour)
283 			$.om.ddmanager.prepareOffsets(this, event);
284 
285 		this.helper.addClass("om-draggable-dragging");
286 		this._mouseDrag(event, true); //Execute the drag once - this causes the helper not to be visible before getting its correct position
287 		
288 		//If the ddmanager is used for droppables, inform the manager that dragging has started (see #5003)
289 		if ( $.om.ddmanager ) $.om.ddmanager.dragStart(this, event);
290 		
291 		return true;
292 	},

294 	_mouseDrag: function(event, noPropagation) {
295 
296 		//Compute the helpers position
297 		this.position = this._generatePosition(event);
298 		this.positionAbs = this._convertPositionTo("absolute");
299 
300 		//Call plugins and callbacks and use the resulting position if something is returned
301 		if (!noPropagation) {
302 			var ui = this._uiHash();
303 			if(this._trigger('onDrag', event, ui) === false) {
304 				this._mouseUp({});
305 				return false;
306 			}
307 			this.position = ui.position;
308 		}
309 
310 		if(!this.options.axis || this.options.axis != "y") this.helper[0].style.left = this.position.left+'px';
311 		if(!this.options.axis || this.options.axis != "x") this.helper[0].style.top = this.position.top+'px';
312 		if($.om.ddmanager) $.om.ddmanager.drag(this, event);
313 
314 		return false;
315 	},

317 	_mouseStop: function(event) {
318 
319 		//If we are using droppables, inform the manager about the drop
320 		var dropped = false;
321 		if ($.om.ddmanager && !this.options.dropBehaviour)
322 			dropped = $.om.ddmanager.drop(this, event);
323 
324 		//if a drop comes from outside (a sortable)
325 		if(this.dropped) {
326 			dropped = this.dropped;
327 			this.dropped = false;
328 		}
329 		
330 		//if the original element is removed, don't bother to continue if helper is set to "original"
331 		if((!this.element[0] || !this.element[0].parentNode) && this.options.helper == "original")
332 			return false;
333 
334 		if((this.options.revert == "invalid" && !dropped) || (this.options.revert == "valid" && dropped) || this.options.revert === true || ($.isFunction(this.options.revert) && this.options.revert.call(this.element, dropped))) {
335 			var self = this;
336 			$(this.helper).animate(this.originalPosition, 500, function() {
337 				if(self._trigger("onStop", event) !== false) {
338 					self._clear();
339 				}
340 			});
341 		} else {
342 			if(this._trigger("onStop", event) !== false) {
343 				this._clear();
344 			}
345 		}
346 
347 		return false;
348 	},

383 	_createHelper: function(event) {
384 
385 		var o = this.options;
386 		var helper = $.isFunction(o.helper) ? $(o.helper.apply(this.element[0], [event])) : (o.helper == 'clone' ? this.element.clone().removeAttr('id') : this.element);

388 		if(!helper.parents('body').length)
389 			helper.appendTo( this.element[0].parentNode);
390 
391 		if(helper[0] != this.element[0] && !(/(fixed|absolute)/).test(helper.css("position")))
392 			helper.css("position", "absolute");
393 
394 		return helper;

400 	_getParentOffset: function() {
401 
402 		//Get the offsetParent and cache its position
403 		this.offsetParent = this.helper.offsetParent();
404 		var po = this.offsetParent.offset();
405 
406 		// This is a special case where we need to modify a offset calculated on start, since the following happened:
407 		// 1. The position of the helper is absolute, so it's position is calculated based on the next positioned parent
408 		// 2. The actual offset parent is a child of the scroll parent, and the scroll parent isn't the document, which means that
409 		//    the scroll is included in the initial calculation of the offset of the parent, and never recalculated upon drag
410 		if(this.cssPosition == 'absolute' && this.scrollParent[0] != document && $.contains(this.scrollParent[0], this.offsetParent[0])) {
411 			po.left += this.scrollParent.scrollLeft();
412 			po.top += this.scrollParent.scrollTop();
413 		}
414 
415 		if((this.offsetParent[0] == document.body) //This needs to be actually done for all browsers, since pageX/pageY includes this information
416 		|| (this.offsetParent[0].tagName && this.offsetParent[0].tagName.toLowerCase() == 'html' && $.browser.msie)) //Ugly IE fix
417 			po = { top: 0, left: 0 };
418 
419 		return {
420 			top: po.top + (parseInt(this.offsetParent.css("borderTopWidth"),10) || 0),
421 			left: po.left + (parseInt(this.offsetParent.css("borderLeftWidth"),10) || 0)
422 		};
423 
424 	},
425 
426 	_getRelativeOffset: function() {
427 
428 		if(this.cssPosition == "relative") {
429 			var p = this.element.position();
430 			return {
431 				top: p.top - (parseInt(this.helper.css("top"),10) || 0) + this.scrollParent.scrollTop(),
432 				left: p.left - (parseInt(this.helper.css("left"),10) || 0) + this.scrollParent.scrollLeft()
433 			};
434 		} else {
435 			return { top: 0, left: 0 };
436 		}
437 
438 	},

456 	_setContainment: function() {
457 
458 		var o = this.options;
459 		if(o.containment == 'parent') o.containment = this.helper[0].parentNode;
460 		if(o.containment == 'document' || o.containment == 'window') this.containment = [
461 			o.containment == 'document' ? 0 : $(window).scrollLeft() - this.offset.relative.left - this.offset.parent.left,
462 			o.containment == 'document' ? 0 : $(window).scrollTop() - this.offset.relative.top - this.offset.parent.top,
463 			(o.containment == 'document' ? 0 : $(window).scrollLeft()) + $(o.containment == 'document' ? document : window).width() - this.helperProportions.width - this.margins.left,
464 			(o.containment == 'document' ? 0 : $(window).scrollTop()) + ($(o.containment == 'document' ? document : window).height() || document.body.parentNode.scrollHeight) - this.helperProportions.height - this.margins.top
465 		];
466 
467 		if(!(/^(document|window|parent)$/).test(o.containment) && o.containment.constructor != Array) {
468 		        var c = $(o.containment);
469 			var ce = c[0]; if(!ce) return;
470 			var co = c.offset();
471 			var over = ($(ce).css("overflow") != 'hidden');
472 
473 			this.containment = [
474 				(parseInt($(ce).css("borderLeftWidth"),10) || 0) + (parseInt($(ce).css("paddingLeft"),10) || 0),
475 				(parseInt($(ce).css("borderTopWidth"),10) || 0) + (parseInt($(ce).css("paddingTop"),10) || 0),
476 				(over ? Math.max(ce.scrollWidth,ce.offsetWidth) : ce.offsetWidth) - (parseInt($(ce).css("borderLeftWidth"),10) || 0) - (parseInt($(ce).css("paddingRight"),10) || 0) - this.helperProportions.width - this.margins.left - this.margins.right,
477 				(over ? Math.max(ce.scrollHeight,ce.offsetHeight) : ce.offsetHeight) - (parseInt($(ce).css("borderTopWidth"),10) || 0) - (parseInt($(ce).css("paddingBottom"),10) || 0) - this.helperProportions.height - this.margins.top  - this.margins.bottom
478 			];
479 			this.relative_container = c;
480 
481 		} else if(o.containment.constructor == Array) {
482 			this.containment = o.containment;
483 		}
484

487 	_convertPositionTo: function(d, pos) {
488 
489 		if(!pos) pos = this.position;
490 		var mod = d == "absolute" ? 1 : -1;
491 		var o = this.options, scroll = this.cssPosition == 'absolute' && !(this.scrollParent[0] != document && $.contains(this.scrollParent[0], this.offsetParent[0])) ? this.offsetParent : this.scrollParent, scrollIsRootNode = (/(html|body)/i).test(scroll[0].tagName);
492 
493 		return {
494 			top: (
495 				pos.top																	// The absolute mouse position
496 				+ this.offset.relative.top * mod										// Only for relative positioned nodes: Relative offset from element to offset parent
497 				+ this.offset.parent.top * mod											// The offsetParent's offset without borders (offset + border)
498 				- ($.browser.safari && $.browser.version < 526 && this.cssPosition == 'fixed' ? 0 : ( this.cssPosition == 'fixed' ? -this.scrollParent.scrollTop() : ( scrollIsRootNode ? 0 : scroll.scrollTop() ) ) * mod)
499 			),
500 			left: (
501 				pos.left																// The absolute mouse position
502 				+ this.offset.relative.left * mod										// Only for relative positioned nodes: Relative offset from element to offset parent
503 				+ this.offset.parent.left * mod											// The offsetParent's offset without borders (offset + border)
504 				- ($.browser.safari && $.browser.version < 526 && this.cssPosition == 'fixed' ? 0 : ( this.cssPosition == 'fixed' ? -this.scrollParent.scrollLeft() : scrollIsRootNode ? 0 : scroll.scrollLeft() ) * mod)
505 			)
506 		};
507 
508 	},

510 	_generatePosition: function(event) {
511 
512 		var o = this.options, scroll = this.cssPosition == 'absolute' && !(this.scrollParent[0] != document && $.contains(this.scrollParent[0], this.offsetParent[0])) ? this.offsetParent : this.scrollParent, scrollIsRootNode = (/(html|body)/i).test(scroll[0].tagName);
513 		var pageX = event.pageX;
514 		var pageY = event.pageY;
515 
516 		/*
517 		 * - Position constraining -
518 		 * Constrain the position to a mix of grid, containment.
519 		 */
520 
521 		if(this.originalPosition) { //If we are not dragging yet, we won't check for options
522 		         var containment;
523 		         if(this.containment) {
524 				 if (this.relative_container){
525 				     var co = this.relative_container.offset();
526 				     containment = [ this.containment[0] + co.left,
527 						     this.containment[1] + co.top,
528 						     this.containment[2] + co.left,
529 						     this.containment[3] + co.top ];
530 				 }
531 				 else {
532 				     containment = this.containment;
533 				 }
534 
535 				if(event.pageX - this.offset.click.left < containment[0]) pageX = containment[0] + this.offset.click.left;
536 				if(event.pageY - this.offset.click.top < containment[1]) pageY = containment[1] + this.offset.click.top;
537 				if(event.pageX - this.offset.click.left > containment[2]) pageX = containment[2] + this.offset.click.left;
538 				if(event.pageY - this.offset.click.top > containment[3]) pageY = containment[3] + this.offset.click.top;
539 			}
540 
541 		}
542 
543 		return {
544 			top: (
545 				pageY																// The absolute mouse position
546 				- this.offset.click.top													// Click offset (relative to the element)
547 				- this.offset.relative.top												// Only for relative positioned nodes: Relative offset from element to offset parent
548 				- this.offset.parent.top												// The offsetParent's offset without borders (offset + border)
549 				+ ($.browser.safari && $.browser.version < 526 && this.cssPosition == 'fixed' ? 0 : ( this.cssPosition == 'fixed' ? -this.scrollParent.scrollTop() : ( scrollIsRootNode ? 0 : scroll.scrollTop() ) ))
550 			),
551 			left: (
552 				pageX																// The absolute mouse position

617

619 				if((i.overflowOffset.top + i.scrollParent[0].offsetHeight) - event.pageY < scrollSensitivity)
620 					i.scrollParent[0].scrollTop = scrolled = i.scrollParent[0].scrollTop + scrollSpeed;
621 				else if(event.pageY - i.overflowOffset.top < scrollSensitivity)
622 					i.scrollParent[0].scrollTop = scrolled = i.scrollParent[0].scrollTop - scrollSpeed;
623 			}

627 					i.scrollParent[0].scrollLeft = scrolled = i.scrollParent[0].scrollLeft + scrollSpeed;

629 					i.scrollParent[0].scrollLeft = scrolled = i.scrollParent[0].scrollLeft - scrollSpeed;
630 			}
631 
632 		} else {
633 
634 			if(!o.axis || o.axis != 'x') {
635 				if(event.pageY - $(document).scrollTop() < scrollSensitivity)
636 					scrolled = $(document).scrollTop($(document).scrollTop() - scrollSpeed);
637 				else if($(window).height() - (event.pageY - $(document).scrollTop()) < scrollSensitivity)
638 					scrolled = $(document).scrollTop($(document).scrollTop() + scrollSpeed);
639 			}
640 
641 			if(!o.axis || o.axis != 'y') {
642 				if(event.pageX - $(document).scrollLeft() < scrollSensitivity)
643 					scrolled = $(document).scrollLeft($(document).scrollLeft() - scrollSpeed);
644 				else if($(window).width() - (event.pageX - $(document).scrollLeft()) < scrollSensitivity)
645 					scrolled = $(document).scrollLeft($(document).scrollLeft() + scrollSpeed);
646 			}
647 
648 		}
