157 function focusable( element, isTabIndexNotNaN ) {
158 	var nodeName = element.nodeName.toLowerCase();
159 	if ( "area" === nodeName ) {
160 		var map = element.parentNode,
161 			mapName = map.name,
162 			img;
163 		if ( !element.href || !mapName || map.nodeName.toLowerCase() !== "map" ) {
164 			return false;
165 		}
166 		img = $( "img[usemap=#" + mapName + "]" )[0];
167 		return !!img && visible( img );
168 	}
169 	return ( /input|select|textarea|button|object/.test( nodeName )
170 		? !element.disabled
171 		: "a" == nodeName
172 			? element.href || isTabIndexNotNaN
173 			: isTabIndexNotNaN)
174 		// the element and all of its ancestors must be visible
175 		&& visible( element );
176 }
177 function visible( element ) {
178 	return !$( element ).parents().andSelf().filter(function() {
179 		return $.curCSS( this, "visibility" ) === "hidden" ||
180 			$.expr.filters.hidden( this );
181 	}).length;
182 }

302 $.omWidget.bridge = function( name, object ) {
303 	$.fn[ name ] = function( options ) {
304 		// 如果tabs方法第一个参数是string类型，则认为是调用组件的方法，否则调用options方法  
305 		var isMethodCall = typeof options === "string",
306 			args = Array.prototype.slice.call( arguments, 1 ),
307 			returnValue = this;
308 		// allow multiple hashes to be passed on init
309 		options = !isMethodCall && args.length ?
310 			$.extend.apply( null, [ true, options ].concat(args) ) :
311 			options;
312 		// '_'开头的方法被认为是内部方法，不会被执行，如$('#tab1').tabs('_init')  
313 		if ( isMethodCall && options.charAt( 0 ) === "_" ) {
314 			return returnValue;
315 		}
316 		if ( isMethodCall ) {
317 			this.each(function() {
318 				// 执行组件方法  
319 				var instance = $.data( this, name );
320 				if (options == 'options') {
321 				    returnValue = instance && instance.options;
322 				    return false;
323                 } else {
324     				var	methodValue = instance && $.isFunction( instance[options] ) ?
325     						instance[ options ].apply( instance, args ) : instance;
326     				if ( methodValue !== instance && methodValue !== undefined ) {
327     					returnValue = methodValue;
328     					return false;
329     				}
330                 }
331 			});
332 		} else {
333 			// 调用组件的options方法  
334 			this.each(function() {
335 				var instance = $.data( this, name );
336 				if ( instance ) {
337 					// 设置options后再次调用_init方法，第一次调用是在_createWidget方法里面。这个方法需要开发者去实现。  
338                     // 主要是当改变组件中某些参数后可能需要对组件进行重画  
339                     instance._setOptions( options || {} );
340 				    $.extend(instance.options, options);
341 				    $(instance.beforeInitListeners).each(function(){
342 				        this.call(instance);
343 				    });
344 					instance._init();
345 					$(instance.initListeners).each(function(){
346 				        this.call(instance);
347 				    });
348 				} else {
349 					// 没有实例的话，在这里调用组件类的构造函数，并把构造后的示例保存在dom的data里面。注意这里的this是dom，object是模块类 
350 					$.data( this, name, new object( options, this ) );
351 				}
352 			});
353 		}
354 
355 		return returnValue;
356 	};
357 };

462 		var callback = this.options[ type ];
463 		// 封装js标准event对象为jquery的Event对象
464 		event = $.Event( event );

466 		// copy original event properties over to the new event
467 		// this would happen if we could call $.event.fix instead of $.Event
468 		// but we don't have a way to force an event to be fixed multiple times
469 		if ( event.originalEvent ) {
470 			for ( var i = $.event.props.length, prop; i; ) {
471 				prop = $.event.props[ --i ];
472 				event[ prop ] = event.originalEvent[ prop ];
473 			}

476 		var newArgs = [],
477 			argLength = arguments.length;
478 		for(var i = 2; i < argLength; i++){
479 			newArgs[i-2] = arguments[i];
480 		}
481 		if( argLength > 1){
482 			newArgs[argLength-2] = arguments[1];
483 		}
484 		return !( $.isFunction(callback) &&
485 			callback.apply( this.element, newArgs ) === false ||
486 			event.isDefaultPrevented() );
487 	}
