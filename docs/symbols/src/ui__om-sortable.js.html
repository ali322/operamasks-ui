 24 	_create: function() {
 25

 27 		this.element.addClass("ui-sortable");

 43 	_mouseCapture: function(event, overrideHandle) {
 44 
 45 		if (this.reverting) {
 46 			return false;
 47 		}
 48 
 49 		if(this.options.disabled || this.options.type == 'static') return false;
 50 
 51 		//We have to refresh the items data once first
 52 		this._refreshItems(event);
 53 
 54 		//Find out if the clicked node (or one of its parents) is a actual item in this.items
 55 		var currentItem = null, self = this, nodes = $(event.target).parents().each(function() {
 56 			if($.data(this, 'sortable-item') == self) {
 57 				currentItem = $(this);
 58 				return false;
 59 			}
 60 		});
 61 		if($.data(event.target, 'sortable-item') == self) currentItem = $(event.target);
 62 
 63 		if(!currentItem) return false;
 64 
 65 
 66 		this.currentItem = currentItem;
 67 		return true;
 68

 71 	_mouseStart: function(event, overrideHandle, noActivation) {
 72 
 73 		this.currentContainer = this;
 74 
 75 		//We only need to call refreshPositions, because the refreshItems call has been moved to mouseCapture
 76 		this.refreshPositions();
 77 
 78 		//Create and append the visible helper
 79 		this.helper = this._createHelper(event);
 80 
 81 		//Cache the helper size
 82 		this._cacheHelperProportions();
 83 
 84 		/*
 85 		 * - Position generation -
 86 		 * This block generates everything position related - it's the core of draggables.
 87 		 */
 88 
 89 		//Cache the margins of the original element
 90 		this._cacheMargins();
 91 
 92 		//Get the next scrolling parent
 93 		this.scrollParent = this.helper.scrollParent();
 94 
 95 		//The element's absolute position on the page minus margins
 96 		this.offset = this.currentItem.offset();
 97 		this.offset = {
 98 			top: this.offset.top - this.margins.top,
 99 			left: this.offset.left - this.margins.left
100 		};
101 
102 		// Only after we got the offset, we can change the helper's position to absolute
103 		// TODO: Still need to figure out a way to make relative sorting possible
104 		this.helper.css("position", "absolute");
105 		this.cssPosition = this.helper.css("position");
106 
107 		$.extend(this.offset, {
108 			click: { //Where the click happened, relative to the element
109 				left: event.pageX - this.offset.left,
110 				top: event.pageY - this.offset.top
111 			},
112 			parent: this._getParentOffset(),
113 			relative: this._getRelativeOffset() //This is a relative to absolute position minus the actual position calculation - only used for relative positioned helper
114 		});
115 
116 		//Generate the original position
117 		this.originalPosition = this._generatePosition(event);
118 		this.originalPageX = event.pageX;
119 		this.originalPageY = event.pageY;
120 
121 
122 		//Cache the former DOM position
123 		this.domPosition = { prev: this.currentItem.prev()[0], parent: this.currentItem.parent()[0] };
124 
125 		//If the helper is not the original, hide the original so it's not playing any role during the drag, won't cause anything bad this way
126 		if(this.helper[0] != this.currentItem[0]) {
127 			this.currentItem.hide();
128 		}
129 
130 		//Create the placeholder
131 		this._createPlaceholder();
132 
133 			this.helper.css("zIndex", 1000);
134 
135 
136 		//Prepare scrolling
137 		if(this.scrollParent[0] != document && this.scrollParent[0].tagName != 'HTML')
138 			this.overflowOffset = this.scrollParent.offset();
139 
140 		//Call callbacks
141 		this._trigger("start", event, this._uiHash());
142 
143 		//Recache the helper size
144 		if(!this._preserveHelperProportions)
145 			this._cacheHelperProportions();
146 
147 		this.helper.addClass("ui-sortable-helper");
148 		this._mouseDrag(event); //Execute the drag once - this causes the helper not to be visible before getting its correct position
149 		return true;

153 	_mouseDrag: function(event) {
154 
155 		//Compute the helpers position
156 		this.position = this._generatePosition(event);
157 		this.positionAbs = this._convertPositionTo("absolute");
158 
159 		if (!this.lastPositionAbs) {
160 			this.lastPositionAbs = this.positionAbs;
161 		}
162 
163 		//Do scrolling
164 			var o = this.options, scrolled = false;
165 
166 				if((this.overflowOffset.top + this.scrollParent[0].offsetHeight) - event.pageY < 20)
167 					this.scrollParent[0].scrollTop = scrolled = this.scrollParent[0].scrollTop + 20;
168 				else if(event.pageY - this.overflowOffset.top < 20)
169 					this.scrollParent[0].scrollTop = scrolled = this.scrollParent[0].scrollTop - 20;
170 
171 				if((this.overflowOffset.left + this.scrollParent[0].offsetWidth) - event.pageX < 20)
172 					this.scrollParent[0].scrollLeft = scrolled = this.scrollParent[0].scrollLeft + 20;
173 				else if(event.pageX - this.overflowOffset.left < 20)
174 					this.scrollParent[0].scrollLeft = scrolled = this.scrollParent[0].scrollLeft - 20;
175 
176 		//Regenerate the absolute position used for position checks
177 		this.positionAbs = this._convertPositionTo("absolute");
178 
179 		//Set the helper position
180 		this.helper[0].style.left = this.position.left+'px';
181 		this.helper[0].style.top = this.position.top+'px';
182 
183 		//Rearrange
184 		for (var i = this.items.length - 1; i >= 0; i--) {
185 
186 			//Cache variables and intersection, continue if no intersection
187 			var item = this.items[i], itemElement = item.item[0], intersection = this._intersectsWithPointer(item);
188 			if (!intersection) continue;
189 
190 			if(itemElement != this.currentItem[0] //cannot intersect with itself
191 				&&	this.placeholder[intersection == 1 ? "next" : "prev"]()[0] != itemElement //no useless actions that have been done before
192 				&&	!$.contains(this.placeholder[0], itemElement) //no action if the item moved is the parent of the item checked
193 				&& (this.options.type == 'semi-dynamic' ? !$.contains(this.element[0], itemElement) : true)
194 				//&& itemElement.parentNode == this.placeholder[0].parentNode // only rearrange items within the same container
195 			) {
196 
197 				this.direction = intersection == 1 ? "down" : "up";
198 
199 				if (this._intersectsWithSides(item)) {
200 					this._rearrange(event, item);
201 				} else {
202 					break;
203 				}
204 
205 				break;
206 			}
207 		}
208 
209 		this.lastPositionAbs = this.positionAbs;
210 		return false;
211 
212 	},

225 	_intersectsWith: function(item) {
226 
227 		var x1 = this.positionAbs.left,
228 			x2 = x1 + this.helperProportions.width,
229 			y1 = this.positionAbs.top,
230 			y2 = y1 + this.helperProportions.height;
231 
232 		var l = item.left,
233 			r = l + item.width,
234 			t = item.top,
235 			b = t + item.height;
236 
237 		var dyClick = this.offset.click.top,
238 			dxClick = this.offset.click.left;
239 
240 		var isOverElement = (y1 + dyClick) > t && (y1 + dyClick) < b && (x1 + dxClick) > l && (x1 + dxClick) < r;
241 
242 			return (l < x1 + (this.helperProportions.width / 2) // Right Half
243 				&& x2 - (this.helperProportions.width / 2) < r // Left Half
244 				&& t < y1 + (this.helperProportions.height / 2) // Bottom Half
245 				&& y2 - (this.helperProportions.height / 2) < b ); // Top Half
246 
247 	},

249 	_isOverAxis: function( x, reference, size ) {
250         //Determines when x coordinate is over "b" element axis
251         return ( x > reference ) && ( x < ( reference + size ) );
252     },
253 	
254 	_intersectsWithPointer: function(item) {
255 
256 		var isOverElementHeight = this._isOverAxis(this.positionAbs.top + this.offset.click.top, item.top, item.height),
257 			isOverElementWidth = this._isOverAxis(this.positionAbs.left + this.offset.click.left, item.left, item.width),
258 			isOverElement = isOverElementHeight && isOverElementWidth,
259 			verticalDirection = this._getDragVerticalDirection(),
260 			horizontalDirection = this._getDragHorizontalDirection();
261 
262 		if (!isOverElement)
263 			return false;
264 
265 		return this.floating ?
266 			( ((horizontalDirection && horizontalDirection == "right") || verticalDirection == "down") ? 2 : 1 )
267 			: ( verticalDirection && (verticalDirection == "down" ? 2 : 1) );
268 
269 	},
270 
271 	_intersectsWithSides: function(item) {
272 
273 		var isOverBottomHalf = this._isOverAxis(this.positionAbs.top + this.offset.click.top, item.top + (item.height/2), item.height),
274 			isOverRightHalf = this._isOverAxis(this.positionAbs.left + this.offset.click.left, item.left + (item.width/2), item.width),
275 			verticalDirection = this._getDragVerticalDirection(),
276 			horizontalDirection = this._getDragHorizontalDirection();
277 
278 		if (this.floating && horizontalDirection) {
279 			return ((horizontalDirection == "right" && isOverRightHalf) || (horizontalDirection == "left" && !isOverRightHalf));
280 		} else {
281 			return verticalDirection && ((verticalDirection == "down" && isOverBottomHalf) || (verticalDirection == "up" && !isOverBottomHalf));
282 		}
283 
284 	},
285 
286 	_getDragVerticalDirection: function() {
287 		var delta = this.positionAbs.top - this.lastPositionAbs.top;
288 		return delta != 0 && (delta > 0 ? "down" : "up");
289 	},
290 
291 	_getDragHorizontalDirection: function() {
292 		var delta = this.positionAbs.left - this.lastPositionAbs.left;
293 		return delta != 0 && (delta > 0 ? "right" : "left");
294 	},

302 	_refreshItems: function(event) {
303 
304 		this.items = [];
305 		this.containers = [this];
306 		var items = this.items;
307 		var queries = [$(this.options.items, this.element), this];
308 
309 			var targetData = queries[1];
310 			var _queries = queries[0];
311 
312 			for (var j=0, queriesLength = _queries.length; j < queriesLength; j++) {
313 				var item = $(_queries[j]);
314 
315 				item.data('sortable-item', targetData); // Data for target checking (mouse manager)
316 
317 				items.push({
318 					item: item,
319 					instance: targetData,
320 					width: 0, height: 0,
321 					left: 0, top: 0
322 				});
323 			};
324 
325 	},

327 	refreshPositions: function(fast) {
328 
329 		//This has to be redone because due to the item being moved out/into the offsetParent, the offsetParent's position will change
330 		if(this.offsetParent && this.helper) {
331 			this.offset.parent = this._getParentOffset();
332 		}
333 
334 		for (var i = this.items.length - 1; i >= 0; i--){
335 			var item = this.items[i];
336 
337 			//We ignore calculating positions of all connected containers when we're not over them
338 			if(item.instance != this.currentContainer && this.currentContainer && item.item[0] != this.currentItem[0])
339 				continue;
340 
341 			var t = item.item;
342 
343 			if (!fast) {
344 				item.width = t.outerWidth();
345 				item.height = t.outerHeight();
346 			}
347 
348 			var p = t.offset();
349 			item.left = p.left;
350 			item.top = p.top;
351 		};
352 		return this;
353 	},

356

358

398 	_getParentOffset: function() {
399 
400 
401 		//Get the offsetParent and cache its position
402 		this.offsetParent = this.helper.offsetParent();
403 		var offsetp = this.offsetParent, po = offsetp.offset();
404 
405 		// This is a special case where we need to modify a offset calculated on start, since the following happened:
406 		// 1. The position of the helper is absolute, so it's position is calculated based on the next positioned parent
407 		// 2. The actual offset parent is a child of the scroll parent, and the scroll parent isn't the document, which means that
408 		//    the scroll is included in the initial calculation of the offset of the parent, and never recalculated upon drag
409 		if(this.cssPosition == 'absolute' && this.scrollParent[0] != document && $.contains(this.scrollParent[0], offsetp[0])) {
410 			po.left += this.scrollParent.scrollLeft();
411 			po.top += this.scrollParent.scrollTop();
412 		}
413 
414 		if((offsetp[0] == document.body) //This needs to be actually done for all browsers, since pageX/pageY includes this information
415 		|| (offsetp[0].tagName && offsetp[0].tagName.toLowerCase() == 'html' && $.browser.msie)) //Ugly IE fix
416 			po = { top: 0, left: 0 };
417 
418 		return {
419 			top: po.top + (parseInt(offsetp.css("borderTopWidth"),10) || 0),
420 			left: po.left + (parseInt(offsetp.css("borderLeftWidth"),10) || 0)
421 		};
422 
423 	},
424 
425 	_getRelativeOffset: function() {
426 
427 		if(this.cssPosition == "relative") {
428 			var p = this.currentItem.position();
429 			return {
430 				top: p.top - (parseInt(this.helper.css("top"),10) || 0) + this.scrollParent.scrollTop(),
431 				left: p.left - (parseInt(this.helper.css("left"),10) || 0) + this.scrollParent.scrollLeft()
432 			};
433 		} else {
434 			return { top: 0, left: 0 };
435 		}
436 
437 	},

440 		this.margins = {
441 			left: (parseInt(this.currentItem.css("marginLeft"),10) || 0),
442 			top: (parseInt(this.currentItem.css("marginTop"),10) || 0)
443 		};
444 	},

447 		this.helperProportions = {
448 			width: this.helper.outerWidth(),
449 			height: this.helper.outerHeight()
450 		};

453 	_convertPositionTo: function(d, pos) {
454 
455 		if(!pos) pos = this.position;
456 		var mod = d == "absolute" ? 1 : -1;
457 		var o = this.options, scroll = this.cssPosition == 'absolute' && !(this.scrollParent[0] != document && $.contains(this.scrollParent[0], this.offsetParent[0])) ? this.offsetParent : this.scrollParent, scrollIsRootNode = (/(html|body)/i).test(scroll[0].tagName);
458 
459 		return {
460 			top: (
461 				pos.top																	// The absolute mouse position
462 				+ this.offset.relative.top * mod										// Only for relative positioned nodes: Relative offset from element to offset parent
463 				+ this.offset.parent.top * mod											// The offsetParent's offset without borders (offset + border)
464 				- ($.browser.safari && this.cssPosition == 'fixed' ? 0 : ( this.cssPosition == 'fixed' ? -this.scrollParent.scrollTop() : ( scrollIsRootNode ? 0 : scroll.scrollTop() ) ) * mod)
465 			),
466 			left: (
467 				pos.left																// The absolute mouse position
468 				+ this.offset.relative.left * mod										// Only for relative positioned nodes: Relative offset from element to offset parent
469 				+ this.offset.parent.left * mod											// The offsetParent's offset without borders (offset + border)
470 				- ($.browser.safari && this.cssPosition == 'fixed' ? 0 : ( this.cssPosition == 'fixed' ? -this.scrollParent.scrollLeft() : scrollIsRootNode ? 0 : scroll.scrollLeft() ) * mod)
471 			)
472 		};
473 
474 	},
475 
476 	_generatePosition: function(event) {
477 
478 		var o = this.options, scroll = this.cssPosition == 'absolute' && !(this.scrollParent[0] != document && $.contains(this.scrollParent[0], this.offsetParent[0])) ? this.offsetParent : this.scrollParent, scrollIsRootNode = (/(html|body)/i).test(scroll[0].tagName);
479 
480 		// This is another very weird special case that only happens for relative elements:
481 		// 1. If the css position is relative
482 		// 2. and the scroll parent is the document or similar to the offset parent
483 		// we have to refresh the relative offset during the scroll so there are no jumps
484 		if(this.cssPosition == 'relative' && !(this.scrollParent[0] != document && this.scrollParent[0] != this.offsetParent[0])) {
485 			this.offset.relative = this._getRelativeOffset();
486 		}
487 
488 		var pageX = event.pageX;
489 		var pageY = event.pageY;
490 
491 		return {
492 			top: (
493 				pageY																// The absolute mouse position
494 				- this.offset.click.top													// Click offset (relative to the element)
495 				- this.offset.relative.top												// Only for relative positioned nodes: Relative offset from element to offset parent
496 				- this.offset.parent.top												// The offsetParent's offset without borders (offset + border)
497 				+ ($.browser.safari && this.cssPosition == 'fixed' ? 0 : ( this.cssPosition == 'fixed' ? -this.scrollParent.scrollTop() : ( scrollIsRootNode ? 0 : scroll.scrollTop() ) ))
498 			),
499 			left: (
500 				pageX																// The absolute mouse position
501 				- this.offset.click.left												// Click offset (relative to the element)
502 				- this.offset.relative.left												// Only for relative positioned nodes: Relative offset from element to offset parent
503 				- this.offset.parent.left												// The offsetParent's offset without borders (offset + border)
504 				+ ($.browser.safari && this.cssPosition == 'fixed' ? 0 : ( this.cssPosition == 'fixed' ? -this.scrollParent.scrollLeft() : scrollIsRootNode ? 0 : scroll.scrollLeft() ))
505 			)
506 		};
507 
508 	},

510 	_rearrange: function(event, i, a, hardRefresh) {
511 
512 		a ? a[0].appendChild(this.placeholder[0]) : i.item[0].parentNode.insertBefore(this.placeholder[0], (this.direction == 'down' ? i.item[0] : i.item[0].nextSibling));
513 
514 		//Various things done here to improve the performance:
515 		// 1. we create a setTimeout, that calls refreshPositions
516 		// 2. on the instance, we have a counter variable, that get's higher after every append
517 		// 3. on the local scope, we copy the counter variable, and check in the timeout, if it's still the same
518 		// 4. this lets only the last addition to the timeout stack through
519 		this.counter = this.counter ? ++this.counter : 1;
520 		var self = this, counter = this.counter;
521 
522 		window.setTimeout(function() {
523 			if(counter == self.counter) self.refreshPositions(!hardRefresh); //Precompute after each DOM insertion, NOT on mousemove
524 		},0);
525 
526 	},
527 
528 	_clear: function(event, noPropagation) {
529 
530 		this.reverting = false;
531 		// We delay all events that have to be triggered to after the point where the placeholder has been removed and
532 		// everything else normalized again
533 		var delayedTriggers = [], self = this;
534 
535 		// We first have to update the dom position of the actual currentItem
536 		// Note: don't do it if the current item is already removed (by a user), or it gets reappended (see #4088)
537 		if(!this._noFinalSort && this.currentItem.parent().length) this.placeholder.before(this.currentItem);
538 		this._noFinalSort = null;
539 
540 		if(this.helper[0] == this.currentItem[0]) {
541 			for(var i in this._storedCSS) {
542 				if(this._storedCSS[i] == 'auto' || this._storedCSS[i] == 'static') this._storedCSS[i] = '';
543 			}
544 			this.currentItem.css(this._storedCSS).removeClass("ui-sortable-helper");
545 		} else {
546 			this.currentItem.show();
547 		}
548 
549 		this.helper.css("zIndex", 1000); //Reset z-index
550 
551 		if(this.cancelHelperRemoval) {
552 			if(!noPropagation) {
553 				//this._trigger("beforeStop", event, this._uiHash());
554 				for (var i=0; i < delayedTriggers.length; i++) { delayedTriggers[i].call(this, event); }; //Trigger all delayed events
555 				this._trigger("stop", event, this._uiHash());
556 			}
557 			return false;
558 		}
559 
560 		this.placeholder[0].parentNode.removeChild(this.placeholder[0]);
561 
562 		if(this.helper[0] != this.currentItem[0]) this.helper.remove(); this.helper = null;
563 
564 		if(!noPropagation) {
565 			for (var i=0; i < delayedTriggers.length; i++) { delayedTriggers[i].call(this, event); }; //Trigger all delayed events
566 			this._trigger("stop", event, this._uiHash());
567 		}
568 
569 		this.fromOutside = false;
570 		return true;
571 
572 	},
