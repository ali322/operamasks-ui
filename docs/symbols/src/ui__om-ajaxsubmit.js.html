 14 	Usage Note:
 15 	-----------
 16 	Do not use both omAjaxSubmit and ajaxForm on the same form.  These
 17 	functions are intended to be exclusive.  Use omAjaxSubmit if you want
 18 	to bind your own submit handler to the form.  For example,
 19 
 20 	$(document).ready(function() {
 21 		$('#myForm').bind('submit', function(e) {
 22 			e.preventDefault(); // <-- important
 23 			$(this).omAjaxSubmit({
 24 				target: '#output'
 25 			});
 26 		});
 27 	});
 28 
 29 	Use ajaxForm when you want the plugin to manage all the event binding
 30 	for you.  For example,
 31 
 32 	$(document).ready(function() {
 33 		$('#myForm').ajaxForm({
 34 			target: '#output'
 35 		});
 36 	});
 37 
 38 	When using ajaxForm, the omAjaxSubmit function will be invoked for you
 39 	at the appropriate time.

 42 /**
 43  * @name omAjaxSubmit
 44  * @class
 45  * <div>
 46  * omAjaxSubmit() 提供使用ajax方式提交HTML form的一种机制。本插件会监听表单的submit事件，<br/>
 47  * 覆盖传统的submit事件监听器，而使用ajax方式来处理submit事件。在表单提交之前，本插件会收集<br/>
 48  * 所有的表单字段，并将之序列化后附加在ajax请求的数据域(data)中。支持所有标准的html可提交的<br/>
 49  * 表。元素。
 50  * </div><br/>
 51  * <b>事件回调</b><br/>
 52  * <div>
 53  * 通过丰富的配置参数，omAjaxSubmit可以高度自定制。同时提供多个事件回调函数，在每一次完整<br/>
 54  * 的表单提交的过程中，用户有时机能够对提交的请求进行修改。
 55  * </div>
 56  * <pre>beforeSerialize:</pre>
 57  * <div style="text-indent:2em;">在form序列化之前执行的回调函数。在获取所有form表单字段的值之前，该函数提供了一个操作form的时机</div>
 58  * <pre>beforeSubmit:</pre>
 59  * <div style="text-indent:2em;">在form被提交之前执行的回调函数。该函数提供了一个时机来执行预提交的逻辑，比如表单校验</div><br/>
 60  * <b>工具方法</b><br/>
 61  * <div>omAjaxSubmit还提供了一系列静态工具方法，用于方便地操作表单及其字段。</div>
 62  * <pre>$.fn.formToArray()</pre>
 63  * <div style="text-indent:2em;">将表单所有元素转换成key/value的数组，例如[{name:'username', value:'jack'},{name:'password', value:'secret'}]，<br/>
 64  * 注意:该数组将作为参数传递给beforeSubmit函数</div>
 65  * <pre>$.fn.formSerialize()</pre>
 66  * <div style="text-indent:2em;">将表单数据序列化成可提交的字符串，例如name1=value1&name2=value2</div>
 67  * <pre>$.fn.fieldSerialize()</pre>
 68  * <div style="text-indent:2em;">将表单所有元素序列化成可提交的字符串，例如name1=value1&name2=value2</div>
 69  * <pre>$.fn.fieldValue()</pre>
 70  * <div style="text-indent:2em;">获取当前元素(或元素数组)的值</div>
 71  * <pre>$.fieldValue(successful)</pre>
 72  * <div style="text-indent:2em;">静态工具方法，用于获取元素的值，参数successful的意义同上</div>
 73  * <pre>$.fn.clearForm()</pre>
 74  * <div style="text-indent:2em;">清空当前表单各个元素的值</div>
 75  * <pre>$.fn.clearFields()</pre>
 76  * <div style="text-indent:2em;">清空当前元素(或元素数组)的值</div>
 77  * <pre>$.fn.resetForm()</pre>
 78  * <div style="text-indent:2em;">重置当前表单各个元素的值</div>
 79  * <pre>$.fn.enable(b)</pre>
 80  * <div style="text-indent:2em;">设置当前元素(或元素数组)的使能状态</div>
 81  * <pre>$.fn.selected(selected)</pre>
 82  * <div style="text-indent:2em;">设置当前元素(或元素数组)的选中状态</div><br/>
 83  * <b>示例</b><br/>
 84  * <pre>
 85  *  $(document).ready(function() {
 86  *      $('#myForm').bind('submit', function(e) {
 87  *          e.preventDefault(); //阻止form默认的提交行为
 88  *              $(this).omAjaxSubmit(//使用ajax提交
 89  *                  {
 90  *                      target: '#output'
 91  *                  }
 92  *              );
 93  *      });

 96  * </pre>

499 			}

506 				$form.attr({
507 					encoding: 'multipart/form-data',
508 					enctype:  'multipart/form-data'

835 /**
836  * formToArray() gathers form element data into an array of objects that can
837  * be passed to any of the following ajax functions: $.get, $.post, or load.
838  * Each object in the array has both a 'name' and 'value' property.  An example of
839  * an array for a simple login form might be:
840  *
841  * [ { name: 'username', value: 'jresig' }, { name: 'password', value: 'secret' } ]
842  *
843  * It is this array that is passed to pre-submit callback functions provided to the
844  * omAjaxSubmit() and ajaxForm() methods.
845  */
846 $.fn.formToArray = function(semantic) {
847 	var a = [];
848 	if (this.length === 0) {
849 		return a;
850 	}
851 
852 	var form = this[0];
853 	var els = semantic ? form.getElementsByTagName('*') : form.elements;
854 	if (!els) {
855 		return a;
856 	}
857 
858 	var i,j,n,v,el,max,jmax;
859 	for(i=0, max=els.length; i < max; i++) {
860 		el = els[i];
861 		n = el.name;
862 		if (!n) {
863 			continue;
864 		}
865 
866 		if (semantic && form.clk && el.type == "image") {
867 			// handle image inputs on the fly when semantic == true
868 			if(!el.disabled && form.clk == el) {
869 				a.push({name: n, value: $(el).val()});
870 				a.push({name: n+'.x', value: form.clk_x}, {name: n+'.y', value: form.clk_y});
871 			}
872 			continue;
873 		}
874 
875 		v = $.fieldValue(el, true);
876 		if (v && v.constructor == Array) {
877 			for(j=0, jmax=v.length; j < jmax; j++) {
878 				a.push({name: n, value: v[j]});
879 			}
880 		}
881 		else if (v !== null && typeof v != 'undefined') {
882 			a.push({name: n, value: v});
883 		}
884 	}
885 
886 	if (!semantic && form.clk) {
887 		// input type=='image' are not found in elements array! handle it here
888 		var $input = $(form.clk), input = $input[0];
889 		n = input.name;
890 		if (n && !input.disabled && input.type == 'image') {
891 			a.push({name: n, value: $input.val()});
892 			a.push({name: n+'.x', value: form.clk_x}, {name: n+'.y', value: form.clk_y});
893 		}
894 	}
895 	return a;

932 /**
933  * Returns the value(s) of the element in the matched set.  For example, consider the following form:
934  *
935  *  <form><fieldset>
936  *	  <input name="A" type="text" />
937  *	  <input name="A" type="text" />
938  *	  <input name="B" type="checkbox" value="B1" />
939  *	  <input name="B" type="checkbox" value="B2"/>
940  *	  <input name="C" type="radio" value="C1" />
941  *	  <input name="C" type="radio" value="C2" />
942  *  </fieldset></form>
943  *
944  *  var v = $(':text').fieldValue();
945  *  // if no values are entered into the text inputs
946  *  v == ['','']
947  *  // if values entered into the text inputs are 'foo' and 'bar'
948  *  v == ['foo','bar']
949  *
950  *  var v = $(':checkbox').fieldValue();
951  *  // if neither checkbox is checked
952  *  v === undefined
953  *  // if both checkboxes are checked
954  *  v == ['B1', 'B2']
955  *
956  *  var v = $(':radio').fieldValue();
957  *  // if neither radio is checked
958  *  v === undefined
959  *  // if first radio is checked

975 			continue;
976 		}
977 		v.constructor == Array ? $.merge(val, v) : val.push(v);
978 	}
979 	return val;
