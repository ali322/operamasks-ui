760         _changePage : function(ctype) { // change page
761             if (this.loading) {
762                 return true;
763             }
764             var el=this.element,
765                 pageData = this.pageData,
766                 nowPage=pageData.nowPage,
767                 totalPages=pageData.totalPages,
768                 newPage = nowPage;
769             this._oldPage = nowPage;//保存好旧的页数，有些插件如sort是需要用到的
770             switch (ctype) {
771                 case 'first':
772                     newPage = 1;
773                     break;
774                 case 'prev':
775                     if (nowPage > 1) {
776                         newPage = nowPage - 1;
777                     }
778                     break;
779                 case 'next':
780                     if (nowPage < totalPages) {
781                         newPage = nowPage + 1;
782                     }
783                     break;
784                 case 'last':
785                     newPage = totalPages;
786                     break;
787                 case 'input':
788                     var nv = parseInt($('.pControl input', el.closest('.om-grid')).val());
789                     if (isNaN(nv)) {
790                         nv = nowPage;
791                     }
792                     if (nv < 1) {
793                         nv = 1;
794                     } else if (nv > totalPages) {
795                         nv = totalPages;
796                     }
797                     $('.pControl input', this.pDiv).val(nv);
798                     newPage = nv;
799                     break;
800                 default:
801                     if (/\d/.test(ctype)) {
802                         var nv = parseInt(ctype);
803                         if (isNaN(nv)) {
804                             nv = 1;
805                         }
806                         if (nv < 1) {
807                             nv = 1;
808                         } else if (nv > totalPages) {
809                             nv = totalPages;
810                         }
811                         $('.pControl input', el.closest('.om-grid')).val(nv);
812                         newPage = nv;
813                     }
814             }
815             if (newPage == nowPage) {
816                 return false;
817             }
818             //触发事件
819             if(this._trigger("onPageChange",null,ctype,newPage)===false){
820                 return;
821             }
822             //修改pageData
823             pageData.nowPage=newPage;
824             //刷新数据
825             this._populate();
