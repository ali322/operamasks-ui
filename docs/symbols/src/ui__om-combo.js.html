361                 options.editable = this.options.editable = false;
362             }
363             
364             this._refeshEmptyText(options.emptyText);
365             
366             options.disabled ? inputEl.attr('disabled', true) : inputEl.removeAttr('disabled');
367             (options.readOnly || !options.editable) ? inputEl.attr('readonly', 'readOnly') : inputEl.removeAttr('readonly');
368             
369             if (!options.lazyLoad) {
370                 //load data immediately
371                 this._toggleLoading('add');
372                 if(source && typeof source == 'string'){
373                     this._ajaxLoad(source);
374                 }else if(source && typeof source == 'object'){
375                     this._loadData(source);
376                     this._toggleLoading('remove');
377                 }else{
378                     //neither records nor remote url was found
379                     this.dataHasLoaded = true;
380                     this._toggleLoading('remove');
381                 }
382                 
383             } else {
384                 this.dataHasLoaded = false;
385             }
386             var unusable = options.disabled || options.readOnly;
387             
388             if (unusable) {
389                 this.expandTrigger.addClass('om-state-disabled');
390             } else {
391                 this._bindEvent();

961         _filetrPass:function(filterStrategy,text,record,inputText){
962             if (text === '') {
963                 return true;
964             }
965             if (typeof filterStrategy === 'function') {
966                 return filterStrategy(text, record);
967             } else {
968                 if (filterStrategy === 'first') {
969                     return inputText.indexOf(text) === 0;
970                 } else if (filterStrategy === 'anywhere') {
971                     return inputText.indexOf(text) > -1;
972                 } else if (filterStrategy === 'last') {
973                     var i = inputText.lastIndexOf(text);
974                     return i > -1 && i + text.length == inputText.length;
975                 } else {
976                     return false;
977                 }
978             }
