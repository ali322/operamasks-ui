 22     $.omWidget.addInitListener('om.omGrid',function(){
 23     	var self = this,
 24     		$elem = self.element,
 25     		ops = self.options,
 26     		cm = this._getColModel();
 27     		
 28     	self._triggered = false;//是否已经触发过一次编辑了。
 29     	//如果所有列都不可编辑，那么_globalEditable=false,这时候所有原生的行都不可编辑，但如果此时设置了options.editMode=="insert",
 30     	//则新添加进来的行还是可以编辑的。
 31     	self._globalEditable = false;
 32     	
 33     	/**
 34     	 * <span style='color:red'>(作用于行编辑插件)</span>编辑模式。
 35     	 * 可选值为"all","insert"，默认为"all"。"all"表示所有行都是可编辑的，但如果所有的列都是不可编辑的，那么行编辑插件仍然会失效。
 36     	 * 如果有这样的需求，所有的行都是不可编辑的，但是此表格可以动态添加新行，而且这些新行在持久化到后台之前是可以编辑的，那么就要使用"insert"模式了。 
 37     	 * @name omGrid#editMode
 38     	 * @type String
 39     	 * @default "all"
 40     	 * @example
 41     	 * $('.selector').omGrid({width : 600, editMode:"insert");
 42     	 */
 43     	ops.editMode = ops.editMode || "all";//默认为all模式，即所有行都可以编辑
 44     	self._allEditMode = ops.editMode=="all";
 45     		
 46     	for(var i=0,len=cm.length; i<len; i++){
 47     		self._globalEditable = cm[i]["editor"]? true : false;
 48     		if(self._globalEditable){
 49     			break;
 50     		}
 51     	}
 52

 54     		return ;//不开启编辑功能，该插件任何方法都会失效。
 55     	}
 56 		self._editComps = {};//缓存每一列的编辑组件，key为colModel中的name,value为{model:模型,instance:组件实例,type:类型}
 57 		self._errorHolders = {};//  name:div   每个编辑单元格对应的出错信息容器
 58 		self._colWidthResize;//如果grid可编辑，在拖动标题栏改变列宽时，当前并不处于编辑状态，则设置此值为true,这样在显示编辑条时由此依据要重新计算各个组件的宽度
 59 		
 60 		ops._onRefreshCallbacks.push(_onRefresh);//只有数据刷新后才可以进行初始化
 61 		ops._onResizableStopCallbacks.push(_onResizable);
 62 		ops._onResizeCallbacks.push(_onResize);//动态改变宽高时要重新计算编辑条的大小
 63

 65         	editRenderer.call(this , self);
 66         }); 
 67         this.tbody.delegate('tr.om-grid-row','click',function(event){
 68         	if(self._triggered && self._editView.editing){
 69 				if(self._validator){
 70 					self._validator.valid() && editRenderer.call(this , self);

 73 				}
 74         	}
 75         });

 78         $elem.parent().scroll(function(){
 79         	if(self._triggered){
 80         		if(btnScrollTimer){
 81         			clearTimeout(btnScrollTimer);
 82         		}
 83         		btnScrollTimer = setTimeout(function(){
 84     				var pos = _getEditBtnPosition(self);
 85             		self._editView.editBtn.animate({"left":pos.left,"top":pos.top},self._editView.editing?"fast":0);
 86             		btnScrollTimer = null;
 87     			} , 300);
 88         	}

139 	    	 * @function

154 	    	 * @function
155 	    	 * @param index 行索引，从0开始；或者为行索引数组(一般由getSelections得到)
156 	    	 * @name omGrid#deleteRow
157 	    	 * @returns jQuery对象
158 	    	 * @example
159 	    	 * $(".selector").omGrid("deleteRow" , 0);<br />

197 	        /** <span style='color:red'>(作用于行编辑插件)。</span>获取所有未保存的修改。如果没有指定type,返回的是所有的修改，格式为: {update:[],insert:[],delete:[]}，如果指定了参数，如
198 	    	 *  指定了"update"，则返回 [{},{}]
199 	    	 * @function
200 	    	 * @param type 可选值为："insert","update","delete" 
201 	    	 * @name omGrid#getChanges
202 	    	 * @returns 若指定了类型，返回[]，否则返回{update:[],insert:[],delete:[]}
203 	    	 * @example
204 	    	 * $(".selector").omGrid("getChanges" , "update");
205 	    	 */
206 	    	getChanges : function(type){
207 	    		var data = {"update":[] , "insert":[] , "delete":[]},
208 	    			reqType = type? type : "update",
209 	    			cData = this._changeData;
210 	    			
211 	    		if(reqType === "update"){
212 	    			var uData = cData[reqType];
213 	    			for(var i in uData){
214 	    				own.call(uData , i) && data[reqType].push($.extend(true , {} , this._rowIdDataMap[i] , uData[i]));
215 	    			}
216 	    			reqType = type? type : "insert";
217 	    		}
218 	    		if(reqType === "insert"){
219 	    			var iData = cData[reqType];
220 	    			for(var i in iData){
221 	    				own.call(iData , i) && data[reqType].push(iData[i]);
222 	    			}
223 	    			reqType = type? type : "delete";
224 	    		}
225 	    		if(reqType === "delete"){
226 	    			var dData = cData[reqType];
227 	    			for(var i in dData){
228 	    				own.call(dData , i) && data[reqType].push(dData[i]);
229 	    			}
230 	    		}
231 	    		if(type){
232 	    			return data[type];
233 	    		}else{
234 	    			return data;
235 	    		}
236 	    	},

238 		 	/** <span style='color:red'>(作用于行编辑插件)。</span>在指定位置动态插入一行。
239 	    	 * @function
240 	    	 * @param index 行索引，从0开始，或为"begin","end"分别表示在表格最前和最后插入行。
241 	    	 * @param rowData 插入的新行的初始值
242 	    	 * @param forceAdd 强制添加，设为true表示直接添加，不会弹出编辑框
243 	    	 * @name omGrid#insertRow
244 	    	 * @returns jQuery对象
245 	    	 * @example
246 	    	 * $(".selector").omGrid("insertRow");//插入最前面<br/>
247 	    	 * $(".selector").omGrid("insertRow" , 1);//插入索引1的位置<br/>
248 	    	 * $(".selector").omGrid("insertRow" , "end" , {id:"1"});//在末尾插入，并使用指定数据初始化<br/>
249 	    	 * $(".selector").omGrid("insertRow" , true);//插入最前面，并直接添加，不显示编辑框<br/>
250 	    	 * $(".selector").omGrid("insertRow" , 0 , {id:"2"} , true);//用指定数据在最前面插入新行，并且不显示编辑框<br/>
251 	    	 */
252 	    	insertRow : function(index , rowData , forceAdd){
253 	    		var ops = this.options,
254 	    			cm = this._getColModel(),
255 	    			$elem = this.element;
256 	    		
257 	    		if(!_isEditable(this) || this._rowAdding){
258 	    			return ;
259 	    		}
260 	    		//insertRow({id:10})
261 	    		if($.isPlainObject(index)){
262 	    			rowData = index;
263 	    			index = 0;
264 	    		}
265 	    		//insertRow(true)
266 	    		if(index === true){
267 	    			rowData = {};
268 	    			index = 0;
269 	    			forceAdd = true;
270 	    		}
271 	    		var $trs = this._getTrs(),
272 	    			rd = {};
273 	    		index = ("begin"==index || index==undefined)? 0 : ("end"==index? $trs.length : index);
274 	    
275 	    		for(var i=0,len=cm.length; i<len; i++){
276 	    			rd[cm[i]["name"]] = "";//默认都为空值
277 	        	}
278 	    		this._changeData["insert"][this._guid] = $.extend(true , rd , rowData);
279 	    		
280 	    		//创建新行
281 	    		var rowValues=this._buildRowCellValues(cm,rd,index),
282 	    			trContent = [],
283 	    			rowClasses=ops.rowClasses;
284 	    			isRowClassesFn= (typeof rowClasses === 'function'),
285 	    			rowCls = isRowClassesFn? rowClasses(index,rd):rowClasses[index % rowClasses.length],
286 	    			tdTmp = "<td align='$' abbr='$' class='grid-cell-dirty $'><div align='$' class='$' style='width:$px'>$</div></td>";//td模板
287 	    			    			
288 	    		trContent.push("<tr class='om-grid-row " + rowCls + "' _grid_row_id="+(this._guid++)+" _insert='true'>");
289 	    		this._getHeaderCols().each(function(i){
290 	                var axis = $(this).attr('axis'),
291 	                	wrap=false,
292 	                	html,
293 	                	cols,
294 	                	j;
295 	                if(axis == 'indexCol'){
296 	                    html="<a class='om-icon'>新行</a>";
297 	                }else if(axis == 'checkboxCol'){
298 	                    html = '<span class="checkbox"/>';
299 	                }else if(axis.substring(0,3)=='col'){
300 	                    var colIndex=axis.substring(3);
301 	                    html=rowValues[colIndex];
302 	                    if(cm[colIndex].wrap){
303 							wrap=true;
304 						} 
305 	                }else{
306 	                    html='';
307 	                }
308 	                cols = [this.align , this.abbr , axis , this.align , wrap?'wrap':'', $('div',$(this)).width() , html];
309 	                j=0;
310 	                trContent.push(tdTmp.replace(/\$/g , function(){
311 	                	return cols[j++];
312 	                }));
313 	            });
314 	            trContent.push("</tr>");
315 	    		
316 	    		var $tr = $(trContent.join(" ")),
317 	    			$destTr,
318 	    			$next;
319 	    		if(index==0){
320 	    			$tr.prependTo($elem.find(">tbody"));
321 	    		}else{
322 	    			$destTr = $trs.eq(index-1);
323 	    			$next = $destTr.next();
324 	    			$destTr = $next.hasClass("rowExpand-rowDetails")? $next : $destTr;//处理行详情
325 	    			$destTr.after($tr);
326 	    		}
327 	    		if(!forceAdd){
328 	    			this.editRow(index);
329 	    			//正在添加新行标志，如果这时候校验通不过，新增行在编辑正确并保存之前双击其它行是没有反应的。此外，如果用户点击取消，这时候会删除新增的行,在第一次点击"保存"按钮时要设置此值为false。
330 	    			this._rowAdding = true;
331 	    		}
332 	    	},
333 	    	
334 	    	/** <span style='color:red'>(作用于行编辑插件)。</span>保存客户端数据。注意，此方法不会提交请求到后台，而是会认为所有的数据改变都已经成功提交到后台去了，所以它会清除所有脏数据标志。
335 	    	 * 一般情况下，在您自己的保存方法事件回调中，调用getChanges方法获取当前所有的改变，并自己提交到后台，然后在成功回调方法中再调用本方法。
336 	    	 * 一旦调用本方法后，cancelChanges方法是无法对调用此方法前的所有改变起作用的。
337 	    	 * @function
338 	    	 * @name omGrid#saveChanges
339 	    	 * @returns jQuery对象
340 	    	 * @example
341 	    	 * $(".selector").omGrid("saveChanges");
342 	    	 */
343 	    	saveChanges : function(){
344 	    		this.cancelEdit();
345 				if(_noChanges(this)){
346 					return ;
347 				}
348 				var uData = this._changeData["update"],
349 					$trs = this.element.find("tr.om-grid-row"),
350 					newRowsData = [];
351 				for(var i in uData){
352 					if(own.call(uData , i)){
353 						$.extend(true , this._rowIdDataMap[i] , uData[i]);
354 					}
355 				}
356 				var self = this;
357 				$trs.each(function(index , tr){
358 					var $tr = $(tr);
359 					if($tr.attr("_delete")){
360 						$tr.remove();
361 					}else{
362 						newRowsData.push(_getRowData(self , tr));
363 					}
364 				});
365 				this.pageData.data.rows = newRowsData;
366 				_clearCache(this);
367 				_resetForm(this);//重置编辑表单，清除错误信息
368 				this.refresh();//重用dataSource中的数据进行刷新
369 	    	},

371              * 编辑一行之前执行的方法。
372              * @event

390              * @param rowData 选择的行所代表的JSON对象
391              * @default 无
392              * @example
393              *  $(".selector").omGrid({
394              *      onAfterEdit:function(rowIndex , rowData){
395              *          alert("您刚刚编辑的记录索引为:" + rowIndex);
396              *      }
397              *  });

399              onAfterEdit : function(rowIndex , rowData){},
400              /**
401              * 取消编辑一行时执行的方法。
402              * @event
403              * @name omGrid#onCancelEdit
404              * @default 无
405              * @example
406              *  $(".selector").omGrid({
407              *      onCancelEdit:function(){
408              *          alert("您取消了编辑状态");
409              *      }
410              *  });
411              */
412              onCancelEdit : function(){},
413 	    	/**
414 	    	 * (覆盖)获取最新的所有数据
415 	    	 */
416 	    	getData : function(){
417 	    		var result = this.pageData.data,
418 					$trs = this._getTrs(),
419 					self = this;
420 					
421 	    		if(_isEditable(this) && _hasChange(this)){
422 	    			result = {total:result.total};
423 	    			result.rows = [];
424 					$trs.each(function(index , tr){
425 	    				result.rows.push(self._getRowData(index));
426 					});
427 				}
428 				return result;
429 	    	},
430 	    	/**
431 	    	 * (覆盖)获取最新的行数据。由于可以插入新的数据行，所以此方法要进行重写。
432 	    	 */
433 	    	_getRowData : function(index){
434 	    		var $tr = this._getTrs().eq(index),
435 	    			rowId = _getRowId($tr),
436 	    			rowData;
437 	    		if(_noChanges(this)){
438 	    			return this.pageData.data.rows[index];
439 	    		}
440 	    		if($tr.attr("_insert")){
441 	    			rowData = this._changeData.insert[rowId];
442 	    		}else{
443 	    			var origRowData = this._rowIdDataMap[rowId],
444 	    				uData = this._changeData["update"];
445 	    			rowData = origRowData;
446 	    			if(uData[rowId]){
447 	    				rowData = $.extend(true , {} , origRowData , uData[rowId]);
448 	    			}
449 	    		}
450 	    		return rowData;
451 	    	}
452 	 	});

455     function editRenderer(self){
456     	var $tr = $(this),
457     		$elem = self.element,
458     		$editRow,
459     		$editForm,
460     		scrollLeft,
461     		cm = self._getColModel(),
462     		editComp,//{name:{type:组件类型,model:列对应model,instance:组件实例}}
463     		lastValue,
464     		ops = self.options;
465     	
466     	if(!_isEditable(self)){
467     		return ;
468     	}
469     		
470     	//如果是insertEditMode模式，那么除非该行是新增的，否则不可编辑，直接返回
471     	if(!self._allEditMode && !$tr.attr("_insert")){
472     		return ;
473     	}
474     	
475     	//当前行正处于编辑状态，直接返回
476     	if(self._triggered  
477     		&& self._editView.editing
478     		&& _getRowId($tr) == self._editView.rowId){
479 			return ;        		
480     	}
481     	
482     	if(self._rowAdding){//处于新增行编辑中，在新增行第一次保存前此方法不响应。
483     		return ;
484     	}
485     	
486     	var rowIndex = self._getTrs().index($tr),
487     		rowData = self._getRowData(rowIndex);
488     		
489     	if(ops.onBeforeEdit && ops.onBeforeEdit.call(self , rowIndex , rowData) === false){
490     		return ;
491     	}
492     	_showEditView(self , $tr);
493     	$editRow = self._editView.editRow;

834 	function _saveEditValue(self , tr){
835 		var $tr = $(tr),
836 //			$editRow = self._editView.editRow,
837 			comps = self._editComps,
838 			rowId = _getRowId($tr),
839 			index = self._getTrs().index($tr),
840 			rowData = self._getRowData(index);
841 			
842 		$.each(comps , function(name , comp){
843 			var key = comp.model.name,
844 				newValue = _getCompValue(self , $tr , comp),
845 				originalValue,
846 				html,
847 				updateRowData;
848 				
849 			if($tr.attr("_insert")){
850 				self._changeData.insert[rowId][key] = newValue;
851 			}else{
852 				originalValue = _getRowData(self , tr)[key];
853 				updateRowData = self._changeData.update[rowId];
854 				//注意，""==0为true,false==""为true,这里为了正确进行比较，全部转化为字符串再进行比较。
855 				if(String(newValue) === String(originalValue)){
856 					_toggleDirtyFlag($tr , comp.model , false);
857 					updateRowData && delete updateRowData[key];
858 					$.isEmptyObject(updateRowData) && delete self._changeData.update[rowId];
859 				}else{
860 					_toggleDirtyFlag($tr , comp.model , true);
861 					updateRowData = self._changeData.update[rowId] = updateRowData || {};
862 					updateRowData[key] = newValue;
863 				}
864 			}
865 			//更新回表格
866 			if(comp.model.renderer){
867 				html = comp.model.renderer(newValue , rowData ,index);
868 			}else{
869 				html = newValue==undefined?"" : newValue;
870 			}
871 			// $tr.find("td[abbr='"+key+"'] >div").html(html); // avoid xss attack, use the following line instead
872                         $tr.find("td[abbr='"+key+"'] >div")[0].innerHTML = html;
873 		});
874 	}
