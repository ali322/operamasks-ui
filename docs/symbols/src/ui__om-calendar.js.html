466          * @private

468         _templetShow: function(templet, data) {
469             var str_in,value_s,i,m,value,par;
470             if (data instanceof Array) {
471                 str_in = '';
472                 for (i = 0; i < data.length; i++) {
473                     str_in += arguments.callee(templet, data[i]);
474                 }
475                 templet = str_in;
476             } else {
477                 value_s = templet.match(/{\$(.*?)}/g);
478                 if (data !== undefined && value_s !== null) {
479                     for (i = 0,m = value_s.length; i < m; i++) {
480                         par = value_s[i].replace(/({\$)|}/g, '');
481                         value = (data[par] !== undefined) ? data[par] : '';
482                         templet = templet.replace(value_s[i], value);
483                     }
484                 }
485             }
486             return templet;
487         },
