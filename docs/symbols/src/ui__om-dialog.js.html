414 	/**
415      * 关闭对话框.
416      * @name omDialog#close
417      * @function
418      * @returns 支持链式操作，返回JQuery对象
419      * @example
420      * var store = $("#select").omDialog('close');
421      * 
422      */
423 	close: function(event) {
424 		var self = this,
425 			maxZ, thisZ,
426 			options = this.options,
427 			onBeforeClose = options.onBeforeClose,
428 			onClose = options.onClose;
429 		
430 		if (onBeforeClose && false === self._trigger("onBeforeClose",event)) {
431 			return;
432 		}
433 
434 		if (self.overlay) {
435 			self.overlay.destroy();
436 		}
437 		self.uiDialog.unbind('keypress.om-dialog');
438 
439 		self._isOpen = false;
440 
441 		if (self.options.hide) {
442 			self.uiDialog.hide(self.options.hide, function() {
443                 onClose && self._trigger("onClose",event);
444 			});
445 		} else {
446 			self.uiDialog.hide();
447 			onClose && self._trigger("onClose",event);
448 		}
449 
450 		$.om.omDialog.overlay.resize();
451 
452 		// adjust the maxZ to allow other modal dialogs to continue to work (see #4309)
453 		if (self.options.modal) {
454 			maxZ = 0;
455 			$('.om-dialog').each(function() {
456 				if (this !== self.uiDialog[0]) {
457 					thisZ = $(this).css('z-index');
458 					if(!isNaN(thisZ)) {
459 						maxZ = Math.max(maxZ, thisZ);
460 					}
461 				}
462 			});
463 			$.om.omDialog.maxZ = maxZ;
464 		}
465 
466 		return self;
467 	},

482 	// the force parameter allows us to move modal dialogs to their correct
483 	// position on open
484 	moveToTop: function(force, event) {
485 		var self = this,
486 			options = self.options,
487 			saveScroll;
488 
489 		if ((options.modal && !force) ||
490 			(!options.stack && !options.modal)) {
491 			return self._trigger('onFocus', event);
492 		}
493 
494 		if (options.zIndex > $.om.omDialog.maxZ) {
495 			$.om.omDialog.maxZ = options.zIndex;
496 		}
497 		if (self.overlay) {
498 			$.om.omDialog.maxZ += 1;
499 			self.overlay.$el.css('z-index', $.om.omDialog.overlay.maxZ = $.om.omDialog.maxZ);
500 		}
501 
502 		//Save and then restore scroll since Opera 9.5+ resets when parent z-Index is changed.
503 		//  http://ui.jquery.com/bugs/ticket/3193
504 		saveScroll = { scrollTop: self.element.scrollTop(), scrollLeft: self.element.scrollLeft() };
505 		$.om.omDialog.maxZ += 1;
506 		self.uiDialog.css('z-index', $.om.omDialog.maxZ);
507 		self.element.attr(saveScroll);
508 		self._trigger('onFocus', event);
509 
510 		return self;
511 	},

513 	/**
514      * 打开对话框。
515      * @name omDialog#open
516      * @function
517      * @returns 支持链式操作，返回JQuery对象
518      * @example
519      * var store = $("#select").omDialog('open');
520      * 
521      */
522 	open: function() {
523 		if (this._isOpen) { return; }
524 
525 		var self = this,
526 			options = self.options,
527 			uiDialog = self.uiDialog;
528 
529 		self.overlay = options.modal ? new $.om.omDialog.overlay(self) : null;
530 		self._size();
531 		self._position(options.position);
532 		uiDialog.show(options.show);
533 		self.moveToTop(true);
534 
535 		// prevent tabbing out of modal dialogs
536 		if (options.modal) {
537 			uiDialog.bind('keypress.om-dialog', function(event) {
538 				if (event.keyCode !== $.om.keyCode.TAB) {
539 					return;
540 				}
541 
542 				var tabbables = $(':tabbable', this),
543 					first = tabbables.filter(':first'),
544 					last  = tabbables.filter(':last');
545 
546 				if (event.target === last[0] && !event.shiftKey) {
547 					first.focus(1);
548 					return false;
549 				} else if (event.target === first[0] && event.shiftKey) {
550 					last.focus(1);
551 					return false;
552 				}
553 			});
554 		}
555 
556 		// set focus to the first tabbable element in the content area or the first button
557 		// if there are no tabbable elements, set focus on the dialog itself
558 		$(self.element.find(':tabbable').get().concat(
559 			uiDialog.find('.om-dialog-buttonpane :tabbable').get().concat(
560 				uiDialog.get()))).eq(0).focus();
561 
562 		self._isOpen = true;
563 		var onOpen = options.onOpen;
564 		if(onOpen){
565 		    self._trigger("onOpen");
566 		}
567 		return self;
568 	},

570 	_createButtons: function(buttons) {
571 		var self = this,
572 			hasButtons = false,
573 			uiDialogButtonPane = $('<div></div>')
574 				.addClass(
575 					'om-dialog-buttonpane ' +
576 					'om-helper-clearfix'
577 				),
578 			uiButtonSet = $( "<div></div>" )
579 				.addClass( "om-dialog-buttonset" )
580 				.appendTo( uiDialogButtonPane );
581 
582 		// if we already have a button pane, remove it
583 		self.uiDialog.find('.om-dialog-buttonpane').remove();
584 
585 		if (typeof buttons === 'object' && buttons !== null) {
586 			$.each(buttons, function() {
587 				return !(hasButtons = true);
588 			});
589 		}
590 		if (hasButtons) {
591 			$.each(buttons, function(name, props) {
592 				props = $.isFunction( props ) ?
593 					{ click: props, text: name } :
594 					props;
595 				var button = $('<button type="button"></button>')
596 					.click(function() {
597 						props.click.apply(self.element[0], arguments);
598 					})
599 					.appendTo(uiButtonSet);
600 				// can't use .attr( props, true ) with jQuery 1.3.2.
601 				$.each( props, function( key, value ) {
602 					if ( key === "click" ) {
603 						return;
604 					}
605 					if ( key in attrFn ) {
606 						button[ key ]( value );
607 					} else {
608 						button.attr( key, value );
609 					}
610 				});
611 				if ($.fn.omButton) {
612 					button.omButton();
613 				}
614 			});
615 			uiDialogButtonPane.appendTo(self.uiDialog);
616 		}

619 	_makeDraggable: function() {
620 		var self = this,
621 			options = self.options,
622 			doc = $(document),
623 			heightBeforeDrag = 0;
624 
625 		function filteredUi(ui) {
626 			return {
627 				position: ui.position,
628 				offset: ui.offset
629 			};
630 		}
631 
632 		self.uiDialog.omDraggable({
633 			cancel: '.om-dialog-content, .om-dialog-titlebar-close',
634 			handle: '.om-dialog-titlebar',
635 			containment: 'document',
636 			cursor: 'move',
637 			onStart: function(ui, event) {
638 				heightBeforeDrag = options.height === "auto" ? "auto" : $(this).height();
639 				$(this).height($(this).height()).addClass("om-dialog-dragging");
640 				self._trigger('onDragStart', filteredUi(ui), event);
641 			},
642 			onDrag: function(ui, event) {
643 				self._trigger('onDrag', filteredUi(ui), event);
644 			},
645 			onStop: function(ui, event) {
646 				options.position = [ui.position.left - doc.scrollLeft(),
647 					ui.position.top - doc.scrollTop()];
648 				$(this).removeClass("om-dialog-dragging").height(heightBeforeDrag);
649 				self._trigger('onDragStop', filteredUi(ui), event);
650 				$.om.omDialog.overlay.resize();
651 			}
652 		});
653 	},

655 	_makeResizable: function(handles) {
656 		handles = (handles === undefined ? this.options.resizable : handles);
657 		var self = this,
658 			options = self.options,
659 			// .ui-resizable has position: relative defined in the stylesheet
660 			// but dialogs have to use absolute or fixed positioning
661 			position = self.uiDialog.css('position'),
662 			resizeHandles = (typeof handles === 'string' ?
663 				handles	:
664 				'n,e,s,w,se,sw,ne,nw'
665 			);
666 
667 		function filteredUi(ui) {
668 			return {
669 				originalPosition: ui.originalPosition,
670 				originalSize: ui.originalSize,
671 				position: ui.position,
672 				size: ui.size
673 			};
674 		}
675 
676 		self.uiDialog.omResizable({
677 			cancel: '.om-dialog-content',
678 			containment: 'document',
679 			alsoResize: self.element,
680 			maxWidth: options.maxWidth,
681 			maxHeight: options.maxHeight,
682 			minWidth: options.minWidth,
683 			minHeight: self._minHeight(),
684 			handles: resizeHandles,
685 			start: function(event, ui) {
686 				$(this).addClass("om-dialog-resizing");
687 				self._trigger('onResizeStart', event, filteredUi(ui));
688 			},
689 			resize: function(event, ui) {
690 				self._trigger('onResize', event, filteredUi(ui));
691 			},
692 			stop: function(event, ui) {
693 				$(this).removeClass("om-dialog-resizing");
694 				options.height = $(this).height();
695 				options.width = $(this).width();
696 				self._trigger('onResizeStop', event, filteredUi(ui));
697 				$.om.omDialog.overlay.resize();
698 			}
699 		})
700 		.css('position', position)
701 		.find('.om-resizable-se').addClass('om-icon om-icon-grip-diagonal-se');
702 	},

715 		var myAt = [],
716 			offset = [0, 0],
717 			isVisible;
718 
719 		if (position) {
720 			// deep extending converts arrays to objects in jQuery <= 1.3.2 :-(

722 	//			myAt = $.isArray(position) ? position : position.split(' ');

735 				});

750 		isVisible = this.uiDialog.is(':visible');
751 		if (!isVisible) {
752 			this.uiDialog.show();
753 		}

756 			.css({ top: 0, left: 0 })

758 		if (!isVisible) {

787 	_setOption: function(key, value){
788 		var self = this,
789 			uiDialog = self.uiDialog;
790 
791 		switch (key) {
792 			case "buttons":
793 				self._createButtons(value);
794 				break;
795 			case "closeText":
796 				// ensure that we always pass a string
797 				self.uiDialogTitlebarCloseText.text("" + value);
798 				break;
799 			case "dialogClass":
800 				uiDialog
801 					.removeClass(self.options.dialogClass)
802 					.addClass(uiDialogClasses + value);
803 				break;
804 			case "disabled":
805 				if (value) {
806 					uiDialog.addClass('om-dialog-disabled');
807 				} else {
808 					uiDialog.removeClass('om-dialog-disabled');
809 				}
810 				break;
811 			case "draggable":
812 				var isDraggable = uiDialog.is( ":data(draggable)" );
813 				if ( isDraggable && !value ) {
814 					uiDialog.omDraggable( "destroy" );
815 				}
816 				
817 				if ( !isDraggable && value ) {
818 					self._makeDraggable();
819 				}
820 				break;
821 			case "position":
822 				self._position(value);
823 				break;
824 			case "resizable":
825 				// currently resizable, becoming non-resizable
826 				var isResizable = uiDialog.is( ":data(resizable)" );
827 				if (isResizable && !value) {
828 					uiDialog.omResizable('destroy');
829 				}
830 
831 				// currently resizable, changing handles
832 				if (isResizable && typeof value === 'string') {
833 					uiDialog.omResizable('option', 'handles', value);
834 				}
835 
836 				// currently non-resizable, becoming resizable
837 				if (!isResizable && value !== false) {
838 					self._makeResizable(value);
839 				}
840 				break;
841 			case "title":
842 				// convert whatever was passed in o a string, for html() to not throw up
843 				$(".om-dialog-title", self.uiDialogTitlebar).html("" + (value || ' '));
844 				break;
845 		}
846

850 	_size: function() {
851 		/* If the user has resized the dialog, the .ui-dialog and .ui-dialog-content
852 		 * divs will both have width and height set, so we need to reset them
853 		 */
854 		var options = this.options,
855 			nonContentHeight,
856 			minContentHeight,
857 			isVisible = this.uiDialog.is( ":visible" );
858 
859 		// reset content sizing
860 		this.element.show().css({
861 			width: 'auto',
862 			minHeight: 0,
863 			height: 0
864 		});
865 
866 		if (options.minWidth > options.width) {
867 			options.width = options.minWidth;
868 		}
869 
870 		// reset wrapper sizing
871 		// determine the height of all the non-content elements
872 		nonContentHeight = this.uiDialog.css({
873 				height: 'auto',
874 				width: options.width
875 			})
876 			.height();
877 		minContentHeight = Math.max( 0, options.minHeight - nonContentHeight );
878 		
879 		if ( options.height === "auto" ) {
880 			// only needed for IE6 support
881 			if ( $.support.minHeight ) {
882 				this.element.css({
883 					minHeight: minContentHeight,
884 					height: "auto"
885 				});
886 			} else {
887 				this.uiDialog.show();
888 				var autoHeight = this.element.css( "height", "auto" ).height();
889 				if ( !isVisible ) {
890 					this.uiDialog.hide();
891 				}
892 				this.element.height( Math.max( autoHeight, minContentHeight ) );
893 			}
894 		} else {
895 			this.element.height( Math.max( options.height - nonContentHeight, 0 ) );
896 		}
897 
898 		if (this.uiDialog.is(':data(resizable)')) {
899 			this.uiDialog.omResizable('option', 'minHeight', this._minHeight());
900 		}
901 	}
