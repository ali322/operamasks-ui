191 	// http://docs.jquery.com/Plugins/Validation/validate
192 	validate: function( options ) {
193 
194 		// if nothing is selected, return nothing; can't chain anyway
195 		if (!this.length) {
196 			options && options.debug && window.console && console.warn( "nothing selected, can't validate, returning nothing" );
197 			return;
198 		}
199 
200 		// check if a validator for this form was already created
201 		var validator = $.data(this[0], 'validator');
202 		if ( validator ) {
203 			return validator;
204 		}
205 
206 		validator = new $.validator( options, this[0] );
207 		$.data(this[0], 'validator', validator);
208 
209 		if ( validator.settings.onsubmit ) {
210 
211 			// allow suppresing validation by adding a cancel class to the submit button
212 			this.find("input, button").filter(".cancel").click(function() {
213 				validator.cancelSubmit = true;
214 			});
215 
216 			// when a submitHandler is used, capture the submitting button
217 			if (validator.settings.submitHandler) {
218 				this.find("input, button").filter(":submit").click(function() {
219 					validator.submitButton = this;
220 				});
221 			}
222 
223 			// validate the form on submit
224 			this.submit( function( event ) {
225 				if ( validator.settings.debug )
226 					// prevent form submit to be able to see console output
227 					event.preventDefault();
228 
229 				function handle() {
230 					if ( validator.settings.submitHandler ) {
231 						if (validator.submitButton) {
232 							// insert a hidden input as a replacement for the missing submit button
233 							var hidden = $("<input type='hidden'/>").attr("name", validator.submitButton.name).val(validator.submitButton.value).appendTo(validator.currentForm);
234 						}
235 						validator.settings.submitHandler.call( validator, validator.currentForm );
236 						if (validator.submitButton) {
237 							// and clean up afterwards; thanks to no-block-scope, hidden can be referenced
238 							hidden.remove();
239 						}
240 						return false;
241 					}
242 					return true;
243 				}
244 
245 				// prevent submit for invalid forms or custom submit handlers
246 				if ( validator.cancelSubmit ) {
247 					validator.cancelSubmit = false;
248 					return handle();
249 				}
250 				if ( validator.form() ) {
251 					if ( validator.pendingRequest ) {
252 						validator.formSubmitted = true;
253 						return false;
254 					}
255 					return handle();
256 				} else {
257 					validator.focusInvalid();
258 					return false;
259 				}
260 			});
261 		}
262 
263 		return validator;
264 	},
265 	// http://docs.jquery.com/Plugins/Validation/valid
266 	valid: function() {
267         if ( $(this[0]).is('form')) {
268             return this.validate().form();
269         } else {
270             var valid = true;
271             var validator = $(this[0].form).validate();
272             this.each(function() {
273 				valid &= validator.element(this);
274             });
275             return valid;
276         }
277     },
278 	// attributes: space seperated list of attributes to retrieve and remove
279 	removeAttrs: function(attributes) {
280 		var result = {},
281 			$element = this;
282 		$.each(attributes.split(/\s/), function(index, value) {
283 			result[value] = $element.attr(value);
284 			$element.removeAttr(value);
285 		});
286 		return result;
287 	},
288 	// http://docs.jquery.com/Plugins/Validation/rules
289 	rules: function(command, argument) {
290 		var element = this[0];
291 
292 		if (command) {
293 			var settings = $.data(element.form, 'validator').settings;
294 			var staticRules = settings.rules;
295 			var existingRules = $.validator.staticRules(element);
296 			switch(command) {
297 			case "add":
298 				$.extend(existingRules, $.validator.normalizeRule(argument));
299 				staticRules[element.name] = existingRules;
300 				if (argument.messages)
301 					settings.messages[element.name] = $.extend( settings.messages[element.name], argument.messages );
302 				break;
303 			case "remove":
304 				if (!argument) {
305 					delete staticRules[element.name];
306 					return existingRules;
307 				}
308 				var filtered = {};
309 				$.each(argument.split(/\s/), function(index, method) {
310 					filtered[method] = existingRules[method];
311 					delete existingRules[method];
312 				});
313 				return filtered;
314 			}
315 		}
316 
317 		var data = $.validator.normalizeRules(
318 		$.extend(
319 			{},
320 			$.validator.metadataRules(element),
321 			$.validator.classRules(element),
322 			$.validator.attributeRules(element),
323 			$.validator.staticRules(element)
324 		), element);
325 
326 		// make sure required is at front
327 		if (data.required) {
328 			var param = data.required;
329 			delete data.required;
330 			data = $.extend({required: param}, data);
331 		}
332 
333 		return data;
334 	}

354 $.validator.format = function(source, params) {
355 	if ( arguments.length == 1 )
356 		return function() {
357 			var args = $.makeArray(arguments);
358 			args.unshift(source);
359 			return $.validator.format.apply( this, args );
360 		};
361 	if ( arguments.length > 2 && params.constructor != Array  ) {
362 		params = $.makeArray(arguments).slice(1);
363 	}
364 	if ( params.constructor != Array ) {
365 		params = [ params ];
366 	}
367 	$.each(params, function(i, n) {
368 		source = source.replace(new RegExp("\\{" + i + "\\}", "g"), n);
369 	});
370 	return source;
371 };
