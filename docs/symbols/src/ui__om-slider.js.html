 15      * @name omSlider

 17      * <b>特点：</b><br/>

220         _runSlideEffect : function(slider, index, effect){
221             var _self = this,
222             	vars = this._getSliderVars(slider),
223                 $container = slider.find('ul.om-slider-content:first'),
224                 $item = $container.children(),opts = this.options,
225                 top = 0,
226                 left = 0;
227             var effectnow = effect ? effect : opts.effect;
228             if(opts.effect == 'random'){
229                 $container.addClass('om-slider-effect-'+ effectnow);
230             }
231             $container.stop();
232             if(effectnow == 'slide-v' || effectnow == 'slide-h'){
233                 var slideV = (effectnow == 'slide-v');
234                 if(opts.effect == 'random'){
235                     var cur = 0;
236                     $item.each(function(n){
237                         if(n !== vars.currentSlide){
238                             cur -= ( slideV? $(this).height() : $(this).width() );
239                         }else{
240                             return false;
241                         }
242                     });
243                     $container.css(slideV? "top":"left" , cur);
244                 }
245                 $item.each(function(n){
246                     if(n == index) return false;
247                     slideV? top -= $(this).height() : left -= $(this).width();
248                 });
249             } else{
250                 return false;
251             }
252             vars.running = true;
253             $container.animate({top:top,left:left},opts.animSpeed,function(){
254                 vars.running = false;
255                 _self._trigger("onAfterSlide",null,index);
256             });
257         },
258         
259         _runFadeEffect : function(slider,index){
260             var _self = this,
261             	vars = this._getSliderVars(slider),
262                 items = slider.find('ul.om-slider-content:first').children(),
263                 opts = this.options;
264             items.each(function(n){
265                 var $child = $(this);
266                 if(n == index){
267                     vars.running = true;
268                     $child.fadeIn(opts.animSpeed,function(){
269                         vars.running = false;
270                         _self._trigger("onAfterSlide",null,index);
271                     });
272                 } else if(n == vars.currentSlide){
273                     $child.fadeOut(opts.animSpeed);
274                 }
275             });
276         }, 
277         
278         _runNoEffect : function(slider,index){
279             var _self = this,
280             	vars = this._getSliderVars(slider),
281                 items = slider.find('ul.om-slider-content:first').children();
282             items.each(function(n){
283                 var $child = $(this);
284                 if(n == index){
285                     $child.show();
286                     _self._trigger("onAfterSlide",null,index);
287                 } else if(n == vars.currentSlide){
288                     $child.hide();
289                 }
290             });

296         _toggleControlNav : function(slider, index){
297             var vars = this._getSliderVars(slider);
298             var opts = this.options;
299             var parent = slider;
300             // 如果是自定义导航条，则controlNav的位置在slider外面，所以从body下面找controlNav

302                 parent = $('body');
303             }

305             navItems.each(function(n){
306                 $(this).toggleClass(opts.activeNavCls,n==index);
307             });

310         /**
311          * 切换至指定面板，index从0开始
312          */
313         _slideTo : function(slider, index){
314             var vars = this._getSliderVars(slider),
315             $container = slider.find('ul.om-slider-content:first');
316             var opts = this.options, self = this;
317             if(isNaN(index) || index < 0 || index >= vars.totalSlides){
318                 return;
319             }
320             if (this._trigger("onBeforeSlide",null,index) === false) {
321                 return false;
322             }
323             if(opts.effect == 'random'){
324                 var array=['fade','slide-h','slide-v'];
325                 var effect = array[Math.floor(Math.random()*3)];
326                 $container.removeClass().addClass('om-slider-content');
327                 $container.removeAttr('style');
328                 $container.find('li').removeAttr('style');
329                 if(effect == 'slide-h' || effect == 'slide-v'){
330                     self._runSlideEffect(slider, index, effect);
331                 } else {
332                     self._runFadeEffect(slider, index);
333                 }
334             }else if(opts.effect == 'slide-h' || opts.effect == 'slide-v'){
335                 self._runSlideEffect(slider, index);
336             } else if(opts.effect == 'fade' || opts.effect === true){
337                 self._runFadeEffect(slider, index);
338             } else{
339                 self._runNoEffect(slider, index);
340             }
341             
342             if(vars.controlNav){
343                 self._toggleControlNav(slider, index);
344             }
345             vars.currentSlide = index;
346             return slider;
347         },

349         _next : function(slider){
350             var vars = this._getSliderVars(slider),
351                 next_index = 0;
352             if(vars.currentSlide+2 <= vars.totalSlides){
353                 next_index = vars.currentSlide + 1;
354             }
355             return this._slideTo(slider,next_index);
356         }, 
357         _prev : function(slider){
358             var vars = this._getSliderVars(slider),
359                 index = vars.totalSlides - 1;
360             if(vars.currentSlide != 0){
361                 index = vars.currentSlide - 1;
362             }
363             return this._slideTo(slider,index);
364         }, 
365         _processDirectionNav : function(slider){
366             var vars = this._getSliderVars(slider),
367                 directionNav = $('<div class="om-slider-directionNav">').appendTo(slider),
368                 self = this;
369             $('<a class="om-slider-prevNav"></a>').appendTo(directionNav).click(function(){
370                 if(vars.running)return false;
371                 self._prev(slider);
372             });
373             $('<a class="om-slider-nextNav"></a>').appendTo(directionNav).click(function(){
374                 if(vars.running)return false;
375                 self._next(slider);
376             });
377             slider.hover(function(){
378                 directionNav.show();
379             },function(){
380                 directionNav.hide();
381             });
382         }, 
383         _processControlNav : function(slider){
384             var vars = this._getSliderVars(slider),
385             	opts = this.options,
386             	self = this,
387             	n;
388             if(opts.controlNav === true || opts.controlNav === 'classical'){
389                 var $nav = $('<ul class="om-slider-nav-classical"></ul>');
390                 vars.controlNav = '.om-slider-nav-classical';
391                 for(n=0;n<vars.totalSlides;n++){
392                     var $navItem = $('<li>'+(n+1)+'</li>');
393                     $navItem.data('sid',n);
394                     var hTimer = 0;
395                     $navItem.click(function(){
396                         //if(vars.running)return false;
397                         self._slideTo(slider,$(this).data('sid'));
398                     });
399                     $navItem.hover(function(){
400                         if(vars.running || vars.stop)return false;
401                         var _self = $(this);
402                         if(_self.hasClass(opts.activeNavCls))return false;
403                         //vars.slideTimer:用于组件销毁
404                         vars.slideTimer = hTimer = setTimeout(function(){self._slideTo(slider,_self.data('sid'));},opts.delay);
405                     },function(){
406                         clearTimeout(hTimer);
407                     });
408                     $nav.append($navItem);
409                 }
410                 slider.append($nav);
411             } else if(opts.controlNav === 'dot'){
412                 var $nav = $('<div class="om-slider-nav-dot"></div>');
413                 vars.controlNav = '.om-slider-nav-dot';
414                 for(n=0;n<vars.totalSlides;n++){
415                     var $navItem = $('<a href="javascript:void(0)">'+(n+1)+'</a>');
416                     $navItem.data('sid',n);
417                     var hTimer = 0;
418                     $navItem.click(function(){
419                         //if(vars.running)return false;
420                         self._slideTo(slider,$(this).data('sid'));
421                     });                 
422                     $navItem.hover(function(){
423                         if(vars.running || vars.stop)return false;
424                         var _self = $(this);
425                         if(_self.hasClass(opts.activeNavCls))return false;
426                         vars.slideTimer = hTimer = setTimeout(function(){self._slideTo(slider,_self.data('sid'));},opts.delay);
427                     },function(){
428                         clearTimeout(hTimer);
429                     });
430                     $nav.append($navItem);
431                 }
432                 //$nav.insertAfter(slider);
433                 $nav.appendTo(slider).css({marginLeft:-1*$nav.width()/2});
434             } else{
435                 if($(opts.controlNav).length > 0){
436                     vars.controlNav = opts.controlNav;
437                     vars.customNav = true;
438                     var $nav = $(opts.controlNav);
439                     $nav.children().each(function(n){
440                         var $navItem = $(this);
441                         $navItem.data('sid',n);
442                         var hTimer = 0;
443                         $navItem.bind("click.omSlider" , function(){
444                             //if(vars.running)return false;
445                             self._slideTo(slider,$(this).data('sid'));
446                         });
447                         $navItem.bind("mouseover.omSlider" , function(){
448                         	if(vars.running || vars.stop)return false;
449                             var _self = $(this);
450                             if(_self.hasClass(opts.activeNavCls))return false;
451                             vars.slideTimer = hTimer = setTimeout(function(){self._slideTo(slider,_self.data('sid'));},opts.delay);
452                         }).bind("mouseout.omSlider" , function(){
453                         	clearTimeout(hTimer);
454                         });
455                     });
456                 }
457             }
458         },

460         _create : function() {
461             var timer = 0;
462             var $this = this.element;
463             var self = this;
464             var opts = this.options;
465             var vars = {
466                 currentSlide: 0,
467                 totalSlides: 0,
468                 running: false,
469                 paused: false,
470                 stop: false,
471                 controlNav: '.om-slider-nav-classical'
472             };
473             
474             $this.data('omSlider',$this).data('omSlider:vars',vars).data('omSlider:opts',opts).addClass('om-slider');
475             if(opts.startSlide > 0){
476                 vars.currentSlide = opts.startSlide; 
477             }
478             var kids = $this.children();
479             kids.wrapAll('<ul class="om-slider-content"></ul>').wrap('<li class="om-slider-item"></li>');
480             if(opts.effect == 'slide-v' || opts.effect == 'slide-h'){
481                 $this.find('.om-slider-content').addClass('om-slider-effect-'+opts.effect);
482             }
483             vars.totalSlides = kids.length;
484             this._processControlNav($this);
485             /**
486              * 内部方法。在effect="slide-h||slide-v"且startSlide>0的情况下需要在window.onload中执行。
487              */
488             function _initSlider(slider, startSlide){
489                 if(isNaN(startSlide) || startSlide < 0 || startSlide >= vars.totalSlides){
490                     return;
491                 }
492                 var $container = slider.find('ul.om-slider-content:first'),
493                     $item = $container.children(),
494                     top = 0,
495                     left = 0;
496                 if(opts.effect == 'slide-h'){
497                     $item.each(function(n){
498                         if(n == startSlide) return false;
499                         left -= $(this).width();
500                     });
501                     setTimeout(function(){$container.css({left:left,top:top});},0);
502                 } else if(opts.effect == 'slide-v'){
503                     $item.each(function(n){
504                         if(n == startSlide) return false;
505                         top -= $(this).height();
506                     });
507                     // 修复omSlider在omTabs的隐藏tab下设置样式无效的问题。只在IE7浏览器下发生。
508                     setTimeout(function(){$container.css({left:left,top:top});},0);
509                 } else{
510                     $container.children().eq(startSlide).show();
511                 }
512                 
513                 if(vars.controlNav){
514                     self._toggleControlNav(slider, startSlide);
515                 }
516                 if(opts.autoPlay){
517                 	//vars.autoPlayTimer:用于组件销毁
518                     vars.autoPlayTimer = timer = setInterval(function(){self._next(slider);},opts.interval);
519                 }
520                 if(opts.pauseOnHover){
521                     slider.hover(function(){
522                         vars.paused = true;
523                         clearInterval(timer);
524                     },function(){
525                         vars.paused = false;
526                         if(opts.autoPlay){
527                             vars.autoPlayTimer = timer = setInterval(function(){self._next(slider);},opts.interval);
528                         }
529                     });
530                 }
531                 if(opts.directionNav){
532                     self._processDirectionNav(slider);
533                 }
534             }
535             
536             if(opts.startSlide > 0 && (opts.effect == 'slide-h' || opts.effect == 'slide-v')){
537                 // 考虑到slide的动画效果下如果设置startSlide>0需要计算startSlide前面的面板的大小。
538                 // 如果面板里面包含img标签的话只有在图片下载完后(既window.onload事件)才能获得面板的大小。
539                 $(window).load(function(){
540                     _initSlider($this,opts.startSlide);
541                 });
542             } else{
543                 _initSlider($this,opts.startSlide);
544             } 
545         }

547     });
