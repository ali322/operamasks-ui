 15 (function($) {
 16     $.omWidget.addInitListener('om.omGrid',function(){
 17         var self = this,
 18         	ops = this.options,
 19             rdp = ops.rowDetailsProvider, //渲染行详情的function
 20             cm = ops.colModel,
 21             $thead = this.thead;
 22             
 23         if( !(rdp && cm.length>0) ){
 24             return;
 25         }
 26         
 27         ops._onResizableCallbacks.push(_resizeRowExpander);
 28         ops._onResizableStopCallbacks.push(_resizeRowExpander);
 29         
 30         $thead.find("th:first").before('<th align="center" axis="expenderCol" class="expenderCol" rowspan='
 31         								+($.isArray(cm[0])?cm.length:1)+'><div style="text-align: center; width: 14px;"></div></th>');
 32         var autoExpandColIndex = -1,
 33             allColsWidth = 0;
 34             
 35         cm = this._getColModel();//确保此时cm为最简单形式的colModel
 36         $(cm).each(function(i){
 37         	if(cm[i].width == 'autoExpand'){
 38                 autoExpandColIndex = i;
 39             }else{
 40                 allColsWidth += cm[i].width;
 41             }
 42         });
 43 
 44         var expenderWidth = $thead.find('th[axis="expenderCol"]').width();
 45         if(autoExpandColIndex != -1){ //说明有某列要自动扩充
 46         	$thead.find('th[axis="col'+autoExpandColIndex+'"] >div')
 47         		.css("width" , "-="+expenderWidth);
 48         }else if(ops.autoFit){
 49             var percent = expenderWidth/allColsWidth;
 50             $thead.find('th[axis^="col"] >div')
 51             	.each(function(i){
 52             		$(this).css('width' , "-=" + parseInt(cm[i].width*percent));
 53             	});
 54         }
 55         
 56         var colCount= this._getHeaderCols().size();//总共列数
 57         this.tbody.delegate('td.expenderCol >div','click',function(event){
 58             var _this = $(this),
 59                 $row = _this.closest('tr'),
 60                 $next = $row.next('tr');
 61                 
 62             if($next.hasClass('rowExpand-rowDetails')){ //已经构造过了，直接显示/隐藏
 63                 $next.toggle();
 64             }else{ //没有构造过，则构造并显示
 65                 var rowIndex = self._getTrs().index($row);
 66                     rowData = self._getRowData(rowIndex),
 67                     rowDetails = rdp? rdp(rowData,rowIndex):' ';
 68                 $row.after('<tr class="rowExpand-rowDetails"><td colspan="'+(colCount-1)+'"><div class="rowExpand-rowDetails-content">'+rowDetails+'</div></td></tr>');
 69             }
 70             
 71             _this.toggleClass('rowExpand-expanded').parent()
 72             	.attr('rowspan' , _this.hasClass('rowExpand-expanded')? 2:1);
 73 
 74 			_resizeRowExpander.call(self);
 75             return false; //不触发onRowSelect和onRowClick事件
 76         });
 77     });
 78     
 79     function _resizeRowExpander(){
 80     	var $grid = this.element; 
 81     	var $rowExpander = $grid.find("tr.rowExpand-rowDetails");
 82     	
 83     	if($rowExpander.length == 0){
 84     		return ;
 85     	}
 86     	
 87     	//计算行详情的宽度
 88     	var $ht = this.hDiv.find("table");
 89     	var width = $ht.width() - $ht.find("th.expenderCol").outerWidth() - parseInt($rowExpander.find("> td").css("border-right-width"));
 90 
 91     	$rowExpander.each(function(){
 92     		$(this).find(".rowExpand-rowDetails-content").outerWidth(width);
 93     	});
 94     }
 95 })(jQuery);
