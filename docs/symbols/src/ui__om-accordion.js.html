435     /**
436      * 修正索引，返回值为-1,0,1,2,...,len-1
437      */
438     _correctIndex: function(index){
439     	var $acc = this.element,
440         	panels = $.data(this.element , "panels"),
441         	panel = $acc.children().find(">div#"+index),
442         	len = panels.length,
443         	oldIndex = index;
444         index = panel.length ? $acc.find(">.om-panel").index(panel.parent()) : index;
445         index = index==-1 ? oldIndex : index;
446         //如果id找不到，则认为是索引，进行修正
447         var r = parseInt(index);
448         r = (isNaN(r) && '0' || r)-0;
449         return len==0 || (r === -1 && this.options.collapsible !== false) ? -1: (r<0? 0 : (r>=len?len-1 : r));
450     },
451     _panelCreateCollapse: function(len , index){
452     	var $acc = this.element,
453     		options = this.options,
454     		panel,
455     		num;
456     	if(options.active === -1 && options.collapsible === true){
457     		return true;
458     	}else{
459 			panel = $acc.find(">div#"+options.active);
460 			num = $acc.children().index(panel);
461 			num = (num == -1? options.active : num); 		
462     		var r = parseInt(num);
463     		r = (isNaN(r) && '0' || r)-0;
464     		r = r<0? 0 : (r>=len?len-1 : r);
465     		return r !== index;  
466     	}

468     _renderPanels: function () {
469         var $acc = this.element,
470         	self = this,
471             panels = [],
472             options = this.options,
473         	$headers = $acc.find(options.header),
474         	cfg = [],
475         	first;
476         $headers.find("a").each(function(n){
477             var href  = this.getAttribute('href', 2);
478             var hrefBase = href.split( "#" )[ 0 ],
479                 baseEl;
480             if ( hrefBase && ( hrefBase === location.toString().split( "#" )[ 0 ] ||
481                     ( baseEl = $( "base" )[ 0 ]) && hrefBase === baseEl.href ) ) {
482                 href = this.hash;
483                 this.href = href;
484             }
485             var $anchor = $(this);
486             cfg[n] = {
487                     title : $anchor.text(),
488                     iconCls: $anchor.attr("iconCls"),
489                     onExpand : function(event) {
490                     	self._trigger("onActivate",event,n);
491                     },
492                     tools:[{id:"collapse" , handler: function(panel , event){
493                     	clearInterval(options.autoInterId);
494 	                    self._activate(n , true);
495 	                    self._setAutoInterId(self);
496 	                    event.stopPropagation();
497                     }}],
498                     onCollapse : function(event) {
499                     	self._trigger("onCollapse",event,n);
500                     },
501                     onSuccess : function() {
502                     	self.resize();
503                     },
504                     onError : function() {
505                     	self.resize();
506                     }
507             };
508             var target = $('>' + href, $acc);
