 15 (function($) {
 16 	/**
 17 	 * 分页导航按钮样式与类别
 18 	 */
 19 	var pagerNvg = [
 20 		{cls:"pFirst" , type:"first"},
 21 		{cls:"pPrev" , type:"prev"},
 22 		{cls:"pNext" , type:"next"},
 23 		{cls:"pLast" , type:"last"},
 24 		{cls:"pReload" , type:"reload"}
 25 	];	
 26 	function needChange(self , type){
 27 		var oldPage = self._oldPage,
 28 		    nowPage = self.pageData.nowPage;
 29 		if("input" === type){
 30 			return oldPage != $('.pControl input', self.element.closest('.om-grid')).val();
 31 		}else if("reload" === type){
 32 			return true;
 33 		}else{
 34 			return oldPage != nowPage;
 35 		}
 36 	}
 37 	
 38     $.omWidget.addInitListener('om.omGrid',function(){
 39         var self = this,
 40             cm = this._getColModel(),
 41             tds = this._getHeaderCols().filter("[axis^='col']"),
 42             $pDiv = this.pDiv;
 43         
 44         $(tds).each(function(index){
 45             var sortFn=cm[index].sort;
 46             if(sortFn){
 47                 var _this=$(this).click(function(){
 48                     var sortCol = cm[index].name;
 49                     var removeClass = _this.hasClass('asc')?'asc'
 50                                     : _this.hasClass('desc')?'desc'
 51                                     : null;
 52                     var sortDir=(removeClass=='asc'?'desc':'asc');
 53                     tds.removeClass('asc desc');
 54                     _this.addClass(sortDir);
 55                     
 56                     var extraData = self._extraData;
 57                     delete extraData.sortBy;
 58                     delete extraData.sortDir;
 59                     switch(sortFn){
 60                         case 'serverSide':
 61                             extraData.sortBy=sortCol;
 62                             extraData.sortDir=sortDir;
 63                             self.reload();
 64                             return;
 65                         case 'clientSide':
 66                             sortFn=function(obj1,obj2){
 67                                 var v1=obj1[sortCol],v2=obj2[sortCol];
 68                                 return v1==v2?0:v1>v2?1:-1;
 69                             };
 70                             break;
 71                         default:
 72                             // do nothing,keep sortFn==cm[index].sort
 73                     }
 74                     var datas = self.pageData.data;
 75                     if(removeClass==null){//从未排序变成升序排列
 76                         datas.rows=datas.rows.sort(sortFn);
 77                     }else{//升序变成降序，或降序变成升序，只需要反转数据即可
 78                         datas.rows=datas.rows.reverse();
 79                     }
 80                     self.refresh();
 81                 });
 82                 _this.children().first().append('<img class="om-grid-sortIcon" src="data:image/gif;base64,R0lGODlhAQABAID/AMDAwAAAACH5BAEAAAAALAAAAAABAAEAAAICRAEAOw=="></img>');
 83             }
 84         });
 85         //监听分页条按钮的事件，如果为客户端排序，每次都要清除掉排序的状态
 86         for(var i=0,len=pagerNvg.length; i<len; i++){
 87         	(function(i){
 88         		$pDiv.find("."+pagerNvg[i].cls).click(function(){
 89 					var change = needChange(self , pagerNvg[i].type);
 90 					tds.each(function(index){
 91 						var $headerCol = $(this);
 92 						if(change && ($headerCol.hasClass('asc') || $headerCol.hasClass('desc')) &&  "serverSide" !== cm[index].sort){
 93 							$headerCol.removeClass('asc desc');
 94 						}
 95 					});
 96 				});	 
 97         	})(i);
 98 		}
 99         $('.pControl input', $pDiv).keydown(function(e){
100         	var change = needChange(self , "input");
101         	if (e.keyCode == $.om.keyCode.ENTER) {
102         		tds.each(function(index){
103 					var $headerCol = $(this);
104 					if(change && ($headerCol.hasClass('asc') || $headerCol.hasClass('desc')) &&  "serverSide" !== cm[index].sort){
105 						$headerCol.removeClass('asc desc');
106 					}
107 				});
108         	}
109         });
110         
111         /**
112 	     * 清空omGrid的排序状态。比如先在第一列上进行了降序排列，以后每次取数时都是按这个降序排列条件来取数，如果要清空排序条件，调用本方法即可，调用后再次取数时就跟从未进行过排序一样。<br/>
113 	     * <b>注意：此方法仅清空排序状态，并不立即取数。</b>
114 	     * @function
115 	     * @example
116 	     *  $('.selector').omGrid('clearSort');
117 	     */
118 	    this.clearSort=function(){
119 	        var extraData = this._extraData;
120 	        extraData.sortBy = undefined;
121 	        extraData.sortDir = undefined;
122 	        this._getHeaderCols().removeClass('asc desc');
123 	        //$('tr:first th[axis^="col"]',this.thead).removeClass('asc desc');
124 	    };
125     });
126 })(jQuery);
