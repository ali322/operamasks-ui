 16     // 设置小数精度
 17     var fixPrecision = function(value, c, p) {
 18         var v = value.indexOf(".");       
 19         if (isNaN(value) && value != ".") {
 20             for (; isNaN(value);) {
 21                 value = value.substring(0, value.length - 1);
 22             }
 23         }
 24         if(!p.allowNegative && value.indexOf("-")!= -1){
 25         	var array=value.split("-");
 26         	value=array.join("");
 27         }
 28         if(!p.allowDecimals&&v!=-1 || value.charAt(value.length-1)==='.'){
 29             return value.substring(0, v);
 30          }
 31         if(v!=-1){
 32             value=value.substring(0,v+p.decimalPrecision+1);
 33         }
 34         return value.length > 0 ? parseFloat(value) : "";
 35     };

 37     /** 
 38      * @name omNumberField
 39      * @class 数字输入框组件，只能输入数字，字符自动过滤掉。<br/>
 40      * @constructor
 41      * @description 构造函数. 
 42      * @param p 标准config对象：{}
 43      * @example
 44      * $('numberFielddiv').omNumberField({decimalPrecision:3});
 45      */
 46     $.omWidget("om.omNumberField", {
 47         options: /** @lends omNumberField.prototype */ 
 48         {
 49             /**
 50              * 是否允许输入小数。
 51              * @default true
 52              * @type Boolean
 53              * @example
 54              * $('#input').omNumberField({allowDecimals:true});
 55              */
 56             allowDecimals: true,  //是否允许输入小数
 57             /**
 58              * 是否允许输入负数。
 59              * @default true
 60              * @type Boolean
 61              * @example
 62              * $('#input').omNumberField({allowNegative:true});
 63              */
 64             allowNegative: true,  //是否允许输入负数
 65             /**
 66              * 精确到小数点后几位。
 67              * @default 2
 68              * @type Number
 69              * @example
 70              * $('#input').omNumberField({decimalPrecision:2});
 71              */
 72             decimalPrecision: 2, //精确到小数点后几位
 73             /**
 74              * 是否禁用组件。
 75              * @default false
 76              * @type Boolean
 77              * @example
 78              * $('#input').omNumberField({disabled:true});
 79              */
 80             disabled: false,
 81             /**
 82              * 在输入框失去焦点时触发的方法。
 83              * @event
 84              * @param value 当前输入框的值
 85              * @param event jQuery.Event对象
 86              * @default emptyFn
 87              * @example
 88              * $('#input').omNumberField({onBlur:function(value,event){alert('now the value is'+value);}});
 89              */
 90             onBlur: function(value){},
 91             /**
 92              * 是否只读。
 93              * @default false
 94              * @type Boolean
 95              * @example
 96              * $('#input').omNumberField({readOnly:true});
 97              */
 98             readOnly: false            
 99         },
100 
101         _create : function() {
102             // 允许输入的字符
103             var options = this.options,
104             	self = this;
105             this.element.addClass("om-numberfield om-widget om-state-default om-state-nobg")
106             			.css("ime-mode" , "disabled");
107 			
108             this.element.keypress(function(e) {
109                 if (e.which == null && (e.charCode != null || e.keyCode != null)) {
110                     e.which = e.charCode != null ? e.charCode : e.keyCode;
111                 }
112                 var k = e.which;
113                 if (k === 8 || (k == 46 && e.button == -1) || k === 0) {
114                     return;
115                 }
116                 var character = String.fromCharCode(k);
117                 $.data(this,"character",character);
118                 var allowed = $.data(this, "allowed");
119                 if (allowed.indexOf(character) === -1||($(this).val().indexOf("-") !== -1 && character == "-")
120                         || ($(this).val().indexOf(".") !== -1 && character == ".")) {
121                     e.preventDefault();
122                 }
123             }).focus(function(){
124             	$(this).addClass('om-state-focus');
125             }).blur(function(e){
126                 $(this).removeClass('om-state-focus');
127             	var character = $.data(this,"character");
128                 this.value=fixPrecision(this.value, character, options);
129                 self._trigger("onBlur",e,this.value);
130             }).keydown(function(e){
131             	self._checkLast(this);
132             	
133             	//Chrome并不支持css属性ime-mode,无法阻止拼音输入，但当使用输入法时，事件的e.which===229恒成立.
134             	if(229 === e.which){
135             		e.preventDefault();
136             	}
137             }).keyup(function(e){//在Chrome中文输入法下，输入  ，。等字符不会触发input框的keypress事件
138             	self._checkLast(this);
139             }).bind('cut paste',function(e){
140             	return false;
141             });
142         },
143 		
144         _init : function() {
145             var $ele = this.element,
146                 opts = this.options;
147             
148             if (typeof opts.disabled !== "boolean") {
149                 opts.disabled = $ele.attr("disabled");
150             }
151 
152             if (opts.readOnly) {
153                 $ele.attr("readonly","readonly");
154             }
155 
156             var character = $.data($ele[0], "character");
157             
158             this._buildAllowChars();
159             
160             if (opts.disabled) {
161                 this.disable();
162             } else {
163                 this.enable();
164             }
165         },
166         
167 		_checkLast : function(self){
168 			var v = self.value,
169         		len = v.length;
170         	if(v && $.data(self,"allowed").indexOf(v.charAt(len-1))===-1
171         		|| v.indexOf('.') != v.lastIndexOf('.')
172         		|| v.indexOf('-') != v.lastIndexOf('-')){
173         		self.value = v = v.substring(0 , (len--)-1);
174         	}
175 		},
176 		
177         _buildAllowChars : function() {
178             var allowed = "0123456789";
179 
180             // 允许输入的字符
181             if (this.options.allowDecimals) {
182                 allowed = allowed + ".";
183             }
184             if (this.options.allowNegative) {
185                 allowed = allowed + "-";
186             }
187             if (this.options.readOnly) {
188                 allowed = "";
189             }
190             $.data(this.element[0], "allowed", allowed);
191         },
192         /**
193          * 禁用组件。
194          * @name omNumberField#disable
195          * @function
196          * @example
197          * $('#input').omNumberField("disable")
198          */
199         disable : function() {
200             this.element.attr("disabled", true)
201                     .addClass("om-numberfield-disabled");
202         },
203         /**
204          * 启用组件。
205          * @name omNumberField#enable
206          * @function
207          * @example
208          * $('#input').omNumberField("enable")
209          */
210         enable : function() {
211             this.element.attr("disabled", false)
212                     .removeClass("om-numberfield-disabled");
213         }
214     });
