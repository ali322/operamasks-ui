309 		_create: function(){
310 		    this.element.addClass("om-panel-body om-widget-content")
311 		    	.wrap("<div class='om-widget om-panel'></div>");
312 		},
313 		_init: function(){
314 			var options = this.options,
315 				$body = this.element,
316 				$parent = $body.parent(),
317 				$header;
318 			this._renderHeader();
319 			$header = $body.prev();
320 			if(options.header === false){
321 		 		$body.addClass("om-panel-noheader");
322 		 	}
323 			this._bindEvent();
324 		 	this._resize($parent);
325 		 	var headerHeight = options.header !== false? $header.outerHeight() : 0;
326 		 	if(options.collapsed !== false){
327 		 		"auto"!==options.height && $parent.height(headerHeight);		 		
328 		 		$body.hide();
329 		 		if(options.header !== false){
330 		 			$header.find(">.om-panel-tool >.om-panel-tool-collapse").removeClass("om-panel-tool-collapse")
331 		 				.addClass("om-panel-tool-expand");
332 		 		}
333 		 	}else{
334 		 		$body.show();
335 		 		"auto"!==options.height && $parent.height(headerHeight + $body.outerHeight());
336 		 		if(options.header !== false){
337 		 			$header.find(">.om-panel-tool >.om-panel-tool-expand").removeClass("om-panel-tool-expand")
338 		 				.addClass("om-panel-tool-collapse");
339 		 		}	
340 		 	}
341 		 	options.closed !== false? this._hide($parent) : this._show($parent);
342 		 	this.reload();
343 		},
344 		_hide: function($target){
345 			if("hidden" === this.options._closeMode){
346 				$target.hide();
347 			}else if("visibility" === this.options._closeMode){
348 				$target.addClass("om-helper-hidden-accessible");
349 			}
350 		},
351 		_show: function($target){
352 			if("hidden" === this.options._closeMode){
353 				$target.show();
354 			}else if("visibility" === this.options._closeMode){
355 				$target.removeClass("om-helper-hidden-accessible");
356 			}
357 		},
358 		_bindEvent: function(){
359 			var self = this,
360 				$body = this.element,
361 				options = this.options,
362 				header = $body.prev();
363 			if(options.collapsible !== false){
364 				header.click(function(event){
365 					if($(event.target).is(".om-panel-icon,.om-panel-title,.om-panel-header")){
366 						options.collapsed !== false? self.expand() : self.collapse();
367 					}
368 				}).find(".om-panel-tool-collapse , .om-panel-tool-expand")
369 				.click(function(){
370 					options.collapsed !== false? self.expand() : self.collapse();
371 				});
372 			}
373 			if(options.closable !== false){
374 				header.find(".om-panel-tool-close")
375 					.click(function(e){
376 						self.close();
377 					});				
378 			}
379 		},
380 		_renderHeader: function(){
381 			this.header && this.header.remove();
382 			if(this.options.header === false){
383 				return ;
384 			}
385 			var that = this,
386 				options = this.options,
387 				tools = options.tools,
388 				$header = this.header = $("<div class='om-panel-header'></div>").insertBefore(this.element);
389 			if(options._helpMsg){
390 				$header.parent().addClass('helpMsg');
391 			}
392 			if(options.iconCls){
393 				$("<div class='om-icon om-panel-icon'></div>").addClass(options.iconCls).appendTo($header);
394 			}
395 			$("<div class='om-panel-title'></div>").html(options.title).appendTo($header);
396 			$tool = $("<div class='om-panel-tool'></div>");
397 			if(options.collapsible !== false){
398 				$("<div class='om-icon om-panel-tool-collapse'></div>").appendTo($tool);	
399 			}
400 			//处理自定义头部右边的工具条
401 			if($.isArray(tools)){
402 				for(var i=0,len=tools.length; i<len; i++){
403 					var tool = tools[i],
404 						iconCls;
405 					if(iconCls = this._getInnerToolCls(tool.id)){
406 						$("<div class='om-icon'></div>").addClass(iconCls)
407 							.click(	function(event){
408 								tool.handler.call(this,that,event);
409 							}).appendTo($tool);
410 					}else if(typeof tool.iconCls === 'string'){
411 						$("<div class='om-icon'></div>").addClass(tool.iconCls)
412 							.click(	function(event){
413 								tool.handler.call(this,that,event);
414 							}).appendTo($tool);
415 					}else if($.isArray(tool.iconCls)){
416 						//这里必须要用内部匿名函数，因为hover中用到了tool，否则tool的值很可能已经被改掉了
417 						(function(tool){
418 							$("<div class='om-icon'></div>").addClass(tool.iconCls[0])
419 								.click(function(event){
420 									tool.handler.call(this,that,event);
421 								})
422 								.hover(function(){
423 									if(tool.iconCls[1]){
424 										$(this).toggleClass(tool.iconCls[1]);
425 									}
426 								}).appendTo($tool);
427 						})(tool);
428 					}
429 				}
430 			}else{
431 				try{
432 					$(tools).appendTo($tool);
433 				}catch(error){
434 					throw "bad format of jquery selector.";
435 				}
436 			}
437 			
438 			if(options.closable !== false){
439 				$("<div class='om-icon om-panel-tool-close'></div>").appendTo($tool);	
440 			}
441 			//处理内置工具按钮hover时的样式变换
442 			$tool.find(">div.om-icon").hover(
443 				function(){
444 					var self = this;
445 					$.each(innerToolCls , function(){
446 						if($(self).hasClass(this)){
447 							$(self).toggleClass(this+"-hover");
448 						}
449 					});
450 				}
451 			);
452 			$tool.appendTo($header);
453 		},
454 		/**
455 		 * 初始化panel,header,body的宽和高
456 		 */
457 	 	_resize: function($panel){
458 	 		var $body = this.element,
459 	 			$header = $body.prev(),
460 	 			$panel = $body.parent(),
461 	 			options = this.options;
462 	 		if(options.width == 'fit'){
463 	 			options.width = '100%';
464 	 			$panel.width('100%');
465 	 			$header.css("width" , "");
466 	 			$body.css("width" , "");
467 	 		}else if(options.width !== 'auto'){
468 				$panel.width(options.width);
469 				$header.outerWidth($panel.width());
470 				$body.outerWidth($panel.width());
471 	 		}else{
472 	 			var style = $body.attr("style");
473 	 			if(style && style.indexOf("width") !== -1){
474 	 				$panel.width($body.outerWidth());
475 	 				$header.outerWidth($body.outerWidth());
476 	 			}else{
477 	 				$panel.css("width" , "");
478 		 			$header.css("width" , "");
479 		 			$body.css("width" , "");
480 	 			}
481 	 		}
482 	 		if(options.height == 'fit'){
483 	 			options.height = '100%';
484 	 			$panel.height('100%');
485 	 			$body.outerHeight($panel.height()- (this.options.header!==false?$header.outerHeight():0) );	 
486 	 		}else if(options.height !== 'auto'){
487 				$panel.height(options.height);
488 				$body.outerHeight($panel.height()- (this.options.header!==false?$header.outerHeight():0) );	 
489 	 		}else{
490 	 			var style = $body.attr("style");
491 	 			if(style && style.indexOf("height") !== -1){
492 	 				$panel.height($header.outerHeight() + $body.outerHeight());
493 	 			}else{
494 	 				$panel.css("height" , "");
495 		 			$body.css("height" , "");
496 	 			}
497 	 		}
498 	 	},
499 	 	_getInnerToolCls: function(id){
500 	 		return $.inArray(id , innerToolId)!=-1? 'om-panel-tool-'+id : null;
501 	 	},
502 		_showLoadingMessage: function(){
503 			var options = this.options,
504 				$body = this.element,
505 				$loadMsg = $body.next(".om-panel-loadingMessage"),
506 				position = {
507 					width:$body.innerWidth(),
