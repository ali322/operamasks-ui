113 	function _getColIndex(self , header){
114 		var cm = self._getColModel();
115 		for(var i=0,len=cm.length; i<len; i++){
116 			if(cm[i].name == header.name){
117 				return i;
118 			}
119 		}
120 	}
121 	
122 	function _buildTh(content , cols , tmp){
123     	var j=0;
124 		content.push(tmp.replace(/\$/g , function(){
125 			return cols[j++];
126 		}));
127 	}

129 	/**
130 	 * 创建只包含最基本数据的colModel。也就是创建出没有多表头时的基本colModel。注意，生成的colModel是具有顺序的。
131 	 */
132 	function _buildBasicColModel(self){
133 		//基本colModel缓存引用
134 		self._colModel = [];
135 		var cm = self._getColModel(),
136 			matrix = [],
137 			realRowHeader = [],//colModel中非最后一行的具有真实数据意义的header缓存
138 			rows = 1,
139 			cols = 1,
140 			rowHeader,
141 			item,
142 			colIndex,//每个header从哪一列开始渲染
143 			len = cm.length;
144 
145 		for(var i=0; i<len; i++){
146 			matrix[i] = [];
147 		}
148 		for(var i=0,row=len; i<row; i++){
149 			rowHeader = cm[i];
150 			
151 			for(var j=0,col=rowHeader.length; j<col; j++){
152 				item = rowHeader[j];
153 				rows = item.rowspan || 1;
154 				cols = item.colspan || 1;
155 				
156 				colIndex = _checkMatrix(matrix , i , rows , cols);
157 				if(item.name){
158 					realRowHeader.push({header:item , colIndex:colIndex});
159 				}
160 			}
161 		}
162 		realRowHeader.sort(function(first , second){
163 			return first.colIndex - second.colIndex;
164 		});
165 		i=0;
166 		while(realRowHeader[i] && self._colModel.push(realRowHeader[i++].header));
167 	}
168 	
169 	function _checkMatrix(matrix , index , rows , cols){
170 		var i=0;
171 		while(matrix[index][i] && ++i);
172 		for(var j=index; j<index+rows; j++){
173 			for(var k=i; k<i+cols; k++){
174 				matrix[j][k] = true;
175 			}
176 		}
177 		return i;
178 	}
