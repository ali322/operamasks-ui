 14 ;(function($) {
 15     /**
 16      * @name omTooltip
 17      * @class 提示组件，当某个链接、表单、输入框等需要做功能向导提示的时候可以使用本组件。
 18      * <b>特点：</b><br/>
 19      * <ol>
 20      *      <li>提示内容多样化，可以是文字，html，页面dom节点还可以是异步加载的内容</li>
 21      *      <li>异步加载内容的时候支持懒加载</li>
 22      *      <li>最大限度节约资源，只有需要使用的时候才在页面加入dom元素</li>
 23      *      <li>支持所有标准html元素</li>
 24      * </ol>
 25      * 
 26      * <b>示例：</b><br/>
 27      * <pre>
 28      * <script type="text/javascript" >
 29      * $(document).ready(function() {
 30      *     $('#tip').omTooltip({
 31      *         trackMouse : true,
 32      *         html : '<div style="color:red;">欢迎使用omTooltip组件</div>'
 33      *     });
 34      * });
 35      * </script>
 36      * 
 37      * <input id="tip" type="submit" />
 38      * </pre>
 39      * 
 40      * @constructor
 41      * @description 构造函数. 
 42      * @param p 标准config对象：{}
 43      */
 44     $.omWidget('om.omTooltip', {
 45         options: /**@lends omTooltip# */{
 46             /**
 47              * 提示框的宽度。
 48              * @type Number 
 49              * @default 'auto' 
 50              * @example
 51              * $("#tip").omTooltip({width: 100}); 
 52              */
 53             width : 'auto',
 54             /**
 55              * 最小宽度，当提示内容很少的时候为了样式美观显示指定的宽度。
 56              * @type Number 
 57              * @default null 
 58              * @example
 59              * $("#tip").omTooltip({minWidth: 100}); 
 60              */
 61             minWidth : null,
 62             /**
 63              * 最大宽度，当需要根据内容自动适应宽度而又不想因为内容太多而导致太宽的情况，可以使用maxWidht设置一个最大宽度。
 64              * @type Number 
 65              * @default null 
 66              * @example
 67              * $("#tip").omTooltip({maxWidth: 200}); 
 68              */
 69             maxWidth : null,
 70             /**
 71              * 提示框的高度。
 72              * @type Number 
 73              * @default 'auto' 
 74              * @example
 75              * $("#tip").omTooltip({height: 100}); 
 76              */
 77             height : 'auto',
 78             /**
 79              * 最大高度，高度默认自适应，当达到最大高度之后会隐藏内容 。
 80              * @type Number 
 81              * @default null 
 82              * @example
 83              * $("#tip").omTooltip({maxHeight: 100}); 
 84              */
 85             maxHeight : null,
 86             /**
 87              * 最小高度。
 88              * @type Number 
 89              * @default null 
 90              * @example
 91              * $("#tip").omTooltip({minHeight: 100}); 
 92              */
 93             minHeight : null,
 94             /**
 95              * 鼠标移入目标区域之后弹出提示框的延迟时间(ms) ，同时也是隐藏延迟的事件。
 96              * @type Number 
 97              * @default 300 
 98              * @example
 99              * $("#tip").omTooltip({delay: 300}); 
100              */
101             delay : 300,
102             /**
103              * 是否显示anchor，默认不显示，如果trackMouse为true，则显示在左上角，如果不跟随，
104              * 则根据region属性确定，箭头的方向永远指向被提示的目标。
105              * @type Boolean 
106              * @default false 
107              * @example
108              * $("#tip").omTooltip({anchor: false}); 
109              */
110             anchor : false,
111             /**
112              * 提示框是否跟随鼠标移动,默认不跟随鼠标移动，设置为true则跟随鼠标移动，
113              * 比region优先级低。
114              * @type Boolean 
115              * @default false 
116              * @example
117              * $("#tip").omTooltip({trackMouse: false}); 
118              */
119             trackMouse : false,
120             /**
121              * 显示提示框的触发方式,可选值有mouseover、click 
122              * @type String 
123              * @default 'mouseover' 
124              * @example
125              * $("#tip").omTooltip({showOn: click}); 
126              */
127             showOn : "mouseover",
128             /**
129              * ajax方式显示内容的时候是否延迟加载页面，必须配置了url才有效 
130              * @type Boolean 
131              * @default false 
132              * @example
133              * $("#tip").omTooltip({url:'a.html',lazyLoad: true}); 
134              */
135             lazyLoad : false,
136             /**
137              * ajax加载内容的url地址,返回的内容格式dataType为html
138              * @type String 
139              * @default null 
140              * @example
141              * $("#tip").omTooltip({url:'a.html'}); 
142              */
143             url : null,
144             /**
145              * 提示框显示在指定位置的偏离度,提示框有两种基准位置，一是鼠标，二是目标区域,
146              *  第一个参数为top偏差值，第二个为left偏差值。
147              * @type Object 
148              * @default [5,5] 
149              * @example
150              * $("#tip").omTooltip({offset:[10,15]}); 
151              */
152             offset : [5,5],
153             /**
154              * anchor偏移度(px),
155              * 当设置anchor为top、button的时候，偏移度指相对于最左边的距离；
156              * 为left、right的时候偏移度指相对于最上的距离
157              * @type Number
158              * @default 0
159              * @example
160              * $("#tip").omTooltip({anchorOffset:10}); 
161              */
162             anchorOffset : 0,
163             /**
164              * 提示框显示的区域,设置提示框显示的区域，如果不设置，
165              * 则以鼠标的当前位置为基准点显示并跟随鼠标移动，
166              * 如果设置了则以目标为基准点显示,可选值有left、right、top、bottom，
167              * 比如选择了left，则提示框永远出现在目标的左边，不会进入目标。
168              * @type String
169              * @default null
170              * @example
171              * $("#tip").omTooltip({region:'bottom'}); 
172              */
173             region : null,
174             /**
175              * 提示组件显示的内容，可以是html和普通字符,优先级比contentEL属性高。
176              * @type String
177              * @default null
178              * @example
179              * $("#tip").omTooltip({html:'<span style="color:red;">我是omTooltip</span>'}); 
180              */
181             html : null,
182             /**
183              * jquery选择器，可以获取页面的dom元素作为提示内容，格式为#ID、.className、tagName等，优先级比html属性低。
184              * @type selector
185              * @default null
186              * @example
187              * $("#tip").omTooltip({contentEL:'#aa'}); 
188              */
189             contentEL : null
190             
191         },
192         _create:function(){
193             this.tipContent = $('<div class="tip-body"></div>');
194             this.tipAnchor = $('<div class="tip-anchor"></div>');
195             this.regionMap = {'right':'left','top':'bottom','left':'right','bottom':'top'};
196             this.tip = $('<div class="tip"></div>').append(this.tipContent);
197         },
198         /**
199          * 初始化方法，主要设置高宽，加载内容
200          */
201         _init:function(){
202             var self = this , options = self.options;
203             
204             this.tipContent.empty();
205             this.tip.find('div.tip-anchor').remove();
206             
207             if(options.anchor){
208                 self.tip.append(self.tipAnchor);
209             }
210             self.adjustWidth = 6 ; //内外边框
211             self.adjustHeight =  6 ;
212             if(options.width != 'auto'){
213                 this.tip.css('width',options.width-self.adjustWidth);
214             }else{
215                 this.tip.css('width','auto');
216             }
217             if(options.height != 'auto'){
218                 this.tip.css('height',options.height - self.adjustHeight);
219             }else{
220                 this.tip.css('height','auto');
221             }
222             if(options.minWidth) {
223                 this.tip.css('minWidth',options.minWidth - self.adjustWidth);
224             }
225             if(options.maxWidth){
226                 this.tip.css('maxWidth',options.maxWidth - self.adjustWidth);
227             }
228             if(options.maxHeight){
229                 this.tip.css('maxHeight',options.maxHeight - self.adjustHeight);
230             }
231             if(options.minHeight){
232                 this.tip.css('minHeight',options.minHeight - self.adjustHeight);
233             }
234             /**
235              * 内容加载的优先级为url、html、contentEl，前者会覆盖后者
236              */
237             if(options.url && !options.lazyLoad){ //如果是ajax方式并且不是懒加载模式下面，初始化组件的时候就请求数据
238                 self._ajaxLoad(options.url);
239             }else if(options.html){
240                 this.tip.find('.tip-body').append(options.html);
241             }else{
242             	var contentElClone = $(options.contentEL).clone();
243             	$(options.contentEL).remove();
244                 this.tip.find('.tip-body').append(contentElClone.show());
245             }
246             this._bindEvent();
247         },
248         /**
249          * ajax方法异步加载数据，返回的数据格式为html。
250          * @param url
251          */
252         _ajaxLoad:function(url){
253             var self = this;
254             $.ajax({
255                 url: url,
256                 method: 'POST',
257                 dataType: 'html',
258                 success: function(data, textStatus){
259                     self.tip.find('.tip-body').append(data);
260                 },
261                 error: function(XMLHttpRequest, textStatus, errorThrown){
262                     self.tip.find('.tip-body').append($.om.lang.omTooltip.FailedToLoad);
263                 }
264             });
265         },
266         /**
267          * 给提示框绑定事件，同时根据配置的delay参数做延迟显示，延迟隐藏处理
268          * 而且鼠标离开了目标区域，但是还停留在提示框内的时候不能隐藏
269          */
270         _bindEvent : function(){
271             var self = this , options = self.options , element = self.element;
272             if(options.showOn == 'mouseover'){ //mouseover的时候显示
273                 element.bind('mouseover.tooltip',function(e){
274                     if(self.showTime)clearTimeout(self.showTime); 
275                     self.showTime = setTimeout(function(){
276                         self.show(e);
277                     },options.delay);
278                 });
279             }else if(options.showOn == 'click'){ //click的时候显示
280                 self.showTime = element.bind('click.tooltip',function(e){
281                     setTimeout(function(){
282                         self.show(e);
283                     },options.delay);
284                 });
285             }
286             if(options.trackMouse){
287                 element.bind('mousemove.tooltip',function(e){
288                     self._adjustPosition(e);
289                 });
290             }
291             element.bind('mouseleave.tooltip',function(){
292                 if(self.showTime)clearTimeout(self.showTime); 
293                 self.hideTime = setTimeout(function(){ self.hide(); }, options.delay);
294             });
295             self.tip.bind('mouseover.tooltip',function(){
296                 if(self.hideTime){clearTimeout(self.hideTime);}
297             }).bind('mouseleave.tooltip',function(){
298                 setTimeout(function(){ self.hide(); }, options.delay);
299             });
300         },
301         /**
302          * 显示组件，组件只有在显示的时候才添加到dom树，
303          * 第二次显示已经存在dom树，只需要改变显示状态。
304          * @name omTooltip#show
305          * @function
306          * @example
307          * $('#mytip').omTooltip('show');
308          */
309         show : function(e){
310             var self = this ,options = self.options;
311             if($(document.body).find(self.tip).length <= 0){ //组件没有写入dom树的情况下
312                 if(options.url && options.lazyLoad){
313                     self._ajaxLoad(options.url);
314                 }
315                 self.tip.appendTo(document.body).fadeIn();
316                 self._adjustPosition(e);
317             }else{
318                 self._adjustPosition(e);
319                 self.tip.fadeIn();
320             }
321         },
322         /**
323          * 调整提示框的位置，如果传入了event且trackMouse为true，则会根据event计算出鼠标的位置
324          * 作为提示框的基础位置，再根据设置的offset等计算最终的提示框位置
325          * 如果没有传入event或者trackMouse为false，则提示框将以目标位置为基础位置，再根据offset
326          * region等参数配置情况计算最终位置。
327          * 这里会根据浏览器的宽度对显示做调整，如果region配置了right，二浏览器右边已经没有足够的空间
328          * 容纳提示框，则会将提示框显示到左边，region配置了bottom的时候同理。
329          * @param event
330          */
331         _adjustPosition : function(event){
332             var self = this , options = self.options , 
333             element = self.element , top ,left,
334             offSet = $(element).offset();
335            if(event && !options.region){
336                 top = event.pageY + options.offset[0]; 
337                 left = event.pageX + options.offset[1]; 
338                 //anchor距离最顶部的距离，使用用户设置的偏移减去tip的高度和调节高度
339                 self.tip.find('.tip-anchor').css({'top':options.anchorOffset+3,'left':'-9px'}).addClass('tip-anchor-left');
340            }else{
341                var bottomWidth = parseInt($(element).css('borderBottomWidth').replace('px',''));
342                top = offSet.top +  $(element).height() + (isNaN(bottomWidth)?0:bottomWidth) +options.offset[0]; //1px作为调节距离
343                left = offSet.left +options.offset[1];
344                //-----------根据region配置显示固定位置--------------
345                if(options.region == 'right'){ //offset向右下偏 right作为默认配置
346                    left = offSet.left + options.offset[1] + $(element).width() + self.adjustWidth; //元素的offset向右移动$(element).widht()的距离
347                    top = top - self.element.height() + self.adjustHeight + options.offset[0]; 
348                    self.tip.find('.tip-anchor').css({'top':options.anchorOffset+3,'left':'-9px'}); //要减去自身的高度,去掉3px的radius
349                }else if(options.region == 'left'){ //offset向左上偏
350                    left = offSet.left - self.tip.width() - self.adjustWidth - options.offset[1] - 2; //靠左的话就要减去tip自身的宽度
351                    top = top - self.element.height() - self.adjustHeight - options.offset[0];
352                    self.tip.find('.tip-anchor').css({'top':options.anchorOffset+3,'left': self.tip.outerWidth()-2}); //将anchor向右移动tip的宽度
353                }else if(options.region == 'top'){ //offset向右上偏
354                    top = top - (self.element.height() + self.tip.height() + self.adjustHeight + 2) - options.offset[0];
355                    left = left + options.offset[1];
356                    self.tip.find('.tip-anchor').css({'top':self.tip.outerHeight()-3, 'left':options.anchorOffset+3}); //25+4
357                }else if(options.region == 'bottom'){ //offset向右下偏
358                    top = top + options.offset[0];
359                    left = left + options.offset[1];
360                    self.tip.find('.tip-anchor').css({'top': -9, 'left':options.anchorOffset + 3});
361                }
362                self.tip.find('.tip-anchor').addClass('tip-anchor-'+self.regionMap[options.region]);
363            }
364            if ((left + self.tip.width() - $(document).scrollLeft()) > document.documentElement.clientWidth){ //当右边距离过短的时候会将提示框调整到左边
365                left = left - self.tip.width() - 20;
366            }
367            if ((top + self.tip.height() - $(document).scrollTop()) > document.documentElement.clientHeight){ //当下边距离过短的时候会将提示框调整到上边
368                top = top - (self.element.height() + self.tip.height()) - 20;
369            }
370            self.tip.css({'top':top - $(document).scrollTop(),'left':left - $(document).scrollLeft()});
371         },
372         /**
373          * 隐藏组件。
374          * @name omTooltip#hide
375          * @function
376          * @example
377          * $('#mytip').omTooltip('hide');
378          */
379         hide : function(){
380             this.tip.hide();
381         },
382         destroy : function(){
383         	clearTimeout(this.showTime);
384         	clearTimeout(this.hideTime);
385         	this.element.unbind('.tooltip');
386         	this.tip.remove();
387         }
388     });
389     $.om.lang.omTooltip = {
390             FailedToLoad : '加载提示信息出错！'
391     };
392 })(jQuery);
