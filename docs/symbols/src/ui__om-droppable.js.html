 54 		/**

 61          */

 63 		/**

135 	_create: function() {
136 
137 		var o = this.options, accept = o.accept;
138 		this.isover = 0; this.isout = 1;
139 
140 		this.accept = $.isFunction(accept) ? accept : function(d) {
141 			return d.is(accept);
142 		};
143 
144 		//Store the droppable's proportions
145 		this.proportions = { width: this.element[0].offsetWidth, height: this.element[0].offsetHeight };
146 
147 		// Add the reference and positions to the manager
148 		$.om.ddmanager.droppables[o._scope] = $.om.ddmanager.droppables[o._scope] || [];
149 		$.om.ddmanager.droppables[o._scope].push(this);
150 
151 		this.element.addClass("om-droppable");
152

156      * 删除元素的放置功能。
157      * @name omDroppable#destroy

159      * @returns JQuery对象

161      * var $selector = $("#selector").omDroppable('destroy');

164 	destroy: function() {
165 		var drop = $.om.ddmanager.droppables[this.options._scope];
166 		for ( var i = 0; i < drop.length; i++ )
167 			if ( drop[i] == this )
168 				drop.splice(i, 1);
169 
170 		this.element
171 			.removeClass("om-droppable om-droppable-disabled")
172 			.removeData("omDroppable")
173 			.unbind(".droppable");
174 
175 		return this;

197 		//(draggable && this._trigger('onDeactivate', event, draggable.currentItem || draggable.element));

200 	_over: function(event) {
201 
202 		var draggable = $.om.ddmanager.current;
203 		if (!draggable || (draggable.currentItem || draggable.element)[0] == this.element[0]) return; // Bail if draggable and droppable are same element
204 
205 		if (this.accept.call(this.element[0],(draggable.currentItem || draggable.element))) {
206 			if(this.options.hoverClass) this.element.addClass(this.options.hoverClass);
207 			this._trigger('onDragOver', event, draggable.currentItem || draggable.element);
208 		}
209 
210 	},
211 
212 	_out: function(event) {
213 
214 		var draggable = $.om.ddmanager.current;
215 		if (!draggable || (draggable.currentItem || draggable.element)[0] == this.element[0]) return; // Bail if draggable and droppable are same element
216 
217 		if (this.accept.call(this.element[0],(draggable.currentItem || draggable.element))) {
218 			if(this.options.hoverClass) this.element.removeClass(this.options.hoverClass);
219 			this._trigger('onDragOut', event, draggable.currentItem || draggable.element);
220 		}
221 
222 	},
223 
224 	_drop: function(event,custom) {
225 
226 		var draggable = custom || $.om.ddmanager.current;
227 		if (!draggable || (draggable.currentItem || draggable.element)[0] == this.element[0]) return false; // Bail if draggable and droppable are same element
228 
229 		var childrenIntersection = false;
230 		this.element.find(":data(omDroppable)").not(".om-draggable-dragging").each(function() {
231 			var inst = $.data(this, 'omDroppable');
232 			if(
233 				inst.options.greedy
234 				&& !inst.options.disabled
235 				&& inst.options._scope == draggable.options._scope
236 				&& inst.accept.call(inst.element[0], (draggable.currentItem || draggable.element))
237 				&& $.om.intersect(draggable, $.extend(inst, { offset: inst.element.offset() }))
238 			) { childrenIntersection = true; return false; }
239 		});
240 		if(childrenIntersection) return false;
241 
242 		if(this.accept.call(this.element[0],(draggable.currentItem || draggable.element))) {
243 			if(this.options.activeClass) this.element.removeClass(this.options.activeClass);
244 			if(this.options.hoverClass) this.element.removeClass(this.options.hoverClass);
245 			this._trigger('onDrop', event, draggable.currentItem || draggable.element);
246 			return this.element;
247 		}
248 
249 		return false;
250 
251 	}

256 $.om.intersect = function(draggable, droppable) {
257 
258 	if (!droppable.offset) return false;
259 
260 	var x1 = (draggable.positionAbs || draggable.position.absolute).left, x2 = x1 + draggable.helperProportions.width,
261 		y1 = (draggable.positionAbs || draggable.position.absolute).top, y2 = y1 + draggable.helperProportions.height;
262 	var l = droppable.offset.left, r = l + droppable.proportions.width,
263 		t = droppable.offset.top, b = t + droppable.proportions.height;
264 	return (l < x1 + (draggable.helperProportions.width / 2) // Right Half
265 		&& x2 - (draggable.helperProportions.width / 2) < r // Left Half
266 		&& t < y1 + (draggable.helperProportions.height / 2) // Bottom Half
267 		&& y2 - (draggable.helperProportions.height / 2) < b ); // Top Half
268 };

270 /*
271 	This manager tracks offsets of draggables and droppables
272 */
273 $.om.ddmanager = {
274 	current: null,
275 	droppables: { 'default': [] },
276 	prepareOffsets: function(t, event) {
277 
278 		var m = $.om.ddmanager.droppables[t.options._scope] || [];
279 		var type = event ? event.type : null; // workaround for #2317
280 		var list = (t.currentItem || t.element).find(":data(omDroppable)").andSelf();
281 
282 		droppablesLoop: for (var i = 0; i < m.length; i++) {
283 
284 			if(m[i].options.disabled || (t && !m[i].accept.call(m[i].element[0],(t.currentItem || t.element)))) continue;	//No disabled and non-accepted
285 			for (var j=0; j < list.length; j++) { if(list[j] == m[i].element[0]) { m[i].proportions.height = 0; continue droppablesLoop; } }; //Filter out elements in the current dragged item
286 			m[i].visible = m[i].element.css("display") != "none"; if(!m[i].visible) continue; 									//If the element is not visible, continue
287 
288 			if(type == "mousedown") m[i]._activate.call(m[i], event); //Activate the droppable if used directly from draggables
289 
290 			m[i].offset = m[i].element.offset();
291 			m[i].proportions = { width: m[i].element[0].offsetWidth, height: m[i].element[0].offsetHeight };
292 
293 		}
294 
295 	},
296 	drop: function(draggable, event) {
297 
298 		var dropped = false;
299 		$.each($.om.ddmanager.droppables[draggable.options._scope] || [], function() {
300 
301 			if(!this.options) return;
302 			if (!this.options.disabled && this.visible && $.om.intersect(draggable, this))
303 				dropped = dropped || this._drop.call(this, event);
304 
305 			if (!this.options.disabled && this.visible && this.accept.call(this.element[0],(draggable.currentItem || draggable.element))) {
306 				this.isout = 1; this.isover = 0;
307 				this._deactivate.call(this, event);
308 			}
309 
310 		});
311 		return dropped;
312 
313 	},
314 	dragStart: function( draggable, event ) {
315 		//Listen for scrolling so that if the dragging causes scrolling the position of the droppables can be recalculated (see #5003)
316 		draggable.element.parentsUntil( "body" ).bind( "scroll.droppable", function() {
317 			if( !draggable.options.refreshPositions ) $.om.ddmanager.prepareOffsets( draggable, event );
318 		});
319 	},
320 	drag: function(draggable, event) {
321 
322 		//If you have a highly dynamic page, you might try this option. It renders positions every time you move the mouse.
323 		if(draggable.options.refreshPositions) $.om.ddmanager.prepareOffsets(draggable, event);
324 
325 		//Run through all droppables and check their positions based on specific tolerance options
326 		$.each($.om.ddmanager.droppables[draggable.options._scope] || [], function() {
327 
328 			if(this.options.disabled || this.greedyChild || !this.visible) return;
329 			var intersects = $.om.intersect(draggable, this);
330 
331 			var c = !intersects && this.isover == 1 ? 'isout' : (intersects && this.isover == 0 ? 'isover' : null);
332 			if(!c) return;
333 
334 			var parentInstance;
335 			if (this.options.greedy) {
336 				var parent = this.element.parents(':data(omDroppable):eq(0)');
337 				if (parent.length) {
338 					parentInstance = $.data(parent[0], 'omDroppable');
339 					parentInstance.greedyChild = (c == 'isover' ? 1 : 0);
340 				}
341 			}
342 
343 			// we just moved into a greedy child
344 			if (parentInstance && c == 'isover') {
345 				parentInstance['isover'] = 0;
346 				parentInstance['isout'] = 1;
347 				parentInstance._out.call(parentInstance, event);
348 			}
349 
350 			this[c] = 1; this[c == 'isout' ? 'isover' : 'isout'] = 0;
351 			this[c == "isover" ? "_over" : "_out"].call(this, event);
352 
353 			// we just moved out of a greedy child
354 			if (parentInstance && c == 'isout') {
355 				parentInstance['isout'] = 0;
356 				parentInstance['isover'] = 1;
357 				parentInstance._over.call(parentInstance, event);
358 			}
359 		});
360 
361 	},
362 	dragStop: function( draggable, event ) {
363 		draggable.element.parentsUntil( "body" ).unbind( "scroll.droppable" );
364 		//Call prepareOffsets one final time since IE does not fire return scroll events when overflow was caused by drag (see #5003)
365 		if( !draggable.options.refreshPositions ) $.om.ddmanager.prepareOffsets( draggable, event );
366 	}
