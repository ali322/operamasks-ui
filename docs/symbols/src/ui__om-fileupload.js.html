332             var uploader = document.getElementById(this.element.attr('id') + 'Uploader');
333             if (key == 'actionData') {
334                 var actionDataString = '';

357         	var self = this;
358             var element = this.element;
359             var settings = $.extend({}, this.options);
360             // 内置属性，不公布
361             settings.wmode = 'opaque'; // The wmode of the flash file
362             settings.expressInstall = null;
363             settings.displayData = 'percentage';
364             settings.folder = ''; // The path to the upload folder
365             settings.simUploadLimit = 1; // The number of simultaneous uploads allowed
366             settings.scriptAccess = 'sameDomain'; // Set to "always" to allow script access across domains
367             settings.queueID = false; // The optional ID of the queue container
368             settings.onInit = function(){}; // Function to run when omFileUpload is initialized
369             settings.onSelectOnce = function(){}; // Function to run once when files are added to the queue
370             settings.onClearQueue = function(){}; // Function to run when the queue is manually cleared
371             settings.id = this.element.attr('id');
372             
373             $(element).data('settings',settings);
374             var pagePath = location.pathname;
375             pagePath = pagePath.split('/');
376             pagePath.pop();
377             pagePath = pagePath.join('/') + '/';
378             var data = {};
379             data.omFileUploadID = settings.id;
380             
381             data.pagepath = pagePath;
382             if (settings.buttonImg) data.buttonImg = escape(settings.buttonImg);
383             data.buttonText = encodeURI($.om.lang._get(settings,"omFileUpload","buttonText"));
384             if (settings.rollover) data.rollover = true;
385             data.action = settings.action;
386             data.folder = escape(settings.folder);
387             
388             var actionDataString = '';
389             var cookieArray = document.cookie.split(';')
390             for (var i = 0; i < cookieArray.length; i++){
391                 actionDataString += '&' + cookieArray[i];
392             }
393             if (settings.actionData) {
394                 for (var name in settings.actionData) {
395                     actionDataString += '&' + name + '=' + settings.actionData[name];
396                 }
397             }
398             data.actionData = escape(encodeURI(actionDataString.substr(1)));
399             data.width = settings.width;
400             data.height = settings.height;
401             data.wmode = settings.wmode;
402             data.method = settings.method;
403             data.queueSizeLimit = settings.queueSizeLimit;
404             data.simUploadLimit = settings.simUploadLimit;
405             if (settings.hideButton) data.hideButton = true;
406             if (settings.fileDesc) data.fileDesc = settings.fileDesc;
407             if (settings.fileExt) data.fileExt = settings.fileExt;
408             if (settings.multi) data.multi = true;
409             if (settings.autoUpload) data.autoUpload = true;
410             if (settings.sizeLimit) data.sizeLimit = settings.sizeLimit;
411             if (settings.checkScript) data.checkScript = settings.checkScript;
412             if (settings.fileDataName) data.fileDataName = settings.fileDataName;
413             if (settings.queueID) data.queueID = settings.queueID;
414             if (settings.onInit() !== false) {
415                 element.css('display','none');
416                 element.after('<div id="' + element.attr('id') + 'Uploader"></div>');
417                 swfobject.embedSWF(settings.swf, settings.id + 'Uploader', settings.width, settings.height, '9.0.24', settings.expressInstall, data, {'quality':'high','wmode':settings.wmode,'allowScriptAccess':settings.scriptAccess},{},function(event) {
418                     if (typeof(settings.onSWFReady) == 'function' && event.success) settings.onSWFReady();
419                 });

427                 element.bind("omFileUploadOpen", settings.onOpen);
428             }
429             element.bind("omFileUploadSelect", {'action': settings.onSelect, 'queueID': settings.queueID}, function(event, ID, fileObj) {
430                 if (self._trigger("onSelect",event,ID,fileObj) !== false) {
431                     var byteSize = Math.round(fileObj.size / 1024 * 100) * .01;
432                     var suffix = 'KB';
433                     if (byteSize > 1000) {
434                         byteSize = Math.round(byteSize *.001 * 100) * .01;
435                         suffix = 'MB';
436                     }
437                     var sizeParts = byteSize.toString().split('.');
438                     if (sizeParts.length > 1) {

441                         byteSize = sizeParts[0];
442                     }

445                     } else {
446                         fileName = fileObj.name;
447                     }
448                     queue = '#' + $(this).attr('id') + 'Queue';
449                     if (event.data.queueID) {
450                         queue = '#' + event.data.queueID;
451                     }

499                     $("#" + $(this).attr('id') + ID).find('.om-fileupload-progress').hide();

514                     if (event.data.toDisplay == 'speed') displayData = ' - ' + data.speed + 'KB/s';
515                     if (event.data.toDisplay == null) displayData = ' ';
