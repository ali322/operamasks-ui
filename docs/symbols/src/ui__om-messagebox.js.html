 29 	                '</div>'+
 30 	                '<div class="om-messageBox-buttonpane om-widget-content om-corner-bottom om-helper-clearfix">'+

 34 	var _height = function(){
 35         // handle IE 6
 36         if ($.browser.msie && $.browser.version < 7) {
 37             var scrollHeight = Math.max(document.documentElement.scrollHeight, document.body.scrollHeight),
 38                 offsetHeight = Math.max(document.documentElement.offsetHeight, document.body.offsetHeight);
 39             return (scrollHeight < offsetHeight) ?  $(window).height() : scrollHeight;
 40         // handle "good" browsers
 41         } else {
 42             return $(document).height();
 43         }
 44 	};
 45 	var _width = function() {
 46         // handle IE
 47         if ( $.browser.msie ) {
 48             var scrollWidth = Math.max(document.documentElement.scrollWidth, document.body.scrollWidth),
 49                 offsetWidth = Math.max(document.documentElement.offsetWidth, document.body.offsetWidth);
 50             return (scrollWidth < offsetWidth) ? $(window).width() : scrollWidth;
 51         // handle "good" browsers
 52         } else {
 53             return $(document).width();
 54         }
 55     };
 56 	var close = function(messageBox, mask, handler, value){
 57 	    if (messageBox.hasClass('om-messageBox-waiting')) {
 58 	        return;
 59 	    }
 60 	    handler ? handler(value) : jQuery.noop();
 61 	    messageBox.remove();
 62 	    mask.remove();
 63 	};
 64     var _show = function(config){
 65         var onClose = config.onClose;
 66         var messageBox = $(tmpl).appendTo(document.body).css('z-index', 1500).position({
 67             of:window,
 68             collision: 'fit'
 69         }).omDraggable({
 70             containment: 'document',
 71             cursor:'move',
 72             handle: '.om-messageBox-titlebar'
 73         }).hide().keydown(function(event){
 74             if (event.keyCode && event.keyCode === $.om.keyCode.ESCAPE) {
 75                 close(messageBox, mask, null, false);
 76                 event.preventDefault();
 77             }
 78         });
 79         var mask = $('<div class="om-widget-overlay"/>').appendTo(document.body).show().css({height:_height(),width:_width()});
 80         var closeBut = messageBox.find('span.om-messageBox-title').html(config.title).next().hover(function(){
 81             $(this).addClass('om-state-hover');
 82         }, function(){
 83             $(this).removeClass('om-state-hover');
 84         }).focus(function(){
 85             $(this).addClass('om-state-focus');
 86         }).blur(function(){
 87             $(this).removeClass('om-state-focus');
 88         }).click(function(event){
 89             close(messageBox, mask, null, false);
 90             return false;
 91         }).bind('mousedown mouseup', function(){
 92             $(this).toggleClass('om-state-mousedown');
 93         });
 94         messageBox.find('div.om-messageBox-image').addClass('om-messageBox-image-' + config.type);
 95         var content = config.content;
 96         if (config.type == 'prompt') {
 97             content = content || '';
 98             content += '<br/><input id="om-messageBox-prompt-input" type="text"/>';
 99         }
100         messageBox.find('td.om-message-content-html').html(content);
101         var buttonSet = messageBox.find('div.om-messageBox-buttonset');
102         switch (config.type) {
103             case 'confirm':
104                 buttonSet.html('<button id="confirm">确定</button><button id="cancel">取消</button>');
105                 if($.fn.omButton){
106 	                buttonSet.find("button#confirm").omButton({
107 	                	width:60,
108 	                	onClick:function(event){
109 	                		close(messageBox, mask, onClose, true);
110 	                	}
111 	                });
112 	                buttonSet.find("button#cancel").omButton({
113 	                	width:60,
114 	                	onClick:function(event){
115 	                		close(messageBox, mask, onClose, false);
116 	                	}
117 	                });
118                 }
119                 break;
120             case 'prompt':
121                 buttonSet.html('<button id="confirm">确定</button><button id="cancel">取消</button>');
122                 if($.fn.omButton){
123 	                buttonSet.find("button#confirm").omButton({
124 	                	width:60,
125 	                	onClick:function(event){
126 	                        var returnValue = onClose ? onClose($('#om-messageBox-prompt-input').val()) : jQuery.noop();
127 	                        if (returnValue !== false) {
128 	                            messageBox.remove();
129 	                            mask.remove();
130 	                        }
131 	                	}
132 	                });
133 	                buttonSet.find("button#cancel").omButton({
134 	                	width:60,
135 	                	onClick:function(event){
136 	                		close(messageBox, mask, onClose, false);
137 	                	}
138 	                });
139                 }
140                 break;
141             case 'waiting':
142                 messageBox.addClass('om-messageBox-waiting');
143                 mask.addClass('om-messageBox-waiting');
144                 closeBut.hide(); //不显示关闭按钮
145                 buttonSet.parent().hide(); //不显示下面的按钮面板
146                 messageBox.find(">.om-messageBox-content").addClass("no-button om-corner-bottom");
147                 break;
148             default:
149                 buttonSet.html('<button id="confirm">确定</button>');
150 	            if($.fn.omButton){
151 		            buttonSet.find("button#confirm").omButton({
152 		            	width:60,
153 		            	onClick:function(event){
154 		            		close(messageBox, mask, onClose, true);
155 		            	}
156 		            });
157 	            }
158         }
159         var buts = $('button',buttonSet);
160         buts.width("100%");
161         messageBox.show();
162         var okBut = buts.first()[0];
163         okBut ? okBut.focus() : messageBox.focus();
164     };
165      /**
166       * @name omMessageBox
167       * @class
168       * omMessageBox用于提供提示信息的弹出窗口，类似于JavaScript中使用alert()、confirm()、prompt()函数时出现的那种提示信息的弹出窗口。<br/><br/>
169       * <br/>
170       * <h2>有以下特点：</h2><br/>
171       * <ul>
172       *     <li>有较好的浏览器兼容性</li>
173       *     <li>可以定义标题、内容，并且标题和内容可以使用html代码</li>
174       *     <li>标题栏有关闭按钮，也可以按Esc键关闭</li>
175       *     <li>支持丰富的提示（图标不同）</li>
176       *     <li>可以监听关闭事件</li>
177       * </ul>
178       * <br/>
179       * <h2>提供了以下工具方法：</h2><br/>
180       * <ul>
181       *     <li>
182       *         <b>$.omMessageBox.alert(config)</b><br/>
183       *         弹出一个Alert提示，仅有一个“确定”按钮。其中config有以下配置项：<br/>
184       *         <ul style="margin-left:40px">
185       *             <li>type：alert提示的类型，类型不同时弹出窗口左边的图标会不同。String类型，可选的值有'alert'、'success'、'error'、'question'、'warning'。默认值为'alert'。</li>
186       *             <li>title：弹出窗口的标题文字，String类型，可以使用普通字符串，也可以使用html代码。默认值为'提示'。</li>
187       *             <li>content：弹出窗口的提示内容，String类型，可以使用普通字符串，也可以使用html代码。无默认值。</li>
188       *             <li>onClose：弹出窗口关闭时的回调函数，Function类型，点击"确定"按钮来关闭弹出窗口时，Function的参数value值为true，按ESC键关闭弹出窗口时，Function的参数value值为false。无默认值。</li>
189       *         </ul>
190       *         <br/>使用方式如下：<br/>
191       *         <pre>
192       *             $.omMessageBox.alert({
193       *                 type:'error',
194       *                 title:'失败',
195       *                 content:'不能删除<font color="red">admin</font>用户',
196       *                 onClose:function(value){
197       *                     alert('do something');
198       *                 }
199       *             });
200       *         </pre>
201       *     </li>
202       *     <li>
203       *         <b>$.omMessageBox.confirm(config)</b><br/>
204       *         弹出一个Confirm提示，有“确定”和“取消”按钮。其中config有以下配置项：<br/>
205       *         <ul style="margin-left:40px">
206       *             <li>title：弹出窗口的标题文字，String类型，可以使用普通字符串，也可以使用html代码。默认值为'确认'。</li>
207       *             <li>content：弹出窗口的提示内容，String类型，可以使用普通字符串，也可以使用html代码。无默认值。</li>
208       *             <li>onClose：弹出窗口关闭时的回调函数，Function类型，点击"确定"按钮来关闭弹出窗口时，Function的参数value值为true，点击“取消”按钮或按ESC键关闭弹出窗口时，Function的参数value值为false。无默认值。</li>
209       *         </ul>
210       *         <br/>使用方式如下：<br/>
211       *         <pre>
212       *             $.omMessageBox.confirm({
213       *                 title:'确认删除',
214       *                 content:'删除用户后，它所有的发帖和回帖将同时删除（不可恢复），你确定要这样做吗？',
215       *                 onClose:function(value){
216       *                     alert(value?'开始删除操作':'不删除了');
217       *                 }
218       *             });
219       *         </pre>
220       *     </li>
221       *     <li>
222       *         <b>$.omMessageBox.prompt(config)</b><br/>
223       *         弹出一个Prompt提示，有一个输入框和“确定”和“取消”按钮。其中config有以下配置项：
224       *         <ul style="margin-left:40px">
225       *             <li>title：弹出窗口的标题文字，String类型，可以使用普通字符串，也可以使用html代码。默认值为'请输入'。</li>
226       *             <li>content：弹出窗口的提示内容，String类型，可以使用普通字符串，也可以使用html代码。无默认值。</li>
227       *             <li>onClose：弹出窗口关闭时的回调函数，Function类型，点击"确定"按钮来关闭弹出窗口时，Function的参数value值为用户在输入框里输入的字符串（一定是字符串），点击“取消”按钮或按ESC键关闭弹出窗口时，Function的参数value值为false。无默认值。<b>注意：在此方法中返回false将会阻止弹出窗口关闭。</b></li>
228       *         </ul>
229       *         <br/>使用方式如下：<br/>
230       *         <pre>
231       *             $.omMessageBox.prompt({
232       *                 title:'商品数量',
233       *                 content:'请输入你要购买的商品的数量（你的余额最多只能购买12千克）：',
234       *                 onClose:function(value){
235       *                     if(value===false){ //按了取消或ESC
236       *                         alert('取消购买');
237       *                         return;
238       *                     }
239       *                     if(value==''){
240       *                         alert('数量不能为空');
241       *                         return false; //不关闭弹出窗口
242       *                     }
243       *                     if(value-0+'' !== value){
244       *                         alert('只能输入数字');
245       *                         return false;  //不关闭弹出窗口
246       *     `               }
247       *                     if(value<0 || value>12){
248       *                         alert('请输入0-12之间的数字（可带小数）');
249       *                         return false; //不关闭弹出窗口
250       *                     }else{
251       *                         alert('开始购买'+value+'千克商品');
252       *                     }
253       *                 }
254       *             });
255       *         </pre>
256       *     </li>
257       *     <li>
258       *         <b>$.omMessageBox.waiting(config | 'close')</b><br/>
259       *         弹出一个Prompt提示，有一个输入框和“确定”和“取消”按钮。该提示窗口没有关闭按钮，也不可以按ESC关闭。如果参数是'close'时表示关闭上次弹出的Waiting提示窗口。如果是config时表示要弹出一个Waiting提示窗口，其中config有以下配置项：
260       *         <ul style="margin-left:40px">
261       *             <li>title：弹出窗口的标题文字，String类型，可以使用普通字符串，也可以使用html代码。默认值为'请稍候'。</li>
262       *             <li>content：弹出窗口的提示内容，String类型，可以使用普通字符串，也可以使用html代码。无默认值。</li>
263       *         </ul>
264       *         <br/>使用方式如下：<br/>
265       *         <pre>
266       *             //弹出提示
267       *             $.omMessageBox.waiting({
268       *                 title:'请稍候',
269       *                 content:'服务器正在处理您的请求，请稍候...',
270       *             });
271       * 
272       *             //关闭提示
273       *             $.omMessageBox.waiting('close');
274       *         </pre>
275       *     </li>
276       * </ul>
277       */
278     $.omMessageBox = {
279         alert: function(config){
280             config = config || {};
281             config.title = config.title || '提示';
282             config.type = config.type || 'alert';
283             _show(config);
284         },
285         confirm: function(config){
286             config = config || {};
287             config.title = config.title || '确认';
288             config.type = 'confirm';
289             _show(config);
290         },
291         prompt: function(config){
292             config = config || {};
293             config.title = config.title || '请输入';
294             config.type = 'prompt';
295             _show(config);
296         },
297         waiting: function(config){
298             if (config === 'close') {
299                 $('.om-messageBox-waiting').remove();
300                 return;
301             }
302             config = config || {};
303             config.title = config.title || '请等待';
304             config.type = 'waiting';
305             _show(config);
306         }
307     };
308 }(jQuery));
