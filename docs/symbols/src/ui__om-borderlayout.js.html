 17     /** 
 18      * @name omBorderLayout
 19      * @class omBroderLayout是页面布局的基础组件.把页面拆分为north,south,west,center,east（上、下、左、中、右）5个区域，除了center是必须设置的之外其他的都是可选的。<br/>
 20      * <b>特点：</b><br/>
 21      * <ol>
 22      * 		<li>以omPanel作为子组件，五个区域(north,south,west,center,east)都用omPanel实现。即每个panel都支持omPanel中的属性设置。</li>
 23      * 		<li>在omPanel的基础上添加两个个属性：region表示所属区域，resizable表示区域是否可拖拉改变大小。</li>
 24      * 		<li>左边(west)和右边(east)的面板还可设置expandToTop或者expandToBottom属性分别拉伸面板至顶部或底部。</li>
 25      * 		<li>可设置每个区域之间的间隔大小。</li>
 26      * 		<li>可设置borderLayout自动适应父容器的大小。</li>
 27      * </ol>
 28      * <b>示例：</b><br/>
 29      * <pre>
 30      * <script type="text/javascript" >
 31      * $(document).ready(function() {
 32      *     $('#page').omBorderLayout({
 33      *     panels:[{ 
 34      *        id:"north-panel", 
 35      *        title:"This is north panel", 
 36      *        region:"north", 
 37      *        resizable:true, 
 38      *        collapsible:true 
 39      *    },{ 
 40      *        id:"center-panel", 
 41      *        title:"This is center panel", 
 42      *        region:"center" 
 43      *    },{ 
 44      *        id:"west-panel", 
 45      *        title:"This is west panel", 
 46      *        region:"west", 
 47      *        resizable:true, 
 48      *        collapsible:true, 
 49      *        width:200 
 50      *    },{ 
 51      *        id:"east-panel", 
 52      *        title:"This is east panel", 
 53      *        region:"east", 
 54      *        resizable:true, 
 55      *        collapsible:true, 
 56      *        width:100 
 57      *    }], 
 58      *    spacing:3 
 59      * });
 60      * </script>
 61      * 
 62      * <div id="page" style="width:800px;height:600px;">
 63      *	<div id="north-panel" />
 64      *	<div id="center-panel" />
 65      *	<div id="west-panel" />
 66      *	<div id="east-panel" />
 67 	 * </div>
 68 	 * </pre>

162 			if(!this.options.panels) return;
163 			// 设置region拖拉改变宽度的最小值

171 		// 获取区域的大小，如果区域被隐藏了则获取代理区域(regionProxy)的大小，如果代理区域也被隐藏则返回0

174 				$proxy = this._getRegionProxy(region),

182 		_resizeRegion : function(init) {
183 			var $centerRegion = this._getRegion("center"),
184 				$northRegion = this._getRegion("north"),
185 				$southRegion = this._getRegion("south"),
186 				$westRegion = this._getRegion("west"),
187 				$eastRegion = this._getRegion("east"),
188 
189 				$northProxy = this._getRegionProxy("north"),
190 				$southProxy = this._getRegionProxy("south"),
191 				$westProxy = this._getRegionProxy("west"),
192 				$eastProxy = this._getRegionProxy("east"),
193 				
194 				northHeight = this._getRegionSize("north").height;
195 				southHeight = this._getRegionSize("south").height;
196 				westWidth = this._getRegionSize("west").width;
197 				eastWidth = this._getRegionSize("east").width;
198 				centerWidth = this._getRegionSize("center").width;
199 				layoutWidth = this.element.width();
200 				layoutHeight = this.element.height();
201 				
202 				westOpt = this._getPanelOpts("west");
203 				eastOpt = this._getPanelOpts("east");
204 			
205 			$centerRegion.css({top:northHeight,left:westWidth});
206 			$centerRegion.find(">.om-panel-body").omPanel("resize",{
207 				height:layoutHeight - northHeight - southHeight
208 			});
209 			// 初始化后需要拖拽动态改变center panel的宽度
210 			if(!init){
211 				$centerRegion.find(">.om-panel-body").omPanel("resize",{
212 					width:layoutWidth - westWidth - eastWidth
213 				});
214 			}
215 
216 			var centerHeight = $centerRegion.outerHeight(true);
217 			if($northRegion){
218 				// 根据左右（west、east）两边的面板是否会拉伸至顶部来计算northPanel的宽度
219 				var northWidth = layoutWidth - (westOpt.expandToTop?westWidth:0) - (eastOpt.expandToTop?eastWidth:0);
220 				$northRegion.find(">.om-panel-body").omPanel("resize",{width:northWidth});
221 				$northRegion.css({left:westOpt.expandToTop?westWidth:0});
222 				if($northProxy){
223 						$northProxy.outerWidth(northWidth)
224 								   .css({left:westOpt.expandToTop?westWidth:0});
225 				}
226 			}
227 			if($southRegion){
228 				// 根据左右（west、east）两边的面板是否会拉伸至底部来计算northPanel的宽度
229 				var southWidth = layoutWidth - (westOpt.expandToBottom?westWidth:0) - (eastOpt.expandToBottom?eastWidth:0);
230 				$southRegion.find(">.om-panel-body").omPanel("resize",{width:southWidth});
231 				$southRegion.css({top:layoutHeight-$southRegion.outerHeight(true),left:westOpt.expandToBottom?westWidth:0});
232 				if($southProxy){
233 					$southProxy.outerWidth(southWidth)
234 							   .css({left:westOpt.expandToBottom?westWidth:0});
235 				}
236 			}
237 			if($westRegion){
238 				var westTop = westOpt.expandToTop?0:northHeight;
239 				var westHeight = centerHeight + (westOpt.expandToBottom?southHeight:0) + (westOpt.expandToTop?northHeight:0);
240 				$westRegion.css({top:westTop});
241 				$westRegion.find(">.om-panel-body").omPanel("resize",{height:westHeight});
242 				if($westProxy){
243 					$westProxy.css({top:westTop});
244 					$westProxy.outerHeight(westHeight);
245 				}
246 			}
247 			if($eastRegion){
248 				var eastTop = eastOpt.expandToTop?0:northHeight;
249 				var eastHeight = centerHeight + (eastOpt.expandToBottom?southHeight:0) + (eastOpt.expandToTop?northHeight:0);
250 				$eastRegion.css({top:eastTop});
251 				$eastRegion.find(">.om-panel-body").omPanel("resize",{height:eastHeight});
252 				if($eastProxy){
253 					$eastProxy.css({top:eastTop});
254 					$eastProxy.outerHeight(eastHeight);
255 				}
256 			}
257 			// 初始化的时候如果center或east或west的宽度没有设置则自动调整宽度为自适应
258 			if(init){
259 				var fitEastWidth = this._getPanelOpts("east") && !this._getPanelOpts("east").width;
260 				var fitWestWidth = this._getPanelOpts("west") && !this._getPanelOpts("west").width;
261 				var fitCenterWidth = !this._getPanelOpts("center").width;
262 				if(fitEastWidth || fitWestWidth || fitCenterWidth){
263 					if(!fitCenterWidth && fitEastWidth && fitWestWidth){
264 						eastWidth = westWidth = (layoutWidth-centerWidth)/2;
265 					} else if(fitCenterWidth && !fitEastWidth && fitWestWidth){
266 						centerWidth = westWidth = (layoutWidth-eastWidth)/2;
267 					} else if(fitCenterWidth && fitEastWidth && !fitWestWidth){
268 						centerWidth = eastWidth = (layoutWidth-westWidth)/2;
269 					} else if(fitCenterWidth && fitEastWidth && fitWestWidth){
270 						eastWidth = westWidth = centerWidth = layoutWidth/3;
271 					}
272 				}
273 				if(fitCenterWidth){
274 					$centerRegion.find(">.om-panel-body").omPanel("resize",{width:Math.floor(layoutWidth - westWidth - eastWidth)});
275 				}
276 				if(fitEastWidth){
277 					$eastRegion.find(">.om-panel-body").omPanel("resize",{width:Math.ceil(layoutWidth - westWidth - centerWidth) - this.options.spacing});
278 				}
279 				if(fitWestWidth){
280 					$westRegion.find(">.om-panel-body").omPanel("resize",{width:Math.ceil(layoutWidth - eastWidth - centerWidth) - this.options.spacing});
281 					$centerRegion.css({left:$westRegion.width() + this.options.spacing});
282 				}
283 			}

332 		// 构建布局框架
333 		_buildRegion : function() {

370 					}
371 				}

381 				
382

385 				
386 				// 初始化north和south的宽度

390

401 					} else if(panel.region == "east"){
402 						handles = "w";
403 						handleClass.width = spacing;
404 						handleClass.left = "-" + spacing;
405 					} else if(panel.region == "south"){
406 						handles = "n";
407 						handleClass.height = spacing;
408 						handleClass.top = "-" + spacing;
409 					} else if(panel.region == "north"){
410 						handles = "s";
411 						handleClass.height = spacing;
412 						handleClass.bottom = "-" + spacing;
413 					}
414 					$panelEl.parent().omResizable({

416 						helper : "om-borderlayout-resizable-helper-" + handles,

430 								maxWidth = $layout.width() - 2*_self._minWidth,
431 								maxHeight = $layout.height() - 2*_self._minHeight;
432 							if(region == "west"){
433 								maxWidth = $layout.width() - (_self._getRegionSize("east").width + _self._minWidth);
434 								ui.element.omResizable( "option", "maxWidth", maxWidth );
435 							} else if(region == "east"){
436 								maxWidth = $layout.width() - (_self._getRegionSize("west").width + _self._minWidth);
437 								ui.element.omResizable( "option", "maxWidth", maxWidth );
438 							} else if(region == "north"){
439 								maxHeight = $layout.height() - (_self._getRegionSize("south").height + _self._minHeight + _self.options.spacing);
440 								ui.element.omResizable( "option", "maxHeight", maxHeight );
441 							} else if(region == "south"){
442 								maxHeight = $layout.height() - (_self._getRegionSize("north").height + _self._minHeight + _self.options.spacing);
443 								ui.element.omResizable( "option", "maxHeight", maxHeight );
444 							}
445 							$('<div class="om-borderlayout-mask"></div>').css({
446 								width:$layout.width(),
447 								height:$layout.height()
448 							}).appendTo($layout);
449 							// 监听拖拽改变panel大小的事件
450 							_self.options.onBeforeDrag && _self._trigger("onBeforeDrag",null,ui.element);
451 						},
452 						minWidth : _self._minWidth,
453 						minHeight : _self._minHeight
454 						
455 					});
456 					$panelEl.parent().find(".om-resizable-handle").css(handleClass);
457 					margin = (panel.region == "south" ? spacing : 0) + " " +
458 							 (panel.region == "west" ? spacing : 0) + " " +
459 							 (panel.region == "north" ? spacing : 0) + " " +
460 							 (panel.region == "east" ? spacing : 0);
461 					
462 					// 如果隐藏收缩panel的按钮，则在panel之间的间隔条上显示收缩/展开trigger按钮
463 					if(showCollapsTrigger){
464 						var $collapsTrigger = $("<div class='om-borderlayout-collaps-trigger-"+panel.region+"'></div>");
465 						(function($panel){$collapsTrigger.click(function(){
466 							_self.collapseRegion($panel.attr("region"));

469 					}
470 				}
471

481 			}

499         /**
500          * 折叠某个区域的panel。
501          * @name omBorderLayout#collapseRegion
502          * @function

504          * @example
