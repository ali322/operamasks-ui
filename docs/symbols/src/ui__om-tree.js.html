 22      *     expanded:true, // 是否默认展开，非必须，默认值是false

400         // handle toggle event
401         // change the target to the treenode (li dom)
402         toggler: function(target) {
403             var self = this,
404                 options = self.options;
405             var nid = target.attr("id");
406             var node = self.findByNId(nid);
407             var hidden = target.hasClass(CLASSES.expandable);
408             
409             if ( hidden ) {
410                 var onBeforeExpand = options.onBeforeExpand;
411                 if(onBeforeExpand && false === self._trigger("onBeforeExpand",null,node)){
412                     return self;
413                 }
414             } else {
415                 var onBeforeCollapse = options.onBeforeCollapse;
416                 if(onBeforeCollapse && false === self._trigger("onBeforeCollapse",null,node)){
417                     return self;
418                 }
419             }
420             
421             // swap classes for hitarea
422             var hitarea = target.find( target.find(">.hitarea") );
423             self._swapClass(hitarea, CLASSES.collapsableHitarea, CLASSES.expandableHitarea);
424             self._swapClass(hitarea, CLASSES.lastCollapsableHitarea, CLASSES.lastExpandableHitarea);
425             
426             // swap classes for li
427             self._swapClass(target, CLASSES.collapsable, CLASSES.expandable);
428             self._swapClass(target, CLASSES.lastCollapsable, CLASSES.lastExpandable);
429             
430             // find child lists
431             target.find( ">ul" )
432                 .each(function(){
433                     if ( hidden ) {
434                         $(this).show();
435                         var onExpand = options.onExpand;
436                         onExpand && self._trigger("onExpand",null,node);
437                     } else {
438                         $(this).hide();
439                         var onCollapse = options.onCollapse;
440                         onCollapse && self._trigger("onCollapse",null,node);
441                     }
442                 });
443         },

445         _init: function() {
446             var self = this, options = self.options,
447                 target = self.element,
448                 source = options.dataSource;
449             target.empty();
450             if(source) {
451                 if(typeof source == 'string'){
452                     self._ajaxLoad(target, source);
453                 }else if(typeof source == 'object'){
454                 	if(options.simpleDataModel){
455                 		source = self._transformToNodes(source);
456                 	}
457                 	target.data("init_dataSource", source);
458                     self._appendNodes.apply(self, [target, source]);
459                     self.updateNode(target);
460                 }
461             }
462         },

465             var self = this,
466                 options = this.options,
467                 onBeforeLoad = options.onBeforeLoad,
468                 onSuccess = options.onSuccess,
469                 onError = options.onError;
470             onBeforeLoad && self._trigger("onBeforeLoad",null,self.findByNId(target.parent().attr("id")));
471             $.ajax({
472                 url: url,
473                 method: 'POST',
474                 dataType: 'json',
475                 cache: false,
476                 success: function(data){
477                 	if(options.simpleDataModel){
478                 		data = self._transformToNodes(data);
479                 	}
480                 	target.data("init_dataSource", data);
481                     self._appendNodes.apply(self, [target, data]);
482                     self.updateNode(target);
483                     onSuccess && self._trigger("onSuccess",null,data);
484                 },
485                 error: function(XMLHttpRequest, textStatus, errorThrown){
486                     onError && self._trigger("onError",null,XMLHttpRequest, textStatus, errorThrown);
487                 }
488             });

790             if(isSingle){

806                 return result;
