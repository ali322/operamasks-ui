 15 (function($) {
 16 	/**
 17       * @name omScrollbar
 18       * @class
 19       * 自定义滚动条，可用于取代浏览器自带的滚动条。<br/>
 20       * <br/>
 21       * <b>特点：</b><br/><br/>
 22       * <ul>
 23       *     <li>使用简单，它会自动管理滚动条的状态，包括显示与隐藏，滚动条的大小等。</li>
 24       * </ul>
 25       * <b>说明：</b><br/><br/>
 26       * <p>
 27       * 	滚动条的使用非常简单，比如页面上已经存在一个div,那么给这个div创建滚动条只需要写上如下代码:<br/>
 28       * 	$(".selector").omScrollbar();<br/>
 29       * </p>
 30       * <p>
 31       * 	有一个要特别注意的地方就是，如果把鼠标移到目标元素上边时出现了滚动条，然后鼠标在目标元素内操作致使目标元素的大小不足以出现滚动条，这时候滚动条是不会自行消失的，
 32       * 这是一个时机问题，只有当鼠标再次移到目标元素上时滚动条的状态才会改变修正。为了解决此问题，你可以监听目标元素内某个节点的事件，然后调用refresh方法即时刷新滚动条的状态。
 33       * </p>
 34       * @constructor
 35       * @description 构造函数
 36       * @param p 标准config对象:{}
 37       */
 38 	$.omWidget("om.omScrollbar" , {
 39 		options:/** @lends omScrollbar#*/{
 40 			/**
 41              * 滚动条的厚度，单位为像素。
 42              * @default 8
 43              * @type Number
 44              * @example
 45              * $('.selector').omScrollbar({thick:20});
 46              */
 47 			thick : 8
 48 		},
 49 		
 50 		_create : function(){
 51 			var ops = this.options;
 52 			this._vScrollbar = $("<div class='om-widget om-scrollbar om-corner-all'></div>").width(ops.thick).appendTo("body").hide();
 53 			this._vScrollbar.type = "v";
 54 			this._hScrollbar = $("<div class='om-widget om-scrollbar om-corner-all'></div>").height(ops.thick).appendTo("body").hide();
 55 			this._hScrollbar.type = "h";
 56 		},
 57 		
 58 		_init : function(){
 59 			this.element.css("overflow" , "hidden");
 60 			this._buildEvent();
 61 		},
 62 		
 63 		destroy : function(){
 64 			this._vScrollbar.remove();
 65 			this._hScrollbar.remove();
 66 		}, 
 67 		
 68 		_buildEvent : function(){
 69 			var self = this,
 70 				$elem = this.element,
 71 				bars = [this._vScrollbar , this._hScrollbar];
 72 				
 73 			$(bars).each(function(index , $bar){
 74 				$bar._hover = false;
 75 				$bar._enable = true;
 76 				
 77 				var type = $bar.type;
 78 				
 79 				var startPos = 0;
 80 				$bar.omDraggable({
 81 					axis : type==='v'? "y":"x",
 82 					containment: $elem,
 83 					onStart : function(ui , event){
 84 						startPos = type==='v'? $elem.scrollTop() : $elem.scrollLeft();
 85 					},
 86 					onDrag: function(ui, event){
 87 						var p = ui.position,
 88 							op = ui.originalPosition;
 89 						type == 'v'?
 90 							$elem.scrollTop( startPos + self._getInt( (p.top-op.top)*$elem.innerHeight()/$bar.outerHeight()) )
 91 							:$elem.scrollLeft( startPos + self._getInt( (p.left-op.left)*$elem.innerWidth()/$bar.outerWidth()) );
 92 					}
 93 				}).hover(function(){
 94 					$(bars).each(function(index , $bar){
 95 						$bar._hover = true;
 96 						clearTimeout($bar._timer);
 97 					});
 98 					$(this).addClass("scrollbar-state-hover");
 99 				},
100 				function(){
101 					$(bars).each(function(index , $bar){
102 						$bar._hover = false;
103 						self._setTimer($bar.type);
104 					});
105 					$(this).removeClass("scrollbar-state-hover");
106 				});
107 			});
108 			
109 			//给垂直滚动条绑定鼠标滑轮滚动事件
110 			var eventName = $.browser.mozilla? "DOMMouseScroll" : "mousewheel";
111 			$elem.bind(eventName , function(e){self._mousewheelListener.call(self,e);});
112 				$(bars).each(function(index , $bar){
113 					$bar.bind(eventName , function(e){self._mousewheelListener.call(self,e);});
114 				});
115 
116 			$elem.hover(function(){
117 				$(bars).each(function(index , $bar){
118 					$bar._hover = true;
119 					clearTimeout($bar._timer);
120 					self._resize($bar);
121 					if($bar._enable){
122 						$bar.fadeIn("fast");						
123 					}
124 				});
125 			},function(){
126 				$(bars).each(function(index , $bar){
127 					$bar._hover = false;
128 					self._setTimer($bar.type);
129 				});
130 			});
131 		},
132 
133 		_mousewheelListener : function(event){
134 			var self = this;
135 			if(!self._vScrollbar._enable){
136 				return ;
137 			}
138 			
139 			//在IE中，滚轮向后，wheelDelta = -120(倍数)
140 			//在FF中，滚轮向后，event.detail = 3(倍数)
141 			//进行处理，统一为 滚轮向后，120(倍数)，滚动向前,-120(倍数)
142 			var delta = 0;
143 			if (event.wheelDelta) {
144 	            delta = -event.wheelDelta;
145 	        } else {
146 	            delta = event.detail * 40;
147 	        }
148 	        
149 	        //每一份(120)表示5个象素点
150 	        //滚动条此次滚动了多少个像素
151 	        var differ = delta / 120 * 10, 
152 	        	$bar = self._vScrollbar,
153 	        	$elem = self.element,
154 	        	elemTop = $elem.offset().top + self._getInt($elem.css("border-top-width")) + self._getInt($elem.css("padding-top")) + self._getInt($elem.css("padding-top-width")),
155 	        	barCurTop = $bar.position().top - elemTop,//滚动条相对于目标元素的内容区域的top
156 	        	contentHeight = $elem.height(),
157 	        	distance = contentHeight - $bar.outerHeight();//滚动条最大滚动距离
158 	        	
159 	        
160 	        if(differ + barCurTop < 0){
161 	        	$bar.css("top" , elemTop);
162 	        	$elem.scrollTop(0);
163 	        }else if(differ + barCurTop > distance){
164 	        	$bar.css("top" , elemTop + distance);
165 	        	$elem.scrollTop( $elem[0].scrollHeight - $elem.height());
166 	        }else{
167 	        	$bar.css("top" , "+="+differ );
168 	        	$elem.scrollTop( $elem.scrollTop() + parseInt(differ * $elem.innerHeight()/$bar.outerHeight()) );
169 	        }
170 	        event.preventDefault();
171 		},
172 		
173 		_getScrollbar : function(type){
174 			return 'h' === type? this._hScrollbar : this._vScrollbar;
175 		},
176 		
177 		_setTimer : function(type){
178 			var self = this;
179 			clearTimeout(this._getScrollbar(type)._timer);
180 			(function(type){
181 				setTimeout(function(){
182 					var $bar = self._getScrollbar(type);
183 					if(!$bar._hover){
184 						$bar.fadeOut("fast");					
185 					}
186 				} , 200);
187 			})(type);
188 		},
189 		/**
190 		 * 刷新滚动条。当进入目标元素时，如果目标元素内容区域过长，会出现滚动条，但如果此时用鼠标进行操作使得目标元素内容过小而不应该出现滚动条，
191 		 * 这时候滚动条是不会自己消失的，要调用此方法进行刷新。
192 		 * @name omScrollbar#refresh
193 		 * @function
194 		 */
195 		refresh : function(){
196 			var bars = [this._vScrollbar , this._hScrollbar],
197 				self = this;
198 			$(bars).each(function(index , $bar){
199 				self._resize($bar);
200 			});
201 		},
202 		
203 		_resize : function($bar){
204 			var $elem = this.element,
205 				type = $bar.type,
206 				size = 0,
207 				offset = $elem.offset(),
208 				w = $elem.width(),
209 				h = $elem.height(),
210 				ow = $elem.outerWidth(),
211 				oh = $elem.outerHeight(),
212 				bl = this._getInt($elem.css("border-left-width")),
213 				br = this._getInt($elem.css("border-right-width")),
214 				bt = this._getInt($elem.css("border-top-width")),
215 				bb = this._getInt($elem.css("border-bottom-width")),
216 				pl = this._getInt($elem.css("padding-left")),
217 				pt = this._getInt($elem.css("padding-top"));
218 				
219 			if('v' === type){
220 				size = $elem[0].scrollHeight;
221 				if(size > h){
222 					$bar._enable = true;
223 					$bar.outerHeight( this._getInt(h*$elem.innerHeight()/size) );
224 					
225 					$bar.css({
226 						"left" : offset.left + ow - $bar.outerWidth() - br,
227 						"top" : offset.top + bt + pt  + this._getInt($elem.scrollTop() * h/ size)
228 					});
229 				}else{
230 					$bar._enable = false;
231 				}
232 			}else if('h' === type){
233 				size = $elem[0].scrollWidth;
234 				if(size > $elem.innerWidth()){
235 					$bar._enable = true;
236 					$bar.outerWidth( this._getInt((pl+w)*w/size) );
237 					$bar.css({
238 						"left" : offset.left + bl + pl + this._getInt($elem.scrollLeft()*w/ size),
239 						"top" : offset.top + oh - $bar.outerHeight() - bb
240 					});
241 				}else{
242 					$bar._enable = false;
243 				}
244 			}
245 			$bar._enable? $bar.fadeIn("fast") : $bar.fadeOut("fast");
246 		},
247 		
248 		_getInt : function(number){
249 			return parseInt(number) || 0;
250 		}
251 	});
252 })(jQuery);
